134 void
135 CoordinateArraySequence::add(size_t i, const Coordinate& coord,
136  bool allowRepeated)
137 {
138  // don't add duplicate coordinates
139  if (! allowRepeated) {
140  size_t sz = size();
141  if (sz > 0) {
142  if (i > 0) {
143  const Coordinate& prev = getAt(i - 1);
144  if (prev.equals2D(coord)) return;
145  }
146  if (i < sz) {
147  const Coordinate& next = getAt(i);
148  if (next.equals2D(coord)) return;
149  }
150  }
151  }
152 
153  vect->insert(vect->begin()+i, coord);
154 }
