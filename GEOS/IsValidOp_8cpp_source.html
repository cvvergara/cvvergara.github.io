64 const Coordinate *
65 IsValidOp::findPtNotNode(const CoordinateSequence *testCoords,
66  const LinearRing *searchRing, GeometryGraph *graph)
67 {
68  // find edge corresponding to searchRing.
69  Edge *searchEdge=graph->findEdge(searchRing);
70  // find a point in the testCoords which is not a node of the searchRing
71  EdgeIntersectionList &eiList=searchEdge->getEdgeIntersectionList();
72  // somewhat inefficient - is there a better way? (Use a node map, for instance?)
73  unsigned int npts = static_cast<int>(testCoords->getSize());
74  for(unsigned int i=0; i<npts; ++i)
75  {
76  const Coordinate& pt=testCoords->getAt(i);
77  if (!eiList.isIntersection(pt)) {
78  return &pt;
79  }
80  }
81  return nullptr;
82 }

124 void
125 IsValidOp::checkValid(const Geometry *g)
126 {
127  assert( validErr == nullptr );
128 
129  if (nullptr == g)
130  return;
131 
132  // empty geometries are always valid!
133  if (g->isEmpty()) return;
134 
135  if ( const Point* x = dynamic_cast<const Point*>(g) )
136  checkValid(x);
137  // LineString also handles LinearRings, so we check LinearRing first
138  else if ( const LinearRing* x = dynamic_cast<const LinearRing*>(g) )
139  checkValid(x);
140  else if ( const LineString* x = dynamic_cast<const LineString*>(g) )
141  checkValid(x);
142  else if ( const Polygon* x = dynamic_cast<const Polygon*>(g) )
143  checkValid(x);
144  else if ( const MultiPolygon* x = dynamic_cast<const MultiPolygon*>(g) )
145  checkValid(x);
146  else if ( const GeometryCollection* x =
147  dynamic_cast<const GeometryCollection*>(g) )
148  {
149  checkValid(x);
150  }
151  else throw util::UnsupportedOperationException();
152 }

504 /*private*/
505 void
506 IsValidOp::checkShellNotNested(const LinearRing *shell, const Polygon *p,
507  GeometryGraph *graph)
508 {
509  const CoordinateSequence *shellPts=shell->getCoordinatesRO();
510 
511  // test if shell is inside polygon shell
512  assert(dynamic_cast<const LinearRing*>(
513  p->getExteriorRing()));
514  const LinearRing *polyShell=static_cast<const LinearRing*>(
515  p->getExteriorRing());
516  const CoordinateSequence *polyPts=polyShell->getCoordinatesRO();
517  const Coordinate *shellPt=findPtNotNode(shellPts,polyShell,graph);
518 
519  // if no point could be found, we can assume that the shell
520  // is outside the polygon
521  if (shellPt==nullptr) return;
522 
523  bool insidePolyShell=CGAlgorithms::isPointInRing(*shellPt, polyPts);
524  if (!insidePolyShell) return;
525 
526  // if no holes, this is an error!
527  int nholes = static_cast<int>(p->getNumInteriorRing());
528  if (nholes<=0) {
529  validErr=new TopologyValidationError(
530  TopologyValidationError::eNestedShells,
531  *shellPt);
532  return;
533  }
534 
542  const Coordinate *badNestedPt=nullptr;
543  for(int i=0; i<nholes; ++i) {
544  assert(dynamic_cast<const LinearRing*>(
545  p->getInteriorRingN(i)));
546  const LinearRing *hole=static_cast<const LinearRing*>(
547  p->getInteriorRingN(i));
548  badNestedPt = checkShellInsideHole(shell, hole, graph);
549  if (badNestedPt==nullptr) return;
550  }
551  validErr=new TopologyValidationError(
552  TopologyValidationError::eNestedShells, *badNestedPt
553  );
554 }

556 /*private*/
557 const Coordinate *
558 IsValidOp::checkShellInsideHole(const LinearRing *shell,
559  const LinearRing *hole,
560  GeometryGraph *graph)
561 {
562  const CoordinateSequence *shellPts=shell->getCoordinatesRO();
563  const CoordinateSequence *holePts=hole->getCoordinatesRO();
564 
565  // TODO: improve performance of this - by sorting pointlists
566  // for instance?
567  const Coordinate *shellPt=findPtNotNode(shellPts, hole, graph);
568 
569  // if point is on shell but not hole, check that the shell is
570  // inside the hole
571  if (shellPt) {
572  bool insideHole=CGAlgorithms::isPointInRing(*shellPt, holePts);
573  if (!insideHole) return shellPt;
574  }
575 
576  const Coordinate *holePt=findPtNotNode(holePts, shell, graph);
577 
578  // if point is on hole but not shell, check that the hole is
579  // outside the shell
580  if (holePt) {
581  bool insideShell=CGAlgorithms::isPointInRing(*holePt, shellPts);
582  if (insideShell) return holePt;
583  return nullptr;
584  }
585  assert(0); // points in shell and hole appear to be equal
586  return nullptr;
587 }
