90 {

92  double deltaY = env.getHeight();
93  double offset = 0.0;
94  if (deltaX > deltaY) {

97  offset = deltaY * 10.0;

100  frameVertex[0] = Vertex((env.getMaxX() + env.getMinX()) / 2.0, env

171 QuadEdge*
172 QuadEdgeSubdivision::locateFromEdge(const Vertex &v,
173  const QuadEdge &startEdge) const
174 {
175  ::geos::ignore_unused_variable_warning(startEdge);
176 
177  int iter = 0;
178  int maxIter = static_cast<int>(quadEdges.size());
179 
180  QuadEdge *e = startingEdges[0];
181 
182  for (;;)
183  {
184  ++iter;
194  if (iter > maxIter) {
195  throw LocateFailureException("");
196  }
197 
198  if ((v.equals(e->orig())) || (v.equals(e->dest()))) {
199  break;
200  } else if (v.rightOf(*e)) {
201  e = &e->sym();
202  } else if (!v.rightOf(e->oNext())) {
203  e = &e->oNext();
204  } else if (!v.rightOf(e->dPrev())) {
205  e = &e->dPrev();
206  } else {
207  // on edge or in triangle containing edge
208  break;
209  }
210  }
211  return e;
212 }

214 QuadEdge*
215 QuadEdgeSubdivision::locate(const Coordinate &p0, const Coordinate &p1)
216 {
217  // find an edge containing one of the points
218  QuadEdge *e = locator->locate(Vertex(p0));
219  if (e == nullptr)
220  return nullptr;
221 
222  // normalize so that p0 is origin of base edge
223  QuadEdge *base = e;
224  if (e->dest().getCoordinate().equals2D(p0))
225  base = &e->sym();
226  // check all edges around origin of base edge
227  QuadEdge *locEdge = base;
228  do {
229  if (locEdge->dest().getCoordinate().equals2D(p1))
230  return locEdge;
231  locEdge = &locEdge->oNext();
232  } while (locEdge != base);

236 QuadEdge&
237 QuadEdgeSubdivision::insertSite(const Vertex &v)
238 {
239  QuadEdge *e = locate(v);
240 
241  if ((v.equals(e->orig(), tolerance)) || (v.equals(e->dest(), tolerance))) {
242  return *e; // point already in subdivision.
243  }
244 
245  // Connect the new point to the vertices of the containing
246  // triangle (or quadrilateral, if the new point fell on an
247  // existing edge.)
248  QuadEdge *base = &makeEdge(e->orig(), v);
249  QuadEdge::splice(*base, *e);
250  QuadEdge *startEdge = base;
251  do {
252  base = &connect(*e, base->sym());
253  e = &base->oPrev();
254  } while (&e->lNext() != startEdge);
255 
256  return *startEdge;
257 }

343 QuadEdge**
344 QuadEdgeSubdivision::fetchTriangleToVisit(QuadEdge *edge,
345  QuadEdgeStack &edgeStack, bool includeFrame, QuadEdgeSet &visitedEdges)
346 {
347  QuadEdge *curr = edge;
348  int edgeCount = 0;
349  bool isFrame = false;
350  do
351  {
352  triEdges[edgeCount] = curr;
353 
354  if (isFrameEdge(*curr))
355  isFrame = true;
356 
357  // push sym edges to visit next
358  QuadEdge *sym = &curr->sym();
359  if (visitedEdges.find(sym) == visitedEdges.end())
360  edgeStack.push(sym);
361 
362  // mark this edge as visited
363  visitedEdges.insert(curr);
364 
365  edgeCount++;
366  curr = &curr->lNext();
367 
368  } while (curr != edge);
369 
370  if (isFrame && !includeFrame)
371  return nullptr;
372  return triEdges;
373 }
