87 SIRtreePointInRing::testLineSegment(const Coordinate& p,LineSegment *seg)
88 {
89  double xInt; // x intersection of segment with ray
90  double x1; // translated coordinates
91  double y1;

93  double y2;
94 
95  /*
96  * Test if segment crosses ray from test point in positive x direction.
97  */
98  Coordinate& p1=seg->p0;
99  Coordinate& p2=seg->p1;
100  x1=p1.x-p.x;
101  y1=p1.y-p.y;
102  x2=p2.x-p.x;
103  y2=p2.y-p.y;
104 
105  if (((y1>0) && (y2<=0)) ||
106  ((y2>0) && (y1<=0))) {
107  /*
108  * segment straddles x axis,so compute intersection.
109  */
110  xInt=RobustDeterminant::signOfDet2x2(x1,y1,x2,y2)/(y2-y1);
111  //xsave=xInt;
112  /*
113  * crosses ray if strictly positive intersection.
114  */
115  if (0.0<xInt) {
116  crossings++;
