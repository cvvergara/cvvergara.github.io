48 inline bool
49 check_valid(const geos::geom::Geometry& g, const std::string& label, bool doThrow=false, bool validOnly=false)
50 {
51  using namespace geos;
52 
53  if ( dynamic_cast<const geos::geom::Lineal*>(&g) ) {
54  if ( ! validOnly ) {
55  operation::IsSimpleOp sop(g, algorithm::BoundaryNodeRule::getBoundaryEndPoint());
56  if ( ! sop.isSimple() )
57  {
58  if ( doThrow ) {
59  throw geos::util::TopologyException(
60  label + " is not simple");
61  }
62  return false;
63  }
64  }
65  } else {
66  operation::valid::IsValidOp ivo(&g);
67  if ( ! ivo.isValid() )
68  {
69  using operation::valid::TopologyValidationError;
70  TopologyValidationError* err = ivo.getValidationError();
71 #ifdef GEOS_DEBUG_CASCADED_UNION
72  std::cerr << label << " is INVALID: "
73  << err->toString()
74  << " (" << std::setprecision(20)
75  << err->getCoordinate() << ")"
76  << std::endl
77 #ifdef GEOS_DEBUG_CASCADED_UNION_PRINT_INVALID
78  << "<a>" << std::endl
79  << g.toString() << std::endl
80  << "</a>" << std::endl
81 #endif
82  ;
83 #endif // GEOS_DEBUG_CASCADED_UNION
84  if ( doThrow ) {
85  throw geos::util::TopologyException(
86  label + " is invalid: " + err->toString(),
87  err->getCoordinate());
88  }
89  return false;
90  }
91  }
92  return true;
93 }

170 geom::Geometry* CascadedPolygonUnion::binaryUnion(GeometryListHolder* geoms,
171  std::size_t start, std::size_t end)
172 {
173  if (end - start <= 1) {
174  return unionSafe(geoms->getGeometry(start), nullptr);
175  }
176  else if (end - start == 2) {
177  return unionSafe(geoms->getGeometry(start), geoms->getGeometry(start + 1));
178  }
179  else {
180  // recurse on both halves of the list
181  std::size_t mid = (end + start) / 2;
182  std::unique_ptr<geom::Geometry> g0 (binaryUnion(geoms, start, mid));
183  std::unique_ptr<geom::Geometry> g1 (binaryUnion(geoms, mid, end));
184  return unionSafe(g0.get(), g1.get());
185  }

189 CascadedPolygonUnion::reduceToGeometries(index::strtree::ItemsList* geomTree)
190 {
191  std::unique_ptr<GeometryListHolder> geoms (new GeometryListHolder());
192 
193  typedef index::strtree::ItemsList::iterator iterator_type;
194  iterator_type end = geomTree->end();
195  for (iterator_type i = geomTree->begin(); i != end; ++i) {
196  if ((*i).get_type() == index::strtree::ItemsListItem::item_is_list) {
197  std::unique_ptr<geom::Geometry> geom (unionTree((*i).get_itemslist()));
198  geoms->push_back_owned(geom.get());
199  geom.release();
200  }
201  else if ((*i).get_type() == index::strtree::ItemsListItem::item_is_geometry) {
202  geoms->push_back(reinterpret_cast<geom::Geometry*>((*i).get_geometry()));
203  }
204  else {
205  assert(!static_cast<bool>("should never be reached"));
206  }
207  }
208 
209  return geoms.release();

226 geom::Geometry*
227 CascadedPolygonUnion::unionOptimized(geom::Geometry* g0, geom::Geometry* g1)
228 {
229  geom::Envelope const* g0Env = g0->getEnvelopeInternal();
230  geom::Envelope const* g1Env = g1->getEnvelopeInternal();
231 
232  if (!g0Env->intersects(g1Env))
233  return geom::util::GeometryCombiner::combine(g0, g1);
234

236  return unionActual(g0, g1);

240  return unionUsingEnvelopeIntersection(g0, g1, commonEnv);
