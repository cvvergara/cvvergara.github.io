169 /*private*/
170 bool
171 OverlayResultValidator::testValid(OverlayOp::OpCode overlayOp,
172  const Coordinate& pt)
173 {
174  std::vector<geom::Location::Value> location(3);
175 
176  location[0] = fpl0.getLocation(pt);
177  location[1] = fpl1.getLocation(pt);
178  location[2] = fplres.getLocation(pt);
179 
180 #if GEOS_DEBUG
181  cerr << setprecision(10) << "Point " << pt << endl
182  << "Loc0: " << location[0] << endl
183  << "Loc1: " << location[1] << endl
184  << "Locr: " << location[2] << endl;
185 #endif
186 
187  /*
188  * If any location is on the Boundary, can't deduce anything,
189  * so just return true
190  */
191  if ( find(location.begin(), location.end(), Location::BOUNDARY) != location.end() )
192  {
193 #if GEOS_DEBUG
194  cerr << "OverlayResultValidator: testpoint " << pt << " is on the boundary, blindly returning a positive answer (is valid)" << endl;
195 #endif
196  return true;
197  }
198 
199  return isValidResult(overlayOp, location);
200 }

204 OverlayResultValidator::isValidResult(OverlayOp::OpCode overlayOp,
205  std::vector<geom::Location::Value>& location)
206 {
207  bool expectedInterior = OverlayOp::isResultOfOp(location[0],
208  location[1], overlayOp);
209 
210  bool resultInInterior = (location[2] == Location::INTERIOR);
211 
212  bool isValid = ! (expectedInterior ^ resultInInterior);
213 
214  return isValid;
