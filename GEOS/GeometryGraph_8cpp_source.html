73 /*
74  * This method implements the Boundary Determination Rule
75  * for determining whether
76  * a component (node or edge) that appears multiple times in elements
77  * of a MultiGeometry is in the boundary or the interior of the Geometry
78  *
79  * The SFS uses the "Mod-2 Rule", which this function implements
80  *
81  * An alternative (and possibly more intuitive) rule would be
82  * the "At Most One Rule":
83  * isInBoundary = (componentCount == 1)
84  */
85 bool
86 GeometryGraph::isInBoundary(int boundaryCount)
87 {
88  // the "Mod-2 Rule"
89  return boundaryCount%2==1;
90 }

229 /*
230  * The left and right topological location arguments assume that the ring
231  * is oriented CW.
232  * If the ring is in the opposite orientation,
233  * the left and right locations must be interchanged.
234  */
235 void
236 GeometryGraph::addPolygonRing(const LinearRing *lr, int cwLeft, int cwRight)
237  // throw IllegalArgumentException (see below)
238 {
239  // skip empty component (see bug #234)
240  if ( lr->isEmpty() ) return;
241 
242  const CoordinateSequence *lrcl = lr->getCoordinatesRO();
243 
244  CoordinateSequence* coord=CoordinateSequence::removeRepeatedPoints(lrcl);
245  if (coord->getSize()<4) {
246  hasTooFewPointsVar=true;
247  invalidPoint=coord->getAt(0); // its now a Coordinate
248  delete coord;
249  return;
250  }
251  int left=cwLeft;
252  int right=cwRight;
253 
254  /*
255  * the isCCW call might throw an
256  * IllegalArgumentException if degenerate ring does
257  * not contain 3 distinct points.
258  */
259  try
260  {
261  if (CGAlgorithms::isCCW(coord)) {
262  left=cwRight;
263  right=cwLeft;
264  }
265  }
266  catch(...)
267  {
268  delete coord;
269  throw;
270  }
271 
272  Edge *e=new Edge(coord, Label(argIndex, Location::BOUNDARY, left, right));
273  lineEdgeMap[lr]=e;
274  insertEdge(e);
275  insertPoint(argIndex,coord->getAt(0), Location::BOUNDARY);
276 }

278 void
279 GeometryGraph::addPolygon(const Polygon *p)
280 {
281  const LineString* ls;
282  const LinearRing* lr;
283 
284  ls = p->getExteriorRing();
285  assert(dynamic_cast<const LinearRing*>(ls));
286  lr = static_cast<const LinearRing*>(ls);
287  addPolygonRing(lr, Location::EXTERIOR, Location::INTERIOR);
288  for (size_t i=0, n=p->getNumInteriorRing(); i<n; ++i)
289  {
290  // Holes are topologically labelled opposite to the shell, since
291  // the interior of the polygon lies on their opposite side
292  // (on the left, if the hole is oriented CW)
293  ls = p->getInteriorRingN(i);
294  assert(dynamic_cast<const LinearRing*>(ls));
295  lr = static_cast<const LinearRing*>(ls);
296  addPolygonRing(lr, Location::INTERIOR, Location::EXTERIOR);
297  }

301 GeometryGraph::addLineString(const LineString *line)
302 {
303  CoordinateSequence* coord=CoordinateSequence::removeRepeatedPoints(line->getCoordinatesRO());
304  if(coord->getSize()<2) {
305  hasTooFewPointsVar=true;
306  invalidPoint=coord->getAt(0);
307  delete coord;
308  return;
309  }
310 
311  Edge *e=new Edge(coord, Label(argIndex, Location::INTERIOR));
312  lineEdgeMap[line]=e;
313  insertEdge(e);
314 
315  /*
316  * Add the boundary points of the LineString, if any.
317  * Even if the LineString is closed, add both points as if they
318  * were endpoints.
319  * This allows for the case that the node already exists and is
320  * a boundary point.
321  */
322  assert(coord->size() >= 2); // found LineString with single point
323  insertBoundaryPoint(argIndex, coord->getAt(0));
324  insertBoundaryPoint(argIndex, coord->getAt(coord->getSize()-1));

458 /*
459  * Adds points using the mod-2 rule of SFS. This is used to add the boundary
460  * points of dim-1 geometries (Curves/MultiCurves). According to the SFS,
461  * an endpoint of a Curve is on the boundary
462  * iff if it is in the boundaries of an odd number of Geometries
463  */
464 void
465 GeometryGraph::insertBoundaryPoint(int argIndex, const Coordinate& coord)
466 {
467  Node *n=nodes->addNode(coord);
468  // nodes always have labels
469  Label& lbl = n->getLabel();
470 
471  // the new point to insert is on a boundary
472  int boundaryCount=1;
473 
474  // determine the current location for the point (if any)
475  int loc = lbl.getLocation(argIndex,Position::ON);
476  if (loc==Location::BOUNDARY) boundaryCount++;
477 
478  // determine the boundary status of the point according to the
479  // Boundary Determination Rule
480  int newLoc = determineBoundary(boundaryNodeRule, boundaryCount);
481  lbl.setLocation(argIndex,newLoc);
482 }
