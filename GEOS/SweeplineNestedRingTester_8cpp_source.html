76 SweeplineNestedRingTester::isInside(LinearRing *innerRing,LinearRing *searchRing)
77 {
78  CoordinateSequence *innerRingPts=innerRing->getCoordinates();
79  CoordinateSequence *searchRingPts=searchRing->getCoordinates();
80 
81  if (!innerRing->getEnvelopeInternal()->intersects(searchRing->getEnvelopeInternal()))
82  return false;
83  const Coordinate *innerRingPt=IsValidOp::findPtNotNode(innerRingPts, searchRing, graph);
84 
85  // Unable to find a ring point not a node of the search ring
86  assert(innerRingPt!=nullptr);
87 
88  bool isInside=CGAlgorithms::isPointInRing(*innerRingPt,searchRingPts);
89  if (isInside) {
90  /*
91  * innerRingPt is const just because the input
92  * CoordinateSequence is const. If the input
93  * Polygon survives lifetime of this object
94  * we are safe.
95  */
96  nestedPt=const_cast<Coordinate *>(innerRingPt);
97  return true;
98  }
99  return false;
