74 void
75 VoronoiDiagramBuilder::create()
76 {
77  if( subdiv.get() ) return;
78 
79  diagramEnv = DelaunayTriangulationBuilder::envelope(*siteCoords);
80  //adding buffer around the final envelope
81  double expandBy = std::max(diagramEnv.getWidth() , diagramEnv.getHeight());
82  diagramEnv.expandBy(expandBy);
83  if(clipEnv)
84  diagramEnv.expandToInclude(clipEnv);
85 
86  std::unique_ptr<IncrementalDelaunayTriangulator::VertexList> vertices (
87  DelaunayTriangulationBuilder::toVertices(*siteCoords)
88  );
89 
90  subdiv.reset( new quadedge::QuadEdgeSubdivision(diagramEnv,tolerance) );
91  IncrementalDelaunayTriangulator triangulator(subdiv.get());
92  triangulator.insertSites(*vertices);
93 }

112 VoronoiDiagramBuilder::getDiagramEdges(const geom::GeometryFactory& geomFact)
113 {
114  create();
115  std::unique_ptr<geom::MultiLineString> edges = subdiv->getVoronoiDiagramEdges(geomFact);

122 std::unique_ptr<geom::GeometryCollection>
123 VoronoiDiagramBuilder::clipGeometryCollection(const geom::GeometryCollection& geom, const geom::Envelope& clipEnv)
124 {
125  std::unique_ptr<geom::Geometry> clipPoly ( geom.getFactory()->toGeometry(&clipEnv) );
126  std::unique_ptr< std::vector<Geometry*> >clipped(new std::vector<Geometry*>);
127  for(std::size_t i=0 ; i < geom.getNumGeometries() ; i++)
128  {
129  const Geometry* g = geom.getGeometryN(i);
130  std::unique_ptr<Geometry> result;
131  // don't clip unless necessary
132  if(clipEnv.contains(g->getEnvelopeInternal()))
133  {
134  result.reset( g->clone() );
135  // TODO: check if userData is correctly cloned here?
136  }
137  else if(clipEnv.intersects(g->getEnvelopeInternal()))
138  {
139  result.reset( clipPoly->intersection(g) );
140  result->setUserData(((Geometry*)g)->getUserData()); // TODO: needed ?
141  }
142 
143  if(result.get() && !result->isEmpty() )
144  {
145  clipped->push_back(result.release());
146  }
147  }
148  return std::unique_ptr<GeometryCollection>(geom.getFactory()->createGeometryCollection(clipped.release()));
149 }
