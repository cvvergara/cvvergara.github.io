89 DirectedEdge*
90 DirectedEdgeStar::getRightmostEdge()
91 {
92  EdgeEndStar::iterator it=begin();
93  if ( it==end() ) return nullptr;
94 
95  assert(*it);
96  assert(dynamic_cast<DirectedEdge*>(*it));
97  DirectedEdge *de0=static_cast<DirectedEdge*>(*it);
98  ++it;
99  if ( it==end() ) return de0;
100 
101  it=end(); --it;
102 
103  assert(*it);
104  assert(dynamic_cast<DirectedEdge*>(*it));
105  DirectedEdge *deLast=static_cast<DirectedEdge*>(*it);
106 
107  assert(de0);
108  int quad0=de0->getQuadrant();
109  assert(deLast);
110  int quad1=deLast->getQuadrant();
111  if (Quadrant::isNorthern(quad0) && Quadrant::isNorthern(quad1))
112  return de0;
113  else if (!Quadrant::isNorthern(quad0) && !Quadrant::isNorthern(quad1))
114  return deLast;
115  else {
116  // edges are in different hemispheres - make sure we return one that is non-horizontal
117  //DirectedEdge *nonHorizontalEdge=NULL;
118  if (de0->getDy()!=0)
119  return de0;
120  else if (deLast->getDy()!=0)
121  return deLast;
122  }
123  assert(0); // found two horizontal edges incident on node
124  return nullptr;
125 }

213 DirectedEdgeStar::linkResultDirectedEdges()
214  // throw(TopologyException *)
215 {
216  // make sure edges are copied to resultAreaEdges list
217  getResultAreaEdges();
218  // find first area edge (if any) to start linking at
219  DirectedEdge *firstOut=nullptr;
220  DirectedEdge *incoming=nullptr;
221  int state=SCANNING_FOR_INCOMING;
222  // link edges in CCW order
223  for (std::vector<DirectedEdge*>::iterator
224  i=resultAreaEdgeList->begin(), iEnd=resultAreaEdgeList->end();
225  i != iEnd;
226  ++i)
227  {
228  DirectedEdge *nextOut=*i;
229  assert(nextOut);
230 
231  // skip de's that we're not interested in
232  if (!nextOut->getLabel().isArea()) continue;
233 
234  DirectedEdge *nextIn=nextOut->getSym();
235  assert(nextIn);
236 
237  // record first outgoing edge, in order to link the last incoming edge
238  if (firstOut==nullptr && nextOut->isInResult()) firstOut=nextOut;
239 
240  switch (state) {
241  case SCANNING_FOR_INCOMING:
242  if (!nextIn->isInResult()) continue;
243  incoming=nextIn;
244  state=LINKING_TO_OUTGOING;
245  break;
246  case LINKING_TO_OUTGOING:
247  if (!nextOut->isInResult()) continue;
248  incoming->setNext(nextOut);
249  state=SCANNING_FOR_INCOMING;
250  break;
251  }
252  }
253  if (state==LINKING_TO_OUTGOING) {
254  if (firstOut==nullptr)
255  {
256  throw util::TopologyException("no outgoing dirEdge found",
257  getCoordinate());
258  }
259  assert(firstOut->isInResult()); // unable to link last incoming dirEdge
260  assert(incoming);
261  incoming->setNext(firstOut);
262  }

344 DirectedEdgeStar::findCoveredLineEdges()
345 {
346  // Since edges are stored in CCW order around the node,
347  // as we move around the ring we move from the right to the left side of the edge
348 
356  int startLoc=Location::UNDEF;
357 
358  EdgeEndStar::iterator endIt=end();
359  for (EdgeEndStar::iterator it=begin(); it!=endIt; ++it)
360  {
361  assert(*it);
362  assert(dynamic_cast<DirectedEdge*>(*it));
363  DirectedEdge *nextOut=static_cast<DirectedEdge*>(*it);
364 
365  DirectedEdge *nextIn=nextOut->getSym();
366  assert(nextIn);
367 
368  if (!nextOut->isLineEdge()) {
369  if (nextOut->isInResult()) {
370  startLoc=Location::INTERIOR;
371  break;
372  }
373  if (nextIn->isInResult()) {
374  startLoc=Location::EXTERIOR;
375  break;
376  }
377  }
378  }
379 
380  // no A edges found, so can't determine if L edges are covered or not
381  if (startLoc==Location::UNDEF) return;
382 
388  int currLoc=startLoc;
389  for (EdgeEndStar::iterator it=begin(); it!=endIt; ++it)
390  {
391  assert(*it);
392  assert(dynamic_cast<DirectedEdge*>(*it));
393  DirectedEdge *nextOut=static_cast<DirectedEdge*>(*it);
394 
395  DirectedEdge *nextIn=nextOut->getSym();
396  assert(nextIn);
397 
398  if (nextOut->isLineEdge()) {
399  nextOut->getEdge()->setCovered(currLoc==Location::INTERIOR);
400  } else { // edge is an Area edge
401  if (nextOut->isInResult())
402  currLoc=Location::EXTERIOR;
403  if (nextIn->isInResult())
404  currLoc=Location::INTERIOR;
405  }
406  }
