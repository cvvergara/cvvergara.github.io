106 /* static public */
107 bool
108 OverlayOp::isResultOfOp(int loc0, int loc1, OverlayOp::OpCode opCode)
109 {
110  if (loc0==Location::BOUNDARY) loc0=Location::INTERIOR;
111  if (loc1==Location::BOUNDARY) loc1=Location::INTERIOR;
112  switch (opCode) {
113  case opINTERSECTION:
114  return loc0==Location::INTERIOR && loc1==Location::INTERIOR;
115  case opUNION:
116  return loc0==Location::INTERIOR || loc1==Location::INTERIOR;
117  case opDIFFERENCE:
118  return loc0==Location::INTERIOR && loc1!=Location::INTERIOR;
119  case opSYMDIFFERENCE:
120  return (loc0==Location::INTERIOR && loc1!=Location::INTERIOR)
121  || (loc0!=Location::INTERIOR && loc1==Location::INTERIOR);
122  }
123  return false;
124 }

512 int
513 OverlayOp::mergeZ(Node *n, const LineString *line) const
514 {
515  const CoordinateSequence *pts = line->getCoordinatesRO();
516  const Coordinate &p = n->getCoordinate();
517  LineIntersector li;
518  for(size_t i=1, size=pts->size(); i<size; ++i)
519  {
520  const Coordinate &p0=pts->getAt(i-1);
521  const Coordinate &p1=pts->getAt(i);
522  li.computeIntersection(p, p0, p1);
523  if (li.hasIntersection())
524  {
525  if ( p == p0 )
526  {
527  n->addZ(p0.z);
528  }
529  else if ( p == p1 )
530  {
531  n->addZ(p1.z);
532  }
533  else
534  {
535  n->addZ(LineIntersector::interpolateZ(p,
536  p0, p1));
537  }
538  return 1;
539  }
540  }
541  return 0;

673 /*private*/
674 void
675 OverlayOp::computeOverlay(OverlayOp::OpCode opCode)
676  //throw(TopologyException *)
677 {
678 
679  // Compute the target envelope
680  const Envelope *env = nullptr;
681  const Envelope *env0 = getArgGeometry(0)->getEnvelopeInternal();
682  const Envelope *env1 = getArgGeometry(1)->getEnvelopeInternal();
683  Envelope opEnv;
684  if ( resultPrecisionModel->isFloating() )
685  {
686  // Envelope-based optimization only works in floating precision
687  switch (opCode) {
688  case opINTERSECTION:
689  env0->intersection(*env1, opEnv);
690  env = &opEnv;
691  break;
692  case opDIFFERENCE:
693  opEnv = *env0;
694  env = &opEnv;
695  break;
696  default:
697  break;
698  }
699  }
700  //env = 0; // WARNING: uncomment to disable env-optimization
701 
702  // copy points from input Geometries.
703  // This ensures that any Point geometries
704  // in the input are considered for inclusion in the result set
705  copyPoints(0, env);
706  copyPoints(1, env);
707 
708  GEOS_CHECK_FOR_INTERRUPTS();
709 
710  // node the input Geometries

910 /*private*/
911 void
912 OverlayOp::computeLabelsFromDepths()
913 {
914  for(auto &e : edgeList.getEdges())
915  {
916  Label& lbl = e->getLabel();
917  Depth &depth = e->getDepth();
918 
919  /*
920  * Only check edges for which there were duplicates,
921  * since these are the only ones which might
922  * be the result of dimensional collapses.
923  */
924  if (depth.isNull()) continue;
925 
926  depth.normalize();
927  for (int i=0;i<2;i++)
928  {
929  if (!lbl.isNull(i) && lbl.isArea() && !depth.isNull(i))
930  {
931  /*
932  * if the depths are equal, this edge is the result of
933  * the dimensional collapse of two or more edges.
934  * It has the same location on both sides of the edge,
935  * so it has collapsed to a line.
936  */
937  if (depth.getDelta(i)==0) {
938  lbl.toLine(i);
939  } else {
940  /*
941  * This edge may be the result of a dimensional collapse,
942  * but it still has different locations on both sides. The
943  * label of the edge must be updated to reflect the resultant
944  * side locations indicated by the depth values.
945  */
946  assert(!depth.isNull(i,Position::LEFT)); // depth of LEFT side has not been initialized
947  lbl.setLocation(i,Position::LEFT,depth.getLocation(i,Position::LEFT));
948  assert(!depth.isNull(i,Position::RIGHT)); // depth of RIGHT side has not been initialized
949  lbl.setLocation(i,Position::RIGHT,depth.getLocation(i,Position::RIGHT));
950  }
951  }
952  }
953  }
