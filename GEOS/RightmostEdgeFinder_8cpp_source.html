52 void
53 RightmostEdgeFinder::findEdge(std::vector<DirectedEdge*>* dirEdgeList)
54 {
55 
56 #ifndef NDEBUG
57  size_t checked=0;
58 #endif
59 
60  /*
61  * Check all forward DirectedEdges only. This is still general,
62  * because each edge has a forward DirectedEdge.
63  */
64  size_t dirEdgeListSize=dirEdgeList->size();
65  for(size_t i=0; i<dirEdgeListSize; ++i)
66  {
67  DirectedEdge *de=(*dirEdgeList)[i];
68  assert(de);
69  if (!de->isForward()) continue;
70  checkForRightmostCoordinate(de);
71 #ifndef NDEBUG
72  ++checked;
73 #endif
74  }
75 
76  if ( ! minDe ) {
77  // I don't know why, but it looks like this can happen
78  // (invalid PlanarGraph, I think)
79  // See http://trac.osgeo.org/geos/ticket/605#comment:17
80  //
81  throw util::TopologyException("No forward edges found in buffer subgraph");
82  }
83 
84 #ifndef NDEBUG
85  assert(checked>0);
86  assert(minIndex>=0);
87  assert(minDe);
88 #endif
89 
90  /*
91  * If the rightmost point is a node, we need to identify which of
92  * the incident edges is rightmost.
93  */
94  assert(minIndex!=0 || minCoord==minDe->getCoordinate());
95  // inconsistency in rightmost processing
96 
97  if (minIndex==0 ) {
98  findRightmostEdgeAtNode();
99  } else {
100  findRightmostEdgeAtVertex();
101  }
102 
103  /*
104  * now check that the extreme side is the R side.
105  * If not, use the sym instead.
106  */
107  orientedDe=minDe;
108  int rightmostSide=getRightmostSide(minDe, minIndex);
109  if (rightmostSide==Position::LEFT) {
110  orientedDe=minDe->getSym();
111  }
112 }

115 void
116 RightmostEdgeFinder::findRightmostEdgeAtNode()
117 {
118  Node *node=minDe->getNode();
119  assert(node);
120 
121  assert(dynamic_cast<DirectedEdgeStar*>(node->getEdges()));
122  DirectedEdgeStar *star=static_cast<DirectedEdgeStar*>(node->getEdges());
123 
124  // Warning! NULL could be returned if the star is empty!
125  minDe=star->getRightmostEdge();
126  assert(minDe);
127 
128  // the DirectedEdge returned by the previous call is not
129  // necessarily in the forward direction. Use the sym edge if it isn't.
130  if (!minDe->isForward())
131  {
132  minDe=minDe->getSym();
133 
134  const Edge* minEdge=minDe->getEdge();
135  assert(minEdge);
136 
137  const CoordinateSequence* minEdgeCoords =
138  minEdge->getCoordinates();
139  assert(minEdgeCoords);
140 
141  minIndex=(int)(minEdgeCoords->getSize())-1;
142  assert(minIndex>=0);
143  }

146 /*private*/
147 void
148 RightmostEdgeFinder::findRightmostEdgeAtVertex()
149 {
150  /*
151  * The rightmost point is an interior vertex, so it has
152  * a segment on either side of it.
153  * If these segments are both above or below the rightmost
154  * point, we need to determine their relative orientation
155  * to decide which is rightmost.
156  */
157 
158  Edge* minEdge=minDe->getEdge();
159  assert(minEdge);
160  const CoordinateSequence *pts=minEdge->getCoordinates();
161  assert(pts);
162 
163  // rightmost point expected to be interior vertex of edge
164  assert(minIndex>0);
165  assert((size_t)minIndex<pts->getSize());
166 
167  const Coordinate& pPrev=pts->getAt(minIndex-1);
168  const Coordinate& pNext=pts->getAt(minIndex+1);
169  int orientation=CGAlgorithms::computeOrientation(
170  minCoord,
171  pNext,
172  pPrev);
173  bool usePrev=false;
174 
175  // both segments are below min point
176  if ( pPrev.y < minCoord.y && pNext.y < minCoord.y
177  && orientation == CGAlgorithms::COUNTERCLOCKWISE)
178  {
179  usePrev=true;
180  }
181  else if ( pPrev.y > minCoord.y && pNext.y > minCoord.y
182  && orientation == CGAlgorithms::CLOCKWISE)
183  {
184  usePrev=true;
185  }
186 
187  // if both segments are on the same side, do nothing - either is safe
188  // to select as a rightmost segment
189  if (usePrev) {
190  minIndex=minIndex-1;
191  }
192 }

196 RightmostEdgeFinder::checkForRightmostCoordinate(DirectedEdge *de)
197 {
198  const Edge* deEdge=de->getEdge();
199  assert(deEdge);
200 
201  const CoordinateSequence *coord=deEdge->getCoordinates();
202  assert(coord);
203 
204  // only check vertices which are the starting point of
205  // a non-horizontal segment
206  size_t n=coord->getSize()-1;
207  for(size_t i=0; i<n; i++)
208  {
209  // only check vertices which are the start or end point
210  // of a non-horizontal segment
211  // <FIX> MD 19 Sep 03 - NO! we can test all vertices,
212  // since the rightmost must have a non-horiz segment adjacent to it
213  if (minCoord.isNull() ||
214  coord->getAt(i).x > minCoord.x )
215  {
216  minDe = de;
217  minIndex = (int)i;
218  minCoord = coord->getAt(i);
219  }
220  }

225 RightmostEdgeFinder::getRightmostSide(DirectedEdge *de, int index)
226 {
227  int side = getRightmostSideOfSegment(de, index);
228 
229  if (side<0) side=getRightmostSideOfSegment(de, index-1);
230 
231  if (side<0)
232  {
233  // reaching here can indicate that segment is horizontal
234  // Assert::shouldNeverReachHere(
235  // "problem with finding rightmost side of segment");
236 
237  minCoord=Coordinate::getNull();
238  checkForRightmostCoordinate(de);
239  }
240

244 /*private*/
245 int
246 RightmostEdgeFinder::getRightmostSideOfSegment(DirectedEdge *de, int i)
247 {
248  assert(de);
249 
250  const Edge *e=de->getEdge();
251  assert(e);
252 
253  const CoordinateSequence *coord=e->getCoordinates();
254  assert(coord);
255 
256  if ( i < 0 || i+1 >= (int)coord->getSize() ) return -1;
257 
258  // indicates edge is parallel to x-axis
259  if (coord->getAt(i).y == coord->getAt(i+1).y) return -1;
260 
261  int pos=Position::LEFT;
262  if (coord->getAt(i).y < coord->getAt(i+1).y) pos=Position::RIGHT;
263  return pos;
264 }
