89 SimpleMCSweepLineIntersector::add(Edge *edge, void* edgeSet)
90 {
91  MonotoneChainEdge *mce=edge->getMonotoneChainEdge();
92  vector<int> &startIndex=mce->getStartIndexes();
93  size_t n = startIndex.size()-1;
94  events.reserve(events.size()+(n*2));

97  GEOS_CHECK_FOR_INTERRUPTS();

145 void
146 SimpleMCSweepLineIntersector::processOverlaps(int start, int end,
147  SweepLineEvent *ev0, SegmentIntersector *si)
148 {
149  MonotoneChain *mc0=(MonotoneChain*) ev0->getObject();
150 
151  /*
152  * Since we might need to test for self-intersections,
153  * include current insert event object in list of event objects to test.
154  * Last index can be skipped, because it must be a Delete event.
155  */
156  for(int i=start; i<end; ++i)
157  {
158  SweepLineEvent *ev1=events[i];
159  if (ev1->isInsert())
160  {
161  MonotoneChain *mc1=(MonotoneChain*) ev1->getObject();
162  // don't compare edges in same group
163  // null group indicates that edges should be compared
164  if (ev0->edgeSet==nullptr || (ev0->edgeSet!=ev1->edgeSet))
165  {
166  mc0->computeIntersections(mc1,si);
167  nOverlaps++;
168  }
169  }
170  }
