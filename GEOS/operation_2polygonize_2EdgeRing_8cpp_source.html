49 EdgeRing *
50 EdgeRing::findEdgeRingContaining(EdgeRing *testEr,
51  vector<EdgeRing*> *shellList)
52 {
53  const LinearRing *testRing=testEr->getRingInternal();
54  if ( ! testRing ) return nullptr;
55  const Envelope *testEnv=testRing->getEnvelopeInternal();
56  Coordinate testPt=testRing->getCoordinateN(0);
57  EdgeRing *minShell=nullptr;
58  const Envelope *minEnv=nullptr;
59 
60  typedef std::vector<EdgeRing*> ERList;
61  for(ERList::size_type i=0, e=shellList->size(); i<e; ++i) {
62  EdgeRing *tryShell=(*shellList)[i];
63  LinearRing *tryRing=tryShell->getRingInternal();
64  const Envelope *tryEnv=tryRing->getEnvelopeInternal();
65  if (minShell!=nullptr) minEnv=minShell->getRingInternal()->getEnvelopeInternal();
66  bool isContained=false;
67 
68  // the hole envelope cannot equal the shell envelope
69 
70  if (tryEnv->equals(testEnv)) continue;
71 
72  const CoordinateSequence *tryCoords =
73  tryRing->getCoordinatesRO();
74 
75  if ( tryEnv->contains(testEnv) ) {
76 
77  // TODO: don't copy testPt !
78  testPt = ptNotInList(testRing->getCoordinatesRO(), tryCoords);
79 
80  if ( CGAlgorithms::isPointInRing(testPt, tryCoords) ) {
81  isContained=true;
82  }
83 
84  }
85 
86  // check if this new containing ring is smaller
87  // than the current minimum ring
88  if (isContained) {
89  if (minShell==nullptr || minEnv->contains(tryEnv)) {
90  minShell=tryShell;
91  }
92  }
93  }
94  return minShell;
95 }
