130 Edge *
131 EdgeIntersectionList::createSplitEdge(EdgeIntersection *ei0,
132  EdgeIntersection *ei1)
133 {
134 #if GEOS_DEBUG
135  cerr<<"["<<this<<"] EdgeIntersectionList::createSplitEdge()"<<endl;
136 #endif // GEOS_DEBUG
137  int npts=ei1->segmentIndex-ei0->segmentIndex+2;
138 
139  const Coordinate& lastSegStartPt=edge->pts->getAt(ei1->segmentIndex);
140 
141  // if the last intersection point is not equal to the its segment
142  // start pt, add it to the points list as well.
143  // (This check is needed because the distance metric is not totally
144  // reliable!). The check for point equality is 2D only - Z values
145  // are ignored
146  bool useIntPt1=ei1->dist>0.0 || !ei1->coord.equals2D(lastSegStartPt);
147 
148  if (!useIntPt1) --npts;
149 
150 #if GEOS_DEBUG
151  cerr<<" npts:"<<npts<<endl;
152 #endif // GEOS_DEBUG
153 
154  vector<Coordinate> *vc=new vector<Coordinate>();
155  vc->reserve(npts);
156 
157  vc->push_back(ei0->coord);
158  for(int i=ei0->segmentIndex+1; i<=ei1->segmentIndex;i++)
159  {
160  if ( ! useIntPt1 && ei1->segmentIndex == i )
161  {
162  vc->push_back(ei1->coord);
163  }
164  else
165  {
166  vc->push_back(edge->pts->getAt(i));
167  }
168  }
169 
170  if (useIntPt1)
171  {
172  vc->push_back(ei1->coord);
173  }
174 
175  CoordinateSequence* pts=new CoordinateArraySequence(vc);
176 
177  return new Edge(pts, edge->getLabel());
178 }
