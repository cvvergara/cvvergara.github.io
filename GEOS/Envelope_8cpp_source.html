353 /*public*/
354 double
355 Envelope::distance(const Envelope* env) const
356 {
357  if (intersects(env)) return 0;
358  double dx=0.0;
359  if(maxx<env->minx) dx=env->minx-maxx;
360  if(minx>env->maxx) dx=minx-env->maxx;
361  double dy=0.0;
362  if(maxy<env->miny) dy=env->miny-maxy;
363  if(miny>env->maxy) dy=miny-env->maxy;
364  // if either is zero, the envelopes overlap either vertically or horizontally
365  if (dx==0.0) return dy;
366  if (dy==0.0) return dx;
367  return sqrt(dx*dx+dy*dy);
368 }

400 vector<string>
401 Envelope::split(const string &str, const string &delimiters)
402 {
403  vector<string> tokens;
404 
405  // Find first "non-delimiter".
406  string::size_type lastPos = 0;
407  string::size_type pos = str.find_first_of(delimiters, lastPos);
408 
409  while (string::npos != pos || string::npos != lastPos)
410  {
411  // Found a token, add it to the vector.
412  tokens.push_back(str.substr(lastPos, pos - lastPos));
413 
414  // Skip delimiters. Note the "not_of"
415  lastPos = str.find_first_not_of(delimiters, pos);
416 
417  // Find next "non-delimiter"
418  pos = str.find_first_of(delimiters, lastPos);
419  }
420 
421  return tokens;
422 }
