103 HotPixel::intersectsScaled(const Coordinate& p0,
104  const Coordinate& p1) const
105 {
106 
107  double const segMinx = std::min(p0.x, p1.x);
108  double const segMaxx = std::max(p0.x, p1.x);
109  double const segMiny = std::min(p0.y, p1.y);
110  double const segMaxy = std::max(p0.y, p1.y);
111 
112  bool isOutsidePixelEnv = maxx < segMinx
113  || minx > segMaxx
114  || maxy < segMiny
115  || miny > segMaxy;
116 
117  if (isOutsidePixelEnv) return false;
118 
119  bool intersects = intersectsToleranceSquare(p0, p1);
120 
121  // Found bad envelope test
122  assert(!(isOutsidePixelEnv && intersects));
123

128 bool
129 HotPixel::intersectsToleranceSquare(const Coordinate& p0,
130  const Coordinate& p1) const
131 {
132  bool intersectsLeft = false;
133  bool intersectsBottom = false;
134 
135  li.computeIntersection(p0, p1, corner[0], corner[1]);
136  if (li.isProper()) return true;
137 
138  li.computeIntersection(p0, p1, corner[1], corner[2]);
139  if (li.isProper()) return true;
140  if (li.hasIntersection()) intersectsLeft = true;
141 
142  li.computeIntersection(p0, p1, corner[2], corner[3]);
143  if (li.isProper()) return true;
144  if (li.hasIntersection()) intersectsBottom = true;
145 
146  li.computeIntersection(p0, p1, corner[3], corner[0]);
147  if (li.isProper()) return true;
148 
149  if (intersectsLeft && intersectsBottom) return true;
150 
151  if (p0.equals2D(pt)) return true;
152  if (p1.equals2D(pt)) return true;
153 
154  return false;
