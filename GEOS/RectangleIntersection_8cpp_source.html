117 bool
118 RectangleIntersection::clip_linestring_parts(const geom::LineString * gi,
119  RectangleIntersectionBuilder & parts,
120  const Rectangle & rect)
121 {
122  using namespace geos::geom;
123 
124  int n = static_cast<int>(gi->getNumPoints());
125 
126  if(gi == nullptr || n<1)
127  return false;
128 
129  // For shorthand code
130 
131  std::vector<Coordinate> cs;
132  gi->getCoordinatesRO()->toVector(cs);
133  //const geom::CoordinateSequence &cs = *(gi->getCoordinatesRO());
134 
135  // Keep a record of the point where a line segment entered
136  // the rectangle. If the boolean is set, we must insert
137  // the point to the beginning of the linestring which then
138  // continues inside the rectangle.
139 
140  double x0 = 0;
141  double y0 = 0;
142  bool add_start = false;
143 
144  // Start iterating
145 
146  int i = 0;
147 
148  while(i<n)
149  {
150  // Establish initial position
151 
152  double x = cs[i].x;
153  double y = cs[i].y;
154  Rectangle::Position pos = rect.position(x,y);
155 
156  if(pos == Rectangle::Outside)
157  {
158  // Skip points as fast as possible until something has to be checked
159  // in more detail.
160 
161  ++i; // we already know it is outside
162 
163  if(x < rect.xmin())
164  while(i < n && cs[i].x < rect.xmin())
165  ++i;
166 
167  else if(x > rect.xmax())
168  while(i < n && cs[i].x > rect.xmax())
169  ++i;
170 
171  else if(y < rect.ymin())
172  while(i < n && cs[i].y < rect.ymin())
173  ++i;
174 
175  else if(y > rect.ymax())
176  while(i < n && cs[i].y > rect.ymax())
177  ++i;
178 
179  if(i >= n)
180  return false;
181 
182  // Establish new position
183  x = cs[i].x;
184  y = cs[i].y;
185  pos = rect.position(x,y);
186 
187  // Handle all possible cases
188  x0 = cs[i-1].x;
189  y0 = cs[i-1].y;
190  clip_to_edges(x0,y0,x,y,rect);
191 
192  if(pos == Rectangle::Inside)
193  {
194  add_start = true; // x0,y0 must have clipped the rectangle
195  // Main loop will enter the Inside/Edge section
196 
197  }
198  else if(pos == Rectangle::Outside)
199  {
200  // From Outside to Outside. We need to check whether
201  // we created a line segment inside the box. In any
202  // case, we will continue the main loop after this
203  // which will then enter the Outside section.
204 
205  // Clip the other end too
206  clip_to_edges(x,y,x0,y0,rect);
207 
208  Rectangle::Position prev_pos = rect.position(x0,y0);
209  pos = rect.position(x,y);
210 
211  if( different(x0,y0,x,y) && // discard corners etc
212  Rectangle::onEdge(prev_pos) && // discard if does not intersect rect
213  Rectangle::onEdge(pos) &&
214  !Rectangle::onSameEdge(prev_pos,pos) // discard if travels along edge
215  )
216  {
217  std::vector<Coordinate> *coords = new std::vector<Coordinate>(2);
218  (*coords)[0] = Coordinate(x0,y0);
219  (*coords)[1] = Coordinate(x,y);
220  CoordinateSequence *seq = _csf->create(coords);
221  geom::LineString * line = _gf->createLineString(seq);
222  parts.add(line);
223  }
224 
225  // Continue main loop outside the rect
226 
227  }
228  else
229  {
230  // From outside to edge. If the edge we hit first when
231  // following the line is not the edge we end at, then
232  // clearly we must go through the rectangle and hence
233  // a start point must be set.
234 
235  Rectangle::Position newpos = rect.position(x0,y0);
236  if(!Rectangle::onSameEdge(pos,newpos))
237  {
238  add_start = true;
239  }
240  else
241  {
242  // we ignore the travel along the edge and continue
243  // the main loop at the last edge point
244  }
245  }
246  }
247 
248  else
249  {
250  // The point is now stricly inside or on the edge.
251  // Keep iterating until the end or the point goes
252  // outside. We may have to output partial linestrings
253  // while iterating until we go strictly outside
254 
255  int start_index = i; // 1st valid original point
256  bool go_outside = false;
257 
258  while(!go_outside && ++i<n)
259  {
260  x = cs[i].x;
261  y = cs[i].y;
262 
263  Rectangle::Position prev_pos = pos;
264  pos = rect.position(x,y);
265 
266  if(pos == Rectangle::Inside)
267  {
268  // Just keep going
269  }
270  else if(pos == Rectangle::Outside)
271  {
272  go_outside = true;
273 
274  // Clip the outside point to edges
275  clip_to_edges(x, y, cs[i-1].x, cs[i-1].y, rect);
276  pos = rect.position(x,y);
277 
278  // Does the line exit through the inside of the box?
279 
280  bool through_box = (different(x,y,cs[i].x,cs[i].y) &&
281  !Rectangle::onSameEdge(prev_pos,pos));
282 
283  // Output a LineString if it at least one segment long
284 
285  if(start_index < i-1 || add_start || through_box)
286  {
287  std::vector<Coordinate> *coords = new std::vector<Coordinate>();
288  if(add_start)
289  {
290  coords->push_back(Coordinate(x0, y0));
291  add_start = false;
292  }
293  //line->addSubLineString(&g, start_index, i-1);
294  coords->insert(coords->end(), cs.begin()+start_index, cs.begin()+i);
295 
296  if(through_box) coords->push_back(Coordinate(x,y));
297 
298  CoordinateSequence *seq = _csf->create(coords);
299  geom::LineString * line = _gf->createLineString(seq);
300  parts.add(line);
301  }
302  // And continue main loop on the outside
303  }
304  else
305  {
306  // on same edge?
307  if(Rectangle::onSameEdge(prev_pos,pos))
308  {
309  // Nothing to output if we haven't been elsewhere
310  if(start_index < i-1 || add_start)
311  {
312  std::vector<Coordinate> *coords = new std::vector<Coordinate>();
313  //geom::LineString * line = new geom::LineString();
314  if(add_start)
315  {
316  //line->addPoint(x0,y0);
317  coords->push_back(Coordinate(x0, y0));
318  add_start = false;
319  }
320  //line->addSubLineString(&g, start_index, i-1);
321  coords->insert(coords->end(), cs.begin()+start_index, cs.begin()+i);
322 
323  CoordinateSequence *seq = _csf->create(coords);
324  geom::LineString * line = _gf->createLineString(seq);
325  parts.add(line);
326  }
327  start_index = i;
328  }
329  else
330  {
331  // On different edge. Must have gone through the box
332  // then - keep collecting points that generate inside
333  // line segments
334  }
335  }
336  }
337 
338  // Was everything in? If so, generate no output but return true in this case only
339  if(start_index == 0 && i >= n)
340  return true;
341 
342  // Flush the last line segment if data ended and there is something to flush
343 
344  if(!go_outside && // meaning data ended
345  (start_index < i-1 || add_start)) // meaning something has to be generated
346  {
347  std::vector<Coordinate> *coords = new std::vector<Coordinate>();
348  //geom::LineString * line = new geom::LineString();
349  if(add_start)
350  {
351  //line->addPoint(x0,y0);
352  coords->push_back(Coordinate(x0, y0));
353  add_start = false;
354  }
355  //line->addSubLineString(&g, start_index, i-1);
356  coords->insert(coords->end(), cs.begin()+start_index, cs.begin()+i);
357 
358  CoordinateSequence *seq = _csf->create(coords);
359  geom::LineString * line = _gf->createLineString(seq);
360  parts.add(line);
361  }
362 
363  }
364  }
365 
366  return false;

374 void
375 RectangleIntersection::clip_polygon_to_linestrings(const geom::Polygon * g,
376  RectangleIntersectionBuilder & toParts,
377  const Rectangle & rect)
378 {
379  if(g == nullptr || g->isEmpty())
380  return;
381 
382  // Clip the exterior first to see what's going on
383 
384  RectangleIntersectionBuilder parts(*_gf);
385 
386  // If everything was in, just clone the original
387 
388  if(clip_linestring_parts(g->getExteriorRing(), parts, rect))
389  {
390  toParts.add(dynamic_cast<geom::Polygon *>(g->clone()));
391  return;
392  }
393 
394  // Now, if parts is empty, our rectangle may be inside the polygon
395  // If not, holes are outside too
396 
397  if(parts.empty())
398  {
399  // We could now check whether the rectangle is inside the outer
400  // ring to avoid checking the holes. However, if holes are much
401  // smaller than the exterior ring just checking the holes
402  // separately could be faster.
403 
404  if(g->getNumInteriorRing() == 0)
405  return;
406 
407  }
408  else
409  {
410  // The exterior must have been clipped into linestrings.
411  // Move them to the actual parts collector, clearing parts.
412 
413  parts.reconnect();
414  parts.release(toParts);
415  }
416 
417  // Handle the holes now:
418  // - Clipped ones become linestrings
419  // - Intact ones become new polygons without holes
420 
421  for(int i=0, n= static_cast<int>(g->getNumInteriorRing()); i<n; ++i)
422  {
423  if(clip_linestring_parts(g->getInteriorRingN(i), parts, rect))
424  {
425  // clones
426  LinearRing *hole = dynamic_cast<LinearRing*>(g->getInteriorRingN(i)->clone());
427  // becomes exterior
428  Polygon *poly = _gf->createPolygon(hole, nullptr);
429  toParts.add(poly);
430  }
431  else if(!parts.empty())
432  {
433  parts.reconnect();
434  parts.release(toParts);
435  }
436  }

443 void
444 RectangleIntersection::clip_polygon_to_polygons(const geom::Polygon * g,
445  RectangleIntersectionBuilder & toParts,
446  const Rectangle & rect)
447 {
448  if(g == nullptr || g->isEmpty())
449  return;
450 
451  // Clip the exterior first to see what's going on
452 
453  RectangleIntersectionBuilder parts(*_gf);
454 
455  // If everything was in, just clone the original
456 
457  const LineString *shell = g->getExteriorRing();
458  if(clip_linestring_parts(shell, parts, rect))
459  {
460  toParts.add(dynamic_cast<geom::Polygon *>(g->clone()));
461  return;
462  }
463 
464  // If there were no intersections, the outer ring might be
465  // completely outside.
466 
467  using geos::algorithm::CGAlgorithms;
468  if( parts.empty() )
469  {
470  Coordinate rectCenter(rect.xmin(), rect.ymin());
471  rectCenter.x += (rect.xmax()-rect.xmin())/2;
472  rectCenter.y += (rect.ymax()-rect.ymin())/2;
473  if ( CGAlgorithms::locatePointInRing(rectCenter,
474  *g->getExteriorRing()->getCoordinatesRO())
475  != Location::INTERIOR )
476  {
477  return;
478  }
479  }
480  else
481  {
482  // TODO: make CCW checking part of clip_linestring_parts ?
483  if ( CGAlgorithms::isCCW(shell->getCoordinatesRO()) ) {
484  parts.reverseLines();
485  }
486  }
487 
488  // Must do this to make sure all end points are on the edges
489 
490  parts.reconnect();
491 
492  // Handle the holes now:
493  // - Clipped ones become part of the exterior
494  // - Intact ones become holes in new polygons formed by exterior parts
495 
496 
497  for(int i=0, n = static_cast<int>(g->getNumInteriorRing()); i<n; ++i)
498  {
499  RectangleIntersectionBuilder holeparts(*_gf);
500  const LineString *hole = g->getInteriorRingN(i);
501  if(clip_linestring_parts(hole, holeparts, rect))
502  {
503  // becomes exterior
504  LinearRing *cloned = dynamic_cast<LinearRing*>(hole->clone());
505  Polygon *poly = _gf->createPolygon(cloned, nullptr);
506  parts.add(poly);
507  }
508  else
509  {
510  if(!holeparts.empty())
511  {
512  // TODO: make CCW checking part of clip_linestring_parts ?
513  if ( ! CGAlgorithms::isCCW(hole->getCoordinatesRO()) ) {
514  holeparts.reverseLines();
515  }
516  holeparts.reconnect();
517  holeparts.release(parts);
518  }
519  else
520  {
521  using geos::algorithm::CGAlgorithms;
522  Coordinate rectCenter(rect.xmin(), rect.ymin());
523  rectCenter.x += (rect.xmax()-rect.xmin())/2;
524  rectCenter.y += (rect.ymax()-rect.ymin())/2;
525  if( CGAlgorithms::isPointInRing(rectCenter,
526  g->getInteriorRingN(i)->getCoordinatesRO()) )
527  {
528  // Completely inside the hole
529  return;
530  }
531  }
532  }
533  }
534 
535  parts.reconnectPolygons(rect);
536  parts.release(toParts);
537 
538 }

641  bool keep_polygons)
642 {

654  return clip_multipolygon(p, parts, rect, keep_polygons);
655  else if ( const GeometryCollection* p = dynamic_cast<const geom::GeometryCollection *>(g) )
656  return clip_geometrycollection(p, parts, rect, keep_polygons);
657  else {
658  throw util::UnsupportedOperationException("Encountered an unknown geometry component when clipping polygons");
