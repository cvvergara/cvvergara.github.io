61 void
62 PointBuilder::extractNonCoveredResultNodes(OverlayOp::OpCode opCode)
63 {
64  map<Coordinate*,Node*,CoordinateLessThen> &nodeMap =
65  op->getGraph().getNodeMap()->nodeMap;
66  map<Coordinate*,Node*,CoordinateLessThen>::iterator it=nodeMap.begin();
67  for (; it!=nodeMap.end(); ++it)
68  {
69  Node *n=it->second;
70 
71  // filter out nodes which are known to be in the result
72  if (n->isInResult()) continue;
73 
74  // if an incident edge is in the result, then
75  // the node coordinate is included already
76  if (n->isIncidentEdgeInResult()) continue;
77 
78  if ( n->getEdges()->getDegree() == 0 ||
79  opCode == OverlayOp::opINTERSECTION )
80  {
81 
87  const Label& label=n->getLabel();
88  if (OverlayOp::isResultOfOp(label, opCode))
89  filterCoveredNodeToPoint(n);
90  }
91  }
92 }
