91  // terminating condition for the recursion

93  {
94  //Debug.println("computeSelect:"+p0+p1);
95  mcs.select(*this, static_cast<unsigned int>(start0));

97  }
98  // nothing to do if the envelopes don't overlap
99  if (!searchEnv.intersects(mcs.tempEnv1))
100  return;
