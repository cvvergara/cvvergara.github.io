44 QuadEdge& IncrementalDelaunayTriangulator::insertSite(const Vertex &v)
45 {
53  QuadEdge *e = subdiv->locate(v);
54 
55  if(!e) {
56  throw LocateFailureException("");
57  }
58 
59  if (subdiv->isVertexOfEdge(*e, v)) {
60  // point is already in subdivision.
61  return *e;
62  }
63  else if (subdiv->isOnEdge(*e, v.getCoordinate())) {
64  // the point lies exactly on an edge, so delete the edge
65  // (it will be replaced by a pair of edges which have the point as a vertex)
66  e = &e->oPrev();
67  subdiv->remove(e->oNext());
68  }
69 
74  QuadEdge* base = &subdiv->makeEdge(e->orig(), v);
75 
76  QuadEdge::splice(*base, *e);
77  QuadEdge *startEdge = base;
78  do {
79  base = &subdiv->connect(*e, base->sym());
80  e = &base->oPrev();
81  } while (&e->lNext() != startEdge);
82 
83 
84  // Examine suspect edges to ensure that the Delaunay condition
85  // is satisfied.
86  for (;;)
87  {
88  QuadEdge* t = &e->oPrev();
89  if (t->dest().rightOf(*e) &&
90  v.isInCircle(e->orig(), t->dest(), e->dest())) {
91  QuadEdge::swap(*e);
92  e = &e->oPrev();
93  } else if (&e->oNext() == startEdge) {
94  return *base; // no more suspect edges.
95  } else {
