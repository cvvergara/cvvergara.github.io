72 Polygon::Polygon(LinearRing *newShell, vector<Geometry *> *newHoles,

212 Polygon::getBoundary() const
213 {
214  /*
215  * We will make sure that what we
216  * return is composed of LineString,
217  * not LinearRings
218  */
219 
220  const GeometryFactory* gf = getFactory();
221 
222  if (isEmpty()) {
223  return gf->createMultiLineString();
224  }
225 
226  if ( ! holes->size() )
227  {
228  return gf->createLineString(*shell).release();
229  }
230 
231  vector<Geometry *> *rings = new vector<Geometry *>(holes->size()+1);
232 
233  (*rings)[0] = gf->createLineString(*shell).release();
234  for(size_t i=0, n=holes->size(); i<n; ++i)
235  {
236  const LinearRing* hole = dynamic_cast<const LinearRing *>((*holes)[i]);
237  assert( hole );
238  LineString* ls = gf->createLineString( *hole ).release();
239  (*rings)[i + 1] = ls;
240  }
241  MultiLineString *ret = getFactory()->createMultiLineString(rings);
242  return ret;
243 }

253 {
254  const Polygon* otherPolygon=dynamic_cast<const Polygon*>(other);
255  if ( ! otherPolygon ) return false;
256 
257  if (!shell->equalsExact(otherPolygon->shell, tolerance)) {
258  return false;
259  }
260 
261  size_t nholes = holes->size();
262 
263  if (nholes != otherPolygon->holes->size()) {
264  return false;
265  }
266

268  {
269  const Geometry* hole=(*holes)[i];
270  const Geometry* otherhole=(*(otherPolygon->holes))[i];
271  if (!hole->equalsExact(otherhole, tolerance))
272  {
273  return false;
274  }
275  }
276 
277  return true;
278 }

344 Polygon::normalize(LinearRing *ring, bool clockwise)
345 {

348  }
349  CoordinateSequence* uniqueCoordinates=ring->getCoordinates();
350  uniqueCoordinates->deleteAt(uniqueCoordinates->getSize()-1);

352  CoordinateSequence::scroll(uniqueCoordinates, minCoordinate);
353  uniqueCoordinates->add(uniqueCoordinates->getAt(0));
354  if (algorithm::CGAlgorithms::isCCW(uniqueCoordinates)==clockwise) {
355  CoordinateSequence::reverse(uniqueCoordinates);
356  }
357  ring->setPoints(uniqueCoordinates);
358  delete(uniqueCoordinates);

475 bool
476 Polygon::isRectangle() const
477 {
478  if ( getNumInteriorRing() != 0 ) return false;
479  assert(shell!=nullptr);
480  if ( shell->getNumPoints() != 5 ) return false;
481 
482  const CoordinateSequence &seq = *(shell->getCoordinatesRO());
483 
484  // check vertices have correct values
485  const Envelope &env = *getEnvelopeInternal();
486  for (int i=0; i<5; i++) {
487  double x = seq.getX(i);
488  if (! (x == env.getMinX() || x == env.getMaxX())) return false;
489  double y = seq.getY(i);
490  if (! (y == env.getMinY() || y == env.getMaxY())) return false;
491  }
492 
493  // check vertices are in right order
494  double prevX = seq.getX(0);
495  double prevY = seq.getY(0);
496  for (int i = 1; i <= 4; i++) {
497  double x = seq.getX(i);
498  double y = seq.getY(i);
499  bool xChanged = (x != prevX);
500  bool yChanged = (y != prevY);
501  if (xChanged == yChanged) return false;
502  prevX = x;
503  prevY = y;
504  }
505  return true;
506 }
