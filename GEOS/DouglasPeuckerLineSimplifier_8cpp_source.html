59 DouglasPeuckerLineSimplifier::CoordsVectAutoPtr
60 DouglasPeuckerLineSimplifier::simplify()
61 {
62  CoordsVectAutoPtr coordList(new CoordsVect());
63 
64  // empty coordlist is the simplest, won't simplify further
65  if ( ! pts.size() ) return coordList;
66 
67  usePt = BoolVectAutoPtr(new BoolVect(pts.size(), true));
68  simplifySection(0, pts.size() - 1);
69 
70  for (std::size_t i=0, n=pts.size(); i<n; ++i)
71  {
72  if ( usePt->operator[](i) )
73  {
74  coordList->push_back(pts[i]);
75  }
76  }
77 
78  // unique_ptr transfer ownership to its
79  // returned copy
80  return coordList;
81 }

84 void
85 DouglasPeuckerLineSimplifier::simplifySection(
86  std::size_t i,
87  std::size_t j)
88 {
89  if ( (i+1) == j ) return;
90 
91  geos::geom::LineSegment seg(pts[i], pts[j]);
92  double maxDistance = -1.0;
93 
94  std::size_t maxIndex = i;
95 
96  for (std::size_t k=i+1; k<j; k++)
97  {
98  double distance = seg.distance(pts[k]);
99  if (distance > maxDistance) {
100  maxDistance = distance;
101  maxIndex = k;
102  }
103  }
104  if (maxDistance <= distanceTolerance) {
105  for(std::size_t k =i+1; k<j; k++)
106  {
107  usePt->operator[](k) = false;
108  }
109  }
110  else {
111  simplifySection(i, maxIndex);
112  simplifySection(maxIndex, j);
113  }
114 }
