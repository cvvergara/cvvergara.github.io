89 CoordinateSequence *
90 GeometryCollection::getCoordinates() const
91 {
92  vector<Coordinate> *coordinates = new vector<Coordinate>(getNumPoints());
93 
94  int k = -1;
95  for (size_t i=0; i<geometries->size(); ++i) {
96  CoordinateSequence* childCoordinates=(*geometries)[i]->getCoordinates();
97  size_t npts=childCoordinates->getSize();
98  for (size_t j=0; j<npts; ++j) {
99  k++;
100  (*coordinates)[k] = childCoordinates->getAt(j);
101  }
102  delete childCoordinates;
103  }
104  return CoordinateArraySequenceFactory::instance()->create(coordinates);
105 }
