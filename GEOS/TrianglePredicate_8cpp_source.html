40 bool
41 TrianglePredicate::isInCircleNormalized(
42  const Coordinate &a, const Coordinate &b, const Coordinate &c,
43  const Coordinate &p)
44 {
45  double adx = a.x - p.x;
46  double ady = a.y - p.y;
47  double bdx = b.x - p.x;
48  double bdy = b.y - p.y;
49  double cdx = c.x - p.x;
50  double cdy = c.y - p.y;
51 
52  double abdet = adx * bdy - bdx * ady;
53  double bcdet = bdx * cdy - cdx * bdy;
54  double cadet = cdx * ady - adx * cdy;
55  double alift = adx * adx + ady * ady;
56  double blift = bdx * bdx + bdy * bdy;
57  double clift = cdx * cdx + cdy * cdy;
58 
59  double disc = alift * bcdet + blift * cadet + clift * abdet;
60  return disc > 0;
61 }
