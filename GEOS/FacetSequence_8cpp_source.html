41 double FacetSequence::distance(const FacetSequence & facetSeq) const {

45  if (isPointThis && isPointOther) {

61 double FacetSequence::computePointLineDistance(const Coordinate & pt, const FacetSequence & facetSeq) const {
62  double minDistance = std::numeric_limits<double>::infinity();
63  double dist;
64  Coordinate q0;
65  Coordinate q1;
66 
67  for (size_t i = facetSeq.start; i < facetSeq.end - 1; i++) {
68  facetSeq.pts->getAt(i, q0);
