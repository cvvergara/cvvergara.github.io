224 EdgeRing::computePoints(DirectedEdge *newStart)
225  // throw(const TopologyException &)
226 {
227  startDe=newStart;
228  DirectedEdge *de=newStart;
229  bool isFirstEdge=true;
230  do {
231  //util::Assert::isTrue(de!=NULL,"EdgeRing::computePoints: found null Directed Edge");
232  //assert(de!=NULL); // EdgeRing::computePoints: found null Directed Edge
233  if(de==nullptr)
234  throw util::TopologyException(
235  "EdgeRing::computePoints: found null Directed Edge");
236 
237  if (de->getEdgeRing()==this)
238  throw util::TopologyException(
239  "Directed Edge visited twice during ring-building",
240  de->getCoordinate());
241 
242  edges.push_back(de);
243  const Label& deLabel = de->getLabel();
244  assert(deLabel.isArea());
245  mergeLabel(deLabel);
246  addPoints(de->getEdge(),de->isForward(),isFirstEdge);
247  isFirstEdge=false;
248  setEdgeRing(de,this);
249  de=getNext(de);
250  } while (de!=startDe);
251 
252  testInvariant();

269 EdgeRing::computeMaxNodeDegree()
270 {
271  maxNodeDegree=0;
272  DirectedEdge *de=startDe;
273  do {
274  Node *node=de->getNode();
275  EdgeEndStar* ees = node->getEdges();
276  assert(dynamic_cast<DirectedEdgeStar*>(ees));
277  DirectedEdgeStar* des = static_cast<DirectedEdgeStar*>(ees);

279  if (degree>maxNodeDegree) maxNodeDegree=degree;

333 EdgeRing::addPoints(Edge *edge, bool isForward, bool isFirstEdge)
334 {
335  // EdgeRing::addPoints: can't add points after LinearRing construction
336  assert(ring==nullptr);
337 
338  assert(edge);

340 
341  assert(edgePts);

343 
344  assert(pts);
345 
346  if (isForward) {
347  size_t startIndex=1;
348  if (isFirstEdge) startIndex=0;
349  for (size_t i=startIndex; i<numEdgePts; ++i)
350  {
351  pts->add(edgePts->getAt(i));
352  }
353  }
354 
355  else { // is backward
356  size_t startIndex=numEdgePts-1;
357  if (isFirstEdge) startIndex=numEdgePts;
358  //for (int i=startIndex;i>=0;i--)
359  for (size_t i=startIndex; i>0; --i)
360  {
361  pts->add(edgePts->getAt(i-1));
362  }
363  }
364 
365  testInvariant();
366

371 EdgeRing::containsPoint(const Coordinate& p)
372 {
373 
374  testInvariant();
375 
376  assert(ring);
377 
378  const Envelope* env=ring->getEnvelopeInternal();
379  assert(env);
380  if ( ! env->contains(p) ) return false;
381 
382  if ( ! CGAlgorithms::isPointInRing(p, ring->getCoordinatesRO()) )
383  return false;
384 
385  for (vector<EdgeRing*>::iterator i=holes.begin(); i<holes.end(); ++i)
386  {
387  EdgeRing *hole=*i;
388  assert(hole);
389  if (hole->containsPoint(p))
390  {
391  return false;
392  }
393  }
394  return true;
