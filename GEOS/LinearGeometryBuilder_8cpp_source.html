89 void
90 LinearGeometryBuilder::endLine()
91 {
92  if (!coordList)
93  {
94  return;
95  }
96  if (coordList->size() < 2)
97  {
98  if (ignoreInvalidLines)
99  {
100  if (coordList)
101  {
102  delete coordList;
103  coordList = nullptr;
104  }
105  return;
106  }
107  else if (fixInvalidLines)
108  {
109  assert(!coordList->isEmpty()); // just to be sure
110 
111  // NOTE: we copy the point cause reallocation of
112  // vector memory will invalidate the reference
113  // to one of its elements.
114  //
115  // We wouldn't have such problems with a vector
116  // of pointers (as used in JTS)...
117  //
118  Coordinate firstPoint = (*coordList)[0];
119  add(firstPoint);
120  }
121  }
122 
123  LineString* line = nullptr;
124  try
125  {
126  line = geomFact->createLineString(coordList);
127  }
128  catch (util::IllegalArgumentException ex)
129  {
130  // exception is due to too few points in line.
131  // only propagate if not ignoring short lines
132  if (! ignoreInvalidLines)
133  throw ex;
134  }
135 
136  if (line) lines.push_back(line);
137  coordList = nullptr;
138 }
