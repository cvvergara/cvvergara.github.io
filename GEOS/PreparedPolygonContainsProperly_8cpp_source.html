46 PreparedPolygonContainsProperly::containsProperly( const geom::Geometry * geom)
47 {
48  // Do point-in-poly tests first, since they are cheaper and may result
49  // in a quick negative result.
50  // If a point of any test components does not lie in target,
51  // result is false
52  bool isAllInPrepGeomArea = isAllTestComponentsInTargetInterior( geom);
53  if ( !isAllInPrepGeomArea )
54  return false;
55 
56  // If any segments intersect, result is false
57  noding::SegmentString::ConstVect lineSegStr;
58  noding::SegmentStringUtil::extractSegmentStrings( geom, lineSegStr);
59  bool segsIntersect = prepPoly->getIntersectionFinder()->intersects( &lineSegStr);
60 
61  for ( size_t i = 0, ni = lineSegStr.size(); i < ni; i++ ) {
62  delete lineSegStr[ i ];
63  }
64 
65  if (segsIntersect)
66  return false;
67 
73  if ( geom->getGeometryTypeId() == geos::geom::GEOS_MULTIPOLYGON
74  || geom->getGeometryTypeId() == geos::geom::GEOS_POLYGON )
75  {
76  // TODO: generalize this to handle GeometryCollections
77  bool isTargetGeomInTestArea = isAnyTargetComponentInAreaTest( geom, prepPoly->getRepresentativePoints());
