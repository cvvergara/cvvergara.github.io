253 unsigned int
254 MinimumDiameter::findMaxPerpDistance(const CoordinateSequence *pts,
255  LineSegment* seg, unsigned int startIndex)
256 {
257  double maxPerpDistance=seg->distancePerpendicular(pts->getAt(startIndex));
258  double nextPerpDistance = maxPerpDistance;
259  unsigned int maxIndex = startIndex;
260  unsigned int nextIndex = maxIndex;
261  while (nextPerpDistance >= maxPerpDistance) {
262  maxPerpDistance = nextPerpDistance;
263  maxIndex=nextIndex;
264  nextIndex=getNextIndex(pts, maxIndex);
265  nextPerpDistance = seg->distancePerpendicular(pts->getAt(nextIndex));
266  }
267 
268  // found maximum width for this segment - update global min dist if appropriate
269  if (maxPerpDistance < minWidth) {
270  minPtIndex = maxIndex;
271  minWidth = maxPerpDistance;
272  delete minWidthPt;
273  minWidthPt = new Coordinate(pts->getAt(minPtIndex));
274  delete minBaseSeg;
275  minBaseSeg = new LineSegment(*seg);
276 // System.out.println(minBaseSeg);
277 // System.out.println(minWidth);
278  }
279  return maxIndex;
280 }

290 Geometry* MinimumDiameter::getMinimumRectangle()
291 {
292  computeMinimumDiameter();
293 
294  if ( !minBaseSeg || !convexHullPts )
295  {
296  //return empty polygon
297  return inputGeom->getFactory()->createPolygon();
298  }
299 
300  // check if minimum rectangle is degenerate (a point or line segment)
301  if (minWidth == 0.0) {
302  if (minBaseSeg->p0.equals2D(minBaseSeg->p1)) {
303  return inputGeom->getFactory()->createPoint(minBaseSeg->p0);
304  }
305  return minBaseSeg->toGeometry(*inputGeom->getFactory()).release();
306  }
307 
308  // deltas for the base segment of the minimum diameter
309  double dx = minBaseSeg->p1.x - minBaseSeg->p0.x;
310  double dy = minBaseSeg->p1.y - minBaseSeg->p0.y;
311 
312  double minPara = DoubleMax;
313  double maxPara = -DoubleMax;
314  double minPerp = DoubleMax;
315  double maxPerp = -DoubleMax;
316 
317  // compute maxima and minima of lines parallel and perpendicular to base segment
318  std::size_t const n=convexHullPts->getSize();
319  for (std::size_t i = 0; i < n; ++i) {
320 
321  double paraC = computeC(dx, dy, convexHullPts->getAt(i));
322  if (paraC > maxPara) maxPara = paraC;
323  if (paraC < minPara) minPara = paraC;
324 
325  double perpC = computeC(-dy, dx, convexHullPts->getAt(i));
326  if (perpC > maxPerp) maxPerp = perpC;
327  if (perpC < minPerp) minPerp = perpC;
328  }
329 
330  // compute lines along edges of minimum rectangle
331  LineSegment maxPerpLine = computeSegmentForLine(-dx, -dy, maxPerp);
332  LineSegment minPerpLine = computeSegmentForLine(-dx, -dy, minPerp);
333  LineSegment maxParaLine = computeSegmentForLine(-dy, dx, maxPara);
334  LineSegment minParaLine = computeSegmentForLine(-dy, dx, minPara);
335 
336  // compute vertices of rectangle (where the para/perp max & min lines intersect)
337  Coordinate p0, p1, p2, p3;
338  maxParaLine.lineIntersection(maxPerpLine, p0);
339  minParaLine.lineIntersection(maxPerpLine, p1);
340  minParaLine.lineIntersection(minPerpLine, p2);
341  maxParaLine.lineIntersection(minPerpLine, p3);
342 
343  const CoordinateSequenceFactory *csf =
344  inputGeom->getFactory()->getCoordinateSequenceFactory();
345 
346  geom::CoordinateSequence *seq = csf->create(5, 2);
347  seq->setAt(p0, 0);
348  seq->setAt(p1, 1);
349  seq->setAt(p2, 2);
350  seq->setAt(p3, 3);
351  seq->setAt(p0, 4); // close
352 
353  LinearRing* shell = inputGeom->getFactory()->createLinearRing( seq );
354  return inputGeom->getFactory()->createPolygon( shell, nullptr );
355 }

362 LineSegment MinimumDiameter::computeSegmentForLine(double a, double b, double c)
363 {
364  Coordinate p0;
365  Coordinate p1;
366  /*
367  * Line eqn is ax + by = c
368  * Slope is a/b.
369  * If slope is steep, use y values as the inputs
370  */
371  if (fabs(b) > fabs(a) ) {
372  p0 = Coordinate(0.0, c/b);
373  p1 = Coordinate(1.0, c/b - a/b);
374  }
375  else {
376  p0 = Coordinate(c/a, 0.0);
377  p1 = Coordinate(c/a - b/a, 1.0);
378  }
379  return LineSegment(p0, p1);
380 }
