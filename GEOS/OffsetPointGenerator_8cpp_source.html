85 /*private*/
86 void
87 OffsetPointGenerator::computeOffsets(const Coordinate& p0,
88  const Coordinate& p1)
89 {
90  double dx = p1.x - p0.x;
91  double dy = p1.y - p0.y;
92  double len = sqrt(dx * dx + dy * dy);
93 
94  // u is the vector that is the length of the offset,
95  // in the direction of the segment
96  double ux = offsetDistance * dx / len;
97  double uy = offsetDistance * dy / len;
98 
99  double midX = (p1.x + p0.x) / 2;
100  double midY = (p1.y + p0.y) / 2;
101 
102  Coordinate offsetLeft(midX - uy, midY + ux);
103  Coordinate offsetRight(midX + uy, midY - ux);
104 
105  offsetPts->push_back(offsetLeft);
106  offsetPts->push_back(offsetRight);
107 }
