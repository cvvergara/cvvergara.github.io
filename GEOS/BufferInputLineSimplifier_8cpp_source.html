75 bool
76 BufferInputLineSimplifier::deleteShallowConcavities()
77 {
82  unsigned int index = 1;
83 
84  unsigned int midIndex = findNextNonDeletedIndex(index);
85  unsigned int lastIndex = findNextNonDeletedIndex(midIndex);
86 
87  bool isChanged = false;
88  while (lastIndex < inputLine.size())
89  {
90  // test triple for shallow concavity
91  bool isMiddleVertexDeleted = false;
92  if (isDeletable(index, midIndex, lastIndex,
93  distanceTol))
94  {
95  isDeleted[midIndex] = DELETE;
96  isMiddleVertexDeleted = true;
97  isChanged = true;
98  }
99  // move simplification window forward
100  if (isMiddleVertexDeleted)
101  index = lastIndex;
102  else
103  index = midIndex;
104 
105  midIndex = findNextNonDeletedIndex(index);
106  lastIndex = findNextNonDeletedIndex(midIndex);
107  }
108  return isChanged;
109 }

174 BufferInputLineSimplifier::isShallowSampled(const geom::Coordinate& p0,
175  const geom::Coordinate& p2,
176  int i0, int i2,
177  double distanceTol) const
178 {
179  // check every n'th point to see if it is within tolerance
180  int inc = (i2 - i0) / NUM_PTS_TO_CHECK;
181  if (inc <= 0) inc = 1;
182 
183  for (int i = i0; i < i2; i += inc) {
