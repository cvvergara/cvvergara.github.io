43 {

157 /* private */
158 int
159 PointLocator::locate(const Coordinate& p,const Polygon *poly)
160 {
161  if (poly->isEmpty()) return Location::EXTERIOR;
162 
163  const LinearRing *shell=dynamic_cast<const LinearRing *>(poly->getExteriorRing());
164  assert(shell);
165 
166  int shellLoc=locateInPolygonRing(p, shell);
167  if (shellLoc==Location::EXTERIOR) return Location::EXTERIOR;
168  if (shellLoc==Location::BOUNDARY) return Location::BOUNDARY;
169 
170  // now test if the point lies in or on the holes
171  for(size_t i=0, n=poly->getNumInteriorRing(); i<n; ++i)
172  {
173  const LinearRing *hole=dynamic_cast<const LinearRing *>(poly->getInteriorRingN(i));
174  int holeLoc=locateInPolygonRing(p,hole);
175  if (holeLoc==Location::INTERIOR) return Location::EXTERIOR;
176  if (holeLoc==Location::BOUNDARY) return Location::BOUNDARY;
177  }
178  return Location::INTERIOR;
179 }
