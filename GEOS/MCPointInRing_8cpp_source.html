142 void
143 MCPointInRing::testLineSegment(const Coordinate& p, const LineSegment& seg)
144 {
145  double xInt; // x intersection of segment with ray
146  double x1; // translated coordinates
147  double y1;
148  double x2;
149  double y2;
150 
151  /*
152  * Test if segment crosses ray from test point in positive x direction.
153  */
154  const Coordinate& p1 = seg.p0;
155  const Coordinate& p2 = seg.p1;
156  x1 = p1.x - p.x;
157  y1 = p1.y - p.y;
158  x2 = p2.x - p.x;
159  y2 = p2.y - p.y;
160 
161  if (((y1>0)&&(y2<=0)) || ((y2>0)&&(y1<=0)))
162  {
163 
164  /*
165  * segment straddles x axis, so compute intersection.
166  */
167  xInt=RobustDeterminant::signOfDet2x2(x1,y1,x2,y2)/(y2-y1);
168 
169  /*
170  * crosses ray if strictly positive intersection.
171  */
172  if (0.0<xInt) {
173  crossings++;
174  }
175  }
176 }
