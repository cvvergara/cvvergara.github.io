108 void
109 SegmentNodeList::findCollapsesFromExistingVertices(
110  std::vector<size_t>& collapsedVertexIndexes)
111 {
112  if ( edge.size() < 2 ) return; // or we'll never exit the loop below
113 
114  for (size_t i=0, n=edge.size()-2; i<n; ++i)
115  {
116  const Coordinate& p0 = edge.getCoordinate(static_cast<unsigned int>(i));
117  const Coordinate& p2 = edge.getCoordinate(static_cast<unsigned int>(i + 2));
118  if (p0.equals2D(p2)) {
119  // add base of collapse as node
120  collapsedVertexIndexes.push_back(i + 1);
121  }

126 void
127 SegmentNodeList::findCollapsesFromInsertedNodes(
128  std::vector<size_t>& collapsedVertexIndexes)
129 {
130  size_t collapsedVertexIndex;
131 
132  // there should always be at least two entries in the list,
133  // since the endpoints are nodes
134  iterator it = begin();
135  SegmentNode* eiPrev = *it;
136  ++it;
137  for(iterator itEnd=end(); it!=itEnd; ++it)
138  {
139  SegmentNode *ei=*it;
140  bool isCollapsed = findCollapseIndex(*eiPrev, *ei,
141  collapsedVertexIndex);
142  if (isCollapsed)
143  collapsedVertexIndexes.push_back(collapsedVertexIndex);
144 
145  eiPrev = ei;
146  }

149 /* private */
150 bool
151 SegmentNodeList::findCollapseIndex(SegmentNode& ei0, SegmentNode& ei1,
152  size_t& collapsedVertexIndex)
153 {
154  // only looking for equal nodes
155  if (! ei0.coord.equals2D(ei1.coord)) return false;
156 
157  int numVerticesBetween = ei1.segmentIndex - ei0.segmentIndex;
158  if (! ei1.isInterior()) {
159  numVerticesBetween--;
160  }
161 
162  // if there is a single vertex between the two equal nodes,
163  // this is a collapse
164  if (numVerticesBetween == 1) {
165  collapsedVertexIndex = ei0.segmentIndex + 1;
166  return true;
167  }
168  return false;
169 }

242 SegmentString*
243 SegmentNodeList::createSplitEdge(SegmentNode *ei0, SegmentNode *ei1)
244 {
245  assert(ei0);
246  assert(ei1);
247 
248  size_t npts = ei1->segmentIndex - ei0->segmentIndex + 2;
249 
250  const Coordinate &lastSegStartPt=edge.getCoordinate(ei1->segmentIndex);
251 
252  // if the last intersection point is not equal to the its
253  // segment start pt, add it to the points list as well.
254  // (This check is needed because the distance metric is not
255  // totally reliable!)
256 
257  // The check for point equality is 2D only - Z values are ignored
258 
259  // Added check for npts being == 2 as in that case NOT using second point
260  // would mean creating a SegmentString with a single point
261  // FIXME: check with mbdavis about this, ie: is it a bug in the caller ?
262  //
263  bool useIntPt1 = npts == 2 || (ei1->isInterior() || ! ei1->coord.equals2D(lastSegStartPt));
264 
265  if (! useIntPt1) {
266  npts--;
267  }
268 
269  CoordinateSequence *pts = new CoordinateArraySequence(npts);
270  size_t ipt = 0;
271  pts->setAt(ei0->coord, ipt++);
272  for (size_t i=ei0->segmentIndex+1; i<=ei1->segmentIndex; i++)
273  {
274  pts->setAt(edge.getCoordinate(static_cast<unsigned int>(i)),ipt++);
275  }
276  if (useIntPt1) pts->setAt(ei1->coord, ipt++);
277 
278  // SegmentString takes ownership of CoordinateList 'pts'
279  SegmentString *ret = new NodedSegmentString(pts, edge.getData());
280 
281 #if GEOS_DEBUG
282  std::cerr<<" SegmentString created"<<std::endl;
283 #endif
284 
285  return ret;
286 }
