48  if(lines.size() < 2)

51  geom::LineString * line1 = lines.front();

53

55  const geom::CoordinateSequence &cs2 = *line2->getCoordinatesRO();
56 
57  const auto n1 = cs1.size();
58  const auto n2 = cs2.size();
59 
60  // Safety check against bad input to prevent segfaults
61  if(n1==0 || n2==0)
62  return;
63 
64  if (cs1[0] != cs2[n2-1]) return;
65 
66  // Merge the two linestrings
67 
68  CoordinateSequence *ncs = CoordinateSequence::removeRepeatedPoints(&cs2);
69  ncs->add(&cs1, false, true);
70 
71  delete line1;
72  delete line2;
73 
74  LineString * nline = _gf.createLineString(ncs);
75  lines.pop_front();
76  lines.pop_back();
77

178 double distance(const Rectangle & rect,
179  double x1, double y1,
180  double x2, double y2)
181 {
182  double dist = 0;
183 
184  Rectangle::Position pos = rect.position(x1,y1);
185  Rectangle::Position endpos = rect.position(x2,y2);
186 
187  while(true)
188  {
189  // Close up when we have the same edge and the
190  // points are in the correct clockwise order
191  if((pos & endpos) != 0 &&
192  (
193  (x1==rect.xmin() && y2>=y1) ||
194  (y1==rect.ymax() && x2>=x1) ||
195  (x1==rect.xmax() && y2<=y1) ||
196  (y1==rect.ymin() && x2<=x1))
197  )
198  {
199  dist += fabs(x2-x1) + fabs(y2-y1);
200  break;
201  }
202 
203  pos = Rectangle::nextEdge(pos);
204  if(pos & Rectangle::Left)
205  {
206  dist += x1-rect.xmin();
207  x1 = rect.xmin();
208  }
209  else if(pos & Rectangle::Top)
210  {
211  dist += rect.ymax()-y1;
212  y1 = rect.ymax();
213  }
214  else if(pos & Rectangle::Right)
215  {
216  dist += rect.xmax()-x1;
217  x1 = rect.xmax();
218  }
219  else
220  {
221  dist += y1-rect.ymin();
222  y1 = rect.ymin();
223  }
224  }
225  return dist;
226 }

271 void
272 normalize_ring(std::vector<Coordinate> &ring)
273 {
274  if(ring.empty())
275  return;
276 
277  // Find the "smallest" coordinate
278 
279  int best_pos = 0;
280  int n = static_cast<int>(ring.size());
281  for(int pos = 0; pos<n; ++pos)
282  {
283  // TODO: use CoordinateLessThan ?
284  if(ring[pos].x < ring[best_pos].x)
285  best_pos = pos;
286  else if(ring[pos].x == ring[best_pos].x &&
287  ring[pos].y < ring[best_pos].y )
288  best_pos = pos;
289  }
290 
291  // Quick exit if the ring is already normalized
292  if(best_pos == 0)
293  return;
294 
295  // Flip hands -algorithm to the part without the
296  // duplicate last coordinate at n-1:
297 
298  reverse_points(ring,0,best_pos-1);
299  reverse_points(ring,best_pos,n-2);
300  reverse_points(ring,0,n-2);
301 
302  // And make sure the ring is valid by duplicating the first coordinate
303  // at the end:
304 
305  geom::Coordinate c;
306  c = ring[0];
307  ring[n-1] = c;
308 }

320  while(true)

328  (x1==rect.xmax() && y2<=y1) ||
329  (y1==rect.ymin() && x2<=x1))
330  )

342  else if(pos & Rectangle::Right)
343  x1 = rect.xmax();
344  else
345  y1 = rect.ymin();
346 
347  ring->push_back(Coordinate(x1,y1));

369 void
370 RectangleIntersectionBuilder::reconnectPolygons(const Rectangle & rect)
371 {
372  // Build the exterior rings first
373 
374  typedef std::vector< geom::Geometry *> LinearRingVect;
375  typedef std::pair< geom::LinearRing *, LinearRingVect * > ShellAndHoles;
376  typedef std::list< ShellAndHoles > ShellAndHolesList;
377 
378  ShellAndHolesList exterior;
379 
380  const CoordinateSequenceFactory &_csf = *_gf.getCoordinateSequenceFactory();
381 
382  // If there are no lines, the rectangle must have been
383  // inside the exterior ring.
384 
385  if(lines.empty())
386  {
387  geom::LinearRing * ring = rect.toLinearRing(_gf);
388  exterior.push_back(make_pair(ring, new LinearRingVect()));
389  }
390  else
391  {
392  // Reconnect all lines into one or more linearrings
393  // using box boundaries if necessary
394 
395  std::vector<Coordinate> *ring = nullptr;
396 
397  while(!lines.empty() || ring != nullptr)
398  {
399  if(ring == nullptr)
400  {
401  ring = new std::vector<Coordinate>();
402  LineString *line = lines.front();
403  lines.pop_front();
404  line->getCoordinatesRO()->toVector(*ring);
405  delete line;
406  }
407 
408  // Distance to own endpoint
409  double own_distance = distance(rect, *ring);
410 
411  // Find line to connect to
412  // TODO: should we use LineMerge op ?
413  double best_distance = -1;
414  std::list<LineString*>::iterator best_pos = lines.begin();
415  for(std::list<LineString*>::iterator iter=lines.begin(); iter!=lines.end(); ++iter)
416  {
417  double d = distance(rect, *ring, *iter);
418  if(best_distance < 0 || d<best_distance)
419  {
420  best_distance = d;
421  best_pos = iter;
422  }
423  }
424 
425  // If own end point is closest, close the ring and continue
426  if(best_distance < 0 || own_distance < best_distance)
427  {
428  close_ring(rect,ring);
429  normalize_ring(*ring);
430  geom::CoordinateSequence *shell_cs = _csf.create(ring);
431  geom::LinearRing *shell = _gf.createLinearRing(shell_cs);
432  exterior.push_back(make_pair(shell, new LinearRingVect()));
433  ring = nullptr;
434  }
435  else
436  {
437  LineString * line = *best_pos;
438  auto nr = ring->size();
439  const CoordinateSequence& cs = *line->getCoordinatesRO();
440  close_boundary(rect, ring,
441  (*ring)[nr-1].x,
442  (*ring)[nr-1].y,
443  cs[0].x,
444  cs[0].y);
445  // above function adds the 1st point
446  for (size_t i=1; i<cs.size(); ++i)
447  ring->push_back(cs[i]);
448  //ring->addSubLineString(line,1);
449  delete line;
450  lines.erase(best_pos);
451  }
452  }
453  }
454 
455  // Attach holes to polygons
456 
457  for (std::list<geom::Polygon *>::iterator i=polygons.begin(), e=polygons.end(); i!=e; ++i)
458  {
459  geom::Polygon *poly = *i;
460  const geom::LineString *hole = poly->getExteriorRing();
461 
462  if(exterior.size() == 1)
463  {
464  exterior.front().second->push_back( hole->clone() );
465  }
466  else
467  {
468  using geos::algorithm::CGAlgorithms;
469  geom::Coordinate c;
470  hole->getCoordinatesRO()->getAt(0, c);
471  for (ShellAndHolesList::iterator i=exterior.begin(), e=exterior.end(); i!=e; ++i)
472  {
473  ShellAndHoles &p = *i;
474  const CoordinateSequence *shell_cs = p.first->getCoordinatesRO();
475  if( CGAlgorithms::isPointInRing(c, shell_cs) )
476  {
477  // add hole to shell
478  p.second->push_back(hole->clone());
479  break;
480  }
481  }
482  }
483 
484  delete poly;
485  }
486 
487  // Build the result polygons
488 
489  std::list<geom::Polygon *> new_polygons;
490  for (ShellAndHolesList::iterator i=exterior.begin(), e=exterior.end(); i!=e; ++i)
491  {
492  ShellAndHoles &p = *i;
493  geom::Polygon * poly = _gf.createPolygon(p.first, p.second);
494  new_polygons.push_back(poly);
495  }
496 
497  clear();
498  polygons = new_polygons;
499 }
