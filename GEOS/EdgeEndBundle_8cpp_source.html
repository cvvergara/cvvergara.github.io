73 void EdgeEndBundle::computeLabel(
74  const algorithm::BoundaryNodeRule& boundaryNodeRule)
75 {
76  // create the label. If any of the edges belong to areas,
77  // the label must be an area label
78  bool isArea=false;
79 
80  for(vector<EdgeEnd*>::iterator it=edgeEnds->begin(), itEnd=edgeEnds->end();
81  it != itEnd; it++)
82  {
83  EdgeEnd *e=*it;
84  if (e->getLabel().isArea()) isArea=true;
85  }
86  if (isArea) {
87  label = Label(Location::UNDEF,Location::UNDEF,Location::UNDEF);
88  } else {
89  label = Label(Location::UNDEF);
90  }
91  // compute the On label, and the side labels if present
92  for(int i=0;i<2;i++) {
93  computeLabelOn(i, boundaryNodeRule);
94  if (isArea)
95  computeLabelSides(i);
96  }
97 }

101 EdgeEndBundle::computeLabelOn(int geomIndex, const algorithm::BoundaryNodeRule& boundaryNodeRule)
102 {
103  // compute the ON location value
104  int boundaryCount=0;
105  bool foundInterior=false;
106 
107  for(vector<EdgeEnd*>::iterator it=edgeEnds->begin();it<edgeEnds->end();it++) {
108  EdgeEnd *e=*it;
109  int loc=e->getLabel().getLocation(geomIndex);
110  if (loc==Location::BOUNDARY) boundaryCount++;
111  if (loc==Location::INTERIOR) foundInterior=true;
112  }
113  int loc=Location::UNDEF;
114  if (foundInterior) loc=Location::INTERIOR;
115  if (boundaryCount>0) {
116  loc = GeometryGraph::determineBoundary(boundaryNodeRule,
117  boundaryCount);
118  }
119  label.setLocation(geomIndex,loc);
120 }
