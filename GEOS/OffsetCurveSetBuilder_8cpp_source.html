106 {

108  std::cerr<<__FUNCTION__<<": coords="<<coord->toString()<<std::endl;

110  // don't add null curves!
111  if (coord->getSize() < 2) {
112 #if GEOS_DEBUG

115  delete coord;
116  return;
117  }
118 
119  // add the edge for a coordinate list which is a raw offset curve
120  Label *newlabel = new Label(0, Location::BOUNDARY, leftLoc, rightLoc);
121 
122  // coord ownership transferred to SegmentString

124

127  newLabels.push_back(newlabel);
128  curveList.push_back(e);

215 /*private*/
216 void
217 OffsetCurveSetBuilder::addPolygon(const Polygon *p)
218 {
219  double offsetDistance=distance;
220 
221  int offsetSide=Position::LEFT;
222  if (distance < 0.0)
223  {
224  offsetDistance = -distance;
225  offsetSide = Position::RIGHT;
226  }
227 
228  // FIXME: avoid the C-style cast
229  const LinearRing *shell=(const LinearRing *)p->getExteriorRing();
230 
231  // optimization - don't bother computing buffer
232  // if the polygon would be completely eroded
233  if (distance < 0.0 && isErodedCompletely(shell, distance))
234  {
235 #if GEOS_DEBUG
236  std::cerr<<__FUNCTION__<<": polygon is eroded completely "<<std::endl;
237 #endif
238  return;
239  }
240 
241  // don't attempt to buffer a polygon
242  // with too few distinct vertices
243  CoordinateSequence *shellCoord =
244  CoordinateSequence::removeRepeatedPoints(shell->getCoordinatesRO());
245  if (distance <= 0.0 && shellCoord->size() < 3)
246  {
247  delete shellCoord;
248  return;
249  }
250 
251  addPolygonRing(
252  shellCoord,
253  offsetDistance,
254  offsetSide,
255  Location::EXTERIOR,
256  Location::INTERIOR);
257 
258  delete shellCoord;
259 
260  for (size_t i=0, n=p->getNumInteriorRing(); i<n; ++i)
261  {
262  const LineString *hls=p->getInteriorRingN(i);
263  assert(dynamic_cast<const LinearRing *>(hls));
264  const LinearRing *hole=static_cast<const LinearRing *>(hls);
265 
266  // optimization - don't bother computing buffer for this hole
267  // if the hole would be completely covered
268  if (distance > 0.0 && isErodedCompletely(hole, -distance))
269  {
270  continue;
271  }
272 
273  CoordinateSequence *holeCoord =
274  CoordinateSequence::removeRepeatedPoints(hole->getCoordinatesRO());
275 
276  // Holes are topologically labelled opposite to the shell,
277  // since the interior of the polygon lies on their opposite
278  // side (on the left, if the hole is oriented CCW)
279  addPolygonRing(
280  holeCoord,
281  offsetDistance,
282  Position::opposite(offsetSide),
283  Location::INTERIOR,
284  Location::EXTERIOR);
285 
286  delete holeCoord;
287  }

290 /* private */
291 void
292 OffsetCurveSetBuilder::addPolygonRing(const CoordinateSequence *coord,
293  double offsetDistance, int side, int cwLeftLoc, int cwRightLoc)
294 {
295 
296  // don't bother adding ring if it is "flat" and
297  // will disappear in the output
298  if (offsetDistance == 0.0 && coord->size() < LinearRing::MINIMUM_VALID_SIZE)
299  return;
300 
301  int leftLoc=cwLeftLoc;
302  int rightLoc=cwRightLoc;
303 #if GEOS_DEBUG
304  std::cerr<<"OffsetCurveSetBuilder::addPolygonRing: CCW: "<<CGAlgorithms::isCCW(coord)<<std::endl;
305 #endif
306  if (coord->size() >= LinearRing::MINIMUM_VALID_SIZE
307  && CGAlgorithms::isCCW(coord))
308  {
309  leftLoc=cwRightLoc;
310  rightLoc=cwLeftLoc;
311 #if GEOS_DEBUG
312  std::cerr<<" side "<<side<<" becomes "<<Position::opposite(side)<<std::endl;
313 #endif
314  side=Position::opposite(side);
315  }
316  std::vector<CoordinateSequence*> lineList;
317  curveBuilder.getRingCurve(coord, side, offsetDistance, lineList);
318  addCurves(lineList, leftLoc, rightLoc);
319 }

322 bool
323 OffsetCurveSetBuilder::isErodedCompletely(const LinearRing *ring,
324  double bufferDistance)
325 {
326  const CoordinateSequence *ringCoord = ring->getCoordinatesRO();
327 
328  // degenerate ring has no area
329  if (ringCoord->getSize() < 4)
330  return bufferDistance < 0;
331 
332  // important test to eliminate inverted triangle bug
333  // also optimizes erosion test for triangles
334  if (ringCoord->getSize() == 4)
335  return isTriangleErodedCompletely(ringCoord, bufferDistance);
336 
337  const Envelope* env = ring->getEnvelopeInternal();
338  double envMinDimension = std::min(env->getHeight(), env->getWidth());
339  if (bufferDistance < 0.0 && 2 * std::abs(bufferDistance) > envMinDimension)
340  return true;
341 
355 /* MD 7 Feb 2005 - there's an unknown bug in the MD code,
356  so disable this for now */
357 #if 0
358  MinimumDiameter md(ring); //=new MinimumDiameter(ring);
359  double minDiam = md.getLength();
360  return minDiam < (2 * std::fabs(bufferDistance));
361 #endif
362 
363  return false;
364 }
