147 bool
148 IsSimpleOp::isSimpleMultiPoint(const MultiPoint& mp)
149 {
150  if (mp.isEmpty()) return true;
151  set<const Coordinate*, CoordinateLessThen> points;
152 
153  for (std::size_t i=0, n=mp.getNumGeometries(); i<n; ++i)
154  {
155  const Point *pt = dynamic_cast<const Point*>(mp.getGeometryN(i));
156  assert(pt);
157  const Coordinate *p=pt->getCoordinate();
158  if (points.find(p) != points.end())
159  {
160  nonSimpleLocation.reset(new Coordinate(*p));
161  return false;
162  }
163  points.insert(p);
164  }
165  return true;
166 }

169 IsSimpleOp::isSimpleLinearGeometry(const Geometry *geom)
170 {
171  if (geom->isEmpty()) return true;
172  GeometryGraph graph(0,geom);
173  LineIntersector li;
174  std::unique_ptr<SegmentIntersector> si (graph.computeSelfNodes(&li,true));
175 
176  // if no self-intersection, must be simple
177  if (!si->hasIntersection()) return true;
178 
179  if (si->hasProperIntersection())
180  {
181  nonSimpleLocation.reset(
182  new Coordinate(si->getProperIntersectionPoint())
183  );
184  return false;
185  }
186 
187  if (hasNonEndpointIntersection(graph)) return false;
188 
189  if ( isClosedEndpointsInInterior ) {
190  if (hasClosedEndpointIntersection(graph)) return false;
191  }
192 
193  return true;
