252 /*protected*/
253 Envelope::Ptr
254 LineString::computeEnvelopeInternal() const
255 {
256  if (isEmpty()) {
257  // We don't return NULL here
258  // as it would indicate "unknown"
259  // envelope. In this case we
260  // *know* the envelope is EMPTY.
261  return Envelope::Ptr(new Envelope());
262  }
263 
264  assert(points.get());
265  const Coordinate& c=points->getAt(0);
266  double minx = c.x;
267  double miny = c.y;
268  double maxx = c.x;
269  double maxy = c.y;
270  std::size_t npts=points->getSize();
271  for (std::size_t i=1; i<npts; i++) {
272  const Coordinate &c=points->getAt(i);
273  minx = minx < c.x ? minx : c.x;
274  maxx = maxx > c.x ? maxx : c.x;
275  miny = miny < c.y ? miny : c.y;
276  maxy = maxy > c.y ? maxy : c.y;
277  }
278 
279  // caller expects a newly allocated Envelope.
280  // this function won't be called twice, unless
281  // cached Envelope is invalidated (set to NULL)
282  return Envelope::Ptr(new Envelope(minx, maxx, miny, maxy));
283 }

333 void
334 LineString::normalize()
335 {
336  assert(points.get());
337  std::size_t npts=points->getSize();
338  std::size_t n=npts/2;
339  for (std::size_t i=0; i<n; i++) {
340  std::size_t j = npts - 1 - i;
341  if (!(points->getAt(i)==points->getAt(j))) {
342  if (points->getAt(i).compareTo(points->getAt(j)) > 0) {
343  CoordinateSequence::reverse(points.get());
344  }
345  return;
346  }
347  }

350 int
351 LineString::compareToSameClass(const Geometry *ls) const
352 {
353  const LineString *line=dynamic_cast<const LineString*>(ls);
354  assert(line);
355  // MD - optimized implementation
356  std::size_t mynpts=points->getSize();
357  std::size_t othnpts=line->points->getSize();
358  if ( mynpts > othnpts ) return 1;
359  if ( mynpts < othnpts ) return -1;
360  for (std::size_t i=0; i<mynpts; i++)

363  if (cmp) return cmp;
