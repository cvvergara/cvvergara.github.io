505 Geometry::intersection(const Geometry *other) const
506 {
511  // special case: if one input is empty ==> empty
512  if (isEmpty() || other->isEmpty() )
513  {
514  return getFactory()->createGeometryCollection();
515  }
516

531  return RectangleIntersection::clip(*this, rect).release();
532  }
533 #endif
534 
535  return BinaryOp(this, other, overlayOp(OverlayOp::opINTERSECTION)).release();

539 Geometry::Union(const Geometry *other) const
540  //throw(TopologyException *, IllegalArgumentException *)
541 {
542 
543  // special case: if one input is empty ==> other input
544  if ( isEmpty() ) return other->clone();
545  if ( other->isEmpty() ) return clone();
546 
547  Geometry *out = nullptr;
548 
549 #ifdef SHORTCIRCUIT_PREDICATES
550  // if envelopes are disjoint return a MULTI geom or
551  // a geometrycollection
552  if ( ! getEnvelopeInternal()->intersects(other->getEnvelopeInternal()) )
553  {
554 //cerr<<"SHORTCIRCUITED-UNION engaged"<<endl;
555  const GeometryCollection *coll;
556 
557  size_t ngeomsThis = getNumGeometries();
558  size_t ngeomsOther = other->getNumGeometries();
559 
560  // Allocated for ownership transfer
561  vector<Geometry *> *v = new vector<Geometry *>();
562  v->reserve(ngeomsThis+ngeomsOther);
563 
564 
565  if ( nullptr != (coll = dynamic_cast<const GeometryCollection *>(this)) )
566  {
567  for (size_t i=0; i<ngeomsThis; ++i)
568  v->push_back(coll->getGeometryN(i)->clone());
569  } else {
570  v->push_back(this->clone());
571  }
572 
573  if ( nullptr != (coll = dynamic_cast<const GeometryCollection *>(other)) )
574  {
575  for (size_t i=0; i<ngeomsOther; ++i)
576  v->push_back(coll->getGeometryN(i)->clone());
577  } else {
578  v->push_back(other->clone());
579  }
580 
581  out = _factory->buildGeometry(v);
582  return out;
583  }
584 #endif
585
