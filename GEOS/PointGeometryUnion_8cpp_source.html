39 PointGeometryUnion::Union() const
40 {
41  using namespace geom;
42  using algorithm::PointLocator;
43  using geom::util::GeometryCombiner;
44 
45  PointLocator locater;
46  // use a set to eliminate duplicates, as required for union
47  std::set<Coordinate> exteriorCoords;
48 
49  for (std::size_t i=0, n=pointGeom.getNumGeometries(); i<n; ++i) {
50  const Point* point = dynamic_cast<const Point*>(pointGeom.getGeometryN(i));
51  assert(point);
52  const Coordinate* coord = point->getCoordinate();
53  int loc = locater.locate(*coord, &otherGeom);
54  if (loc == Location::EXTERIOR)
55  exteriorCoords.insert(*coord);
56  }
57 
58  // if no points are in exterior, return the other geom
59  if (exteriorCoords.empty())
60  return std::unique_ptr<Geometry>(otherGeom.clone());
61 
62  // make a puntal geometry of appropriate size
63  std::unique_ptr<Geometry> ptComp;
64 
65  if (exteriorCoords.size() == 1) {
66  ptComp.reset( geomFact->createPoint(*(exteriorCoords.begin())) );
67  }
68  else
69  {
70  std::vector<Coordinate> coords(exteriorCoords.size());
71  std::copy(exteriorCoords.begin(), exteriorCoords.end(), coords.begin());
72  ptComp.reset( geomFact->createMultiPoint(coords) );
73  }
74 
75  // add point component to the other geometry
76  return std::unique_ptr<Geometry> (
77  GeometryCombiner::combine(ptComp.get(), &otherGeom)
78  );
79 }
