101  double vertexDistance(const FacetSequence* fs1, const FacetSequence* fs2) {
102  for (size_t i1 = 0; i1 < fs1->size(); i1++) {
103  for (size_t i2 = 0; i2 < fs2->size(); i2++) {
104  const Coordinate* p1 = fs1->getCoordinate(i1);
105  const Coordinate* p2 = fs2->getCoordinate(i2);
106  if (!p1->equals2D(*p2)) {
107  double d = p1->distance(*p2);
108  if (d < minDist) {
109  minDist = d;
110  minPts[0] = *p1;
111  minPts[1] = *p2;
112  if (d == 0.0)
