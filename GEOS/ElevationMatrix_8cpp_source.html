153 ElevationMatrixCell &
154 ElevationMatrix::getCell(const Coordinate &c)
155 {
156  int col, row;
157 
158  if ( ! cellwidth ) col=0;
159  else
160  {
161  double xoffset = c.x - env.getMinX();
162  col = (int)(xoffset/cellwidth);
163  if ( col == (int)cols ) col = cols-1;
164  }
165  if ( ! cellheight ) row=0;
166  else
167  {
168  double yoffset = c.y - env.getMinY();
169  row = (int)(yoffset/cellheight);
170  if ( row == (int)rows ) row = rows-1;
171  }
172  int celloffset = (cols*row)+col;
173 
174  if (celloffset<0 || celloffset >= (int)(cols*rows))
175  {
176  ostringstream s;
177  s<<"ElevationMatrix::getCell got a Coordinate out of grid extent ("<<env.toString()<<") - cols:"<<cols<<" rows:"<<rows;
178  throw util::IllegalArgumentException(s.str());
179  }
180 
181  return cells[celloffset];
182 }

191 double
192 ElevationMatrix::getAvgElevation() const
193 {
194  if ( avgElevationComputed ) return avgElevation;
195  double ztot=0;
196  int zvals=0;
197  for (unsigned int r=0; r<rows; r++)
198  {
199  for(unsigned int c=0; c<cols; c++)
200  {
201  const ElevationMatrixCell &cell = cells[(r*cols)+c];
202  double e = cell.getAvg();
203  if ( !ISNAN(e) )
204  {
205  zvals++;
206  ztot+=e;
207  }
208  }
209  }
210  if ( zvals ) avgElevation = ztot/zvals;
211  else avgElevation = DoubleNotANumber;
212 
213  avgElevationComputed = true;
214 
215  return avgElevation;
216 }
