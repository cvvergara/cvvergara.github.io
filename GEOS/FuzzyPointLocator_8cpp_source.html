61 FuzzyPointLocator::extractLineWork(const geom::Geometry& geom)
62 {
63  ::geos::ignore_unused_variable_warning(geom);
64 
65  vector<Geometry*>* lineGeoms = new vector<Geometry*>();
66  try { // geoms array will leak if an exception is thrown
67 
68  for (size_t i=0, n=g.getNumGeometries(); i<n; ++i)
69  {
70  const Geometry* gComp = g.getGeometryN(i);
71  Geometry* lineGeom = nullptr;
72 
73  // only get linework for polygonal components
74  if (gComp->getDimension() == 2) {
75  lineGeom = gComp->getBoundary();
76  lineGeoms->push_back(lineGeom);
77  }
78  }
79  return std::unique_ptr<Geometry>(g.getFactory()->buildGeometry(lineGeoms));
80 
81  } catch (...) { // avoid leaks
82  for (size_t i=0, n=lineGeoms->size(); i<n; ++i)
83  {
84  delete (*lineGeoms)[i];
85  }
86  delete lineGeoms;
87  throw;
88  }

128 Location::Value
129 FuzzyPointLocator::getLocation(const Coordinate& pt)
130 {
131  unique_ptr<Geometry> point(g.getFactory()->createPoint(pt));
132 
133  double dist = linework->distance(point.get());
134 
135  // if point is close to boundary, it is considered
136  // to be on the boundary
137  if (dist < tolerance)
138  return Location::BOUNDARY;
139 
140  // now we know point must be clearly inside or outside geometry,
141  // so return actual location value
142 
143  // (the static_cast is needed because PointLocator doesn't cleanly
144  // return a Location::Value - it should !!)
145  return static_cast<Location::Value>(ptLocator.locate(pt, &g));
