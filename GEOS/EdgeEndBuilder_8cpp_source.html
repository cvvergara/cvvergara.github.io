53 EdgeEndBuilder::computeEdgeEnds(Edge *edge, vector<EdgeEnd*> *l)
54 {
55  EdgeIntersectionList &eiList=edge->getEdgeIntersectionList();
56  //Debug.print(eiList);
57  // ensure that the list has entries for the first and last point of the edge
58  eiList.addEndpoints();
59 
60  EdgeIntersectionList::iterator it=eiList.begin();
61  // no intersections, so there is nothing to do
62  if (it==eiList.end()) return;
63 
64  EdgeIntersection *eiPrev=nullptr;
65  EdgeIntersection *eiCurr=nullptr;
66 
67  EdgeIntersection *eiNext=*it;
68  it++;
69  do {
70  eiPrev=eiCurr;

72  eiNext=nullptr;
73  if (it!=eiList.end()) {
74  eiNext=*it;
75  it++;
76  }
77  if (eiCurr!=nullptr) {
78  createEdgeEndForPrev(edge,l,eiCurr,eiPrev);
79  createEdgeEndForNext(edge,l,eiCurr,eiNext);
