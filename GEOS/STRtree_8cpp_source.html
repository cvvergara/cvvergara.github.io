40 static bool yComparator(Boundable *a, Boundable *b)
41 {
42  assert(a);
43  assert(b);
44  const void* aBounds = a->getBounds();
45  const void* bBounds = b->getBounds();
46  assert(aBounds);
47  assert(bBounds);
48  const Envelope* aEnv = static_cast<const Envelope*>(aBounds);
49  const Envelope* bEnv = static_cast<const Envelope*>(bBounds);
50 
51  // NOTE - mloskot:
52  // The problem of instability is directly related to mathematical definition of
53  // "strict weak ordering" as a fundamental requirement for binary predicate:
54  //
55  // if a is less than b then b is not less than a,
56  // if a is less than b and b is less than c
57  // then a is less than c,
58  // and so on.
59  //
60  // For some very low values, this predicate does not fulfill this requiremnet,
61 
62  // NOTE - strk:
63  // It seems that the '<' comparison here gives unstable results.
64  // In particular, when inlines are on (for Envelope::getMinY and getMaxY)
65  // things are fine, but when they are off we can even get a memory corruption !!
66  //return STRtree::centreY(aEnv) < STRtree::centreY(bEnv);
67 
68  // NOTE - mloskot:
69  // This comparison does not answer if a is "lower" than b
70  // what is required for sorting. This comparison only answeres
71  // if a and b are "almost the same" or different
72 
73  /*NOTE - cfis
74  In debug mode VC++ checks the predicate in both directions.
75 
76  If !_Pred(_Left, _Right)
77  Then an exception is thrown if _Pred(_Right, _Left).
78  See xutility around line 320:
79 
80  bool __CLRCALL_OR_CDECL _Debug_lt_pred(_Pr _Pred, _Ty1& _Left, _Ty2& _Right,
81  const wchar_t *_Where, unsigned int _Line)*/
82 
83  //return std::fabs( STRtree::centreY(aEnv) - STRtree::centreY(bEnv) ) < 1e-30
84 
85  // NOTE - strk:
86  // See http://trac.osgeo.org/geos/ticket/293
87  // as for why simple comparison (<) isn't used here
88  return AbstractSTRtree::compareDoubles(STRtree::centreY(aEnv),
89  STRtree::centreY(bEnv));
90 }

165 STRtree::verticalSlices(BoundableList* childBoundables, size_t sliceCount)
166 {
167  size_t sliceCapacity = (size_t) ceil((double)childBoundables->size() / (double) sliceCount);
168  vector<BoundableList*>* slices = new vector<BoundableList*>(sliceCount);
169 
170  size_t i=0, nchilds=childBoundables->size();
171 
172  for (size_t j=0; j<sliceCount; j++)
173  {
174  (*slices)[j]=new BoundableList();
175  (*slices)[j]->reserve(sliceCapacity);
176  size_t boundablesAddedToSlice = 0;
177  while (i<nchilds && boundablesAddedToSlice<sliceCapacity)
178  {
179  Boundable *childBoundable=(*childBoundables)[i];
180  ++i;
181  (*slices)[j]->push_back(childBoundable);
182  ++boundablesAddedToSlice;
183  }
184  }
185  return slices;
186 }

212 std::pair<const void*, const void*> STRtree::nearestNeighbour(BoundablePair* initBndPair, double maxDistance) {
213  double distanceLowerBound = maxDistance;
214  BoundablePair* minPair = nullptr;
215 
216  BoundablePair::BoundablePairQueue priQ;
217  priQ.push(initBndPair);
218 
219  while(!priQ.empty() && distanceLowerBound > 0.0) {
220  BoundablePair* bndPair = priQ.top();
221  double currentDistance = bndPair->getDistance();
222 
230  if (minPair && currentDistance >= distanceLowerBound)
231  break;
232 
233  priQ.pop();
234 
242  if (bndPair->isLeaves()) {
243  distanceLowerBound = currentDistance;
244  minPair = bndPair;
245  } else {
251  bndPair->expandToQueue(priQ, distanceLowerBound);
252  }
253 
254  if (bndPair != initBndPair && bndPair != minPair)
255  delete bndPair;
256  }
257 
258  /* Free any remaining BoundablePairs in the queue */
259  while(!priQ.empty()) {
260  BoundablePair* bndPair = priQ.top();
261  priQ.pop();
262  if (bndPair != initBndPair)
263  delete bndPair;
264  }
265 
266  if (!minPair)
267  throw util::GEOSException("Error computing nearest neighbor");
268 
269  const void* item0 = dynamic_cast<const ItemBoundable*>(minPair->getBoundable(0))->getItem();
270  const void* item1 = dynamic_cast<const ItemBoundable*>(minPair->getBoundable(1))->getItem();
271  if (minPair != initBndPair)
272  delete minPair;
273 
274  return std::pair<const void*, const void*>(item0, item1);
275 }
