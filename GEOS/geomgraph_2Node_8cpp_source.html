117 Node::isIncidentEdgeInResult() const
118 {
119  testInvariant();
120 
121  if (!edges) return false;
122 
123  EdgeEndStar::iterator it=edges->begin();
124  EdgeEndStar::iterator endIt=edges->end();

127  assert(*it);

242 void
243 Node::addZ(double z)
244 {
245 #if GEOS_DEBUG
246  cerr<<"["<<this<<"] Node::addZ("<<z<<")";
247 #endif
248  if ( ISNAN(z) )
249  {
250 #if GEOS_DEBUG
251  cerr<<" skipped"<<endl;
252 #endif
253  return;
254  }
255  if ( find(zvals.begin(), zvals.end(), z) != zvals.end() )
256  {
257 #if GEOS_DEBUG
258  cerr<<" already stored"<<endl;
259 #endif
260  return;
261  }
262  zvals.push_back(z);
263  ztot+=z;
264  coord.z = ztot / static_cast<double>(zvals.size());
265 #if GEOS_DEBUG
