156 /*public static*/
157 double
158 GeometrySnapper::computeOverlaySnapTolerance(const geom::Geometry& g)
159 {
160  double snapTolerance = computeSizeBasedSnapTolerance(g);
161 
171  assert(g.getPrecisionModel());
172  const PrecisionModel& pm = *(g.getPrecisionModel());
173  if (pm.getType() == PrecisionModel::FIXED)
174  {
175  double fixedSnapTol = (1 / pm.getScale()) * 2 / 1.415;
176  if ( fixedSnapTol > snapTolerance )
177  snapTolerance = fixedSnapTol;
178  }
179  return snapTolerance;
180 }
