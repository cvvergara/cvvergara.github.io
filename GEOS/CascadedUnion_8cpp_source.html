84 geom::Geometry* CascadedUnion::binaryUnion(GeometryListHolder* geoms,
85  std::size_t start, std::size_t end)
86 {
87  if (end - start <= 1) {
88  return unionSafe(geoms->getGeometry(start), nullptr);
89  }
90  else if (end - start == 2) {
91  return unionSafe(geoms->getGeometry(start), geoms->getGeometry(start + 1));
