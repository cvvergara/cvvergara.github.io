41 /*public static*/
42 Envelope*
43 Quadtree::ensureExtent(const Envelope *itemEnv, double minExtent)
44 {
45  //The names "ensureExtent" and "minExtent" are misleading -- sounds like
46  //this method ensures that the extents are greater than minExtent.
47  //Perhaps we should rename them to "ensurePositiveExtent" and "defaultExtent".
48  //[Jon Aquino]
49  double minx=itemEnv->getMinX();
50  double maxx=itemEnv->getMaxX();
51  double miny=itemEnv->getMinY();
52  double maxy=itemEnv->getMaxY();
53  // has a non-zero extent
54  if (minx!=maxx && miny!=maxy) return (Envelope *)itemEnv;
55  // pad one or both extents
56  if (minx==maxx) {
57  minx=minx-minExtent/2.0;
58  maxx=minx+minExtent/2.0;
59  }
60  if (miny==maxy) {
61  miny=miny-minExtent/2.0;
62  maxy=miny+minExtent/2.0;
63  }
64  Envelope *newEnv = new Envelope(minx, maxx, miny, maxy);
65  return newEnv;
66 }
