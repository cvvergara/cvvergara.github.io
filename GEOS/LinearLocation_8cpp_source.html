44 /* public static */
45 Coordinate
46 LinearLocation::pointAlongSegmentByFraction(const Coordinate& p0, const Coordinate& p1, double frac)
47 {
48  if (frac <= 0.0) return p0;
49  if (frac >= 1.0) return p1;
50 
51  double x = (p1.x - p0.x) * frac + p0.x;
52  double y = (p1.y - p0.y) * frac + p0.y;
53  // interpolate Z value. If either input Z is NaN, result z will be NaN as well.
54  double z = (p1.z - p0.z) * frac + p0.z;
55  return Coordinate(x, y, z);
56 }

116 /* public */
117 void
118 LinearLocation::snapToVertex(const Geometry* linearGeom, double minDistance)
119 {
120  if (segmentFraction <= 0.0 || segmentFraction >= 1.0)
121  return;
122  double segLen = getSegmentLength(linearGeom);
123  double lenToStart = segmentFraction * segLen;
124  double lenToEnd = segLen - lenToStart;
125  if (lenToStart <= lenToEnd && lenToStart < minDistance)
126  {
127  segmentFraction = 0.0;
128  }
129  else if (lenToEnd <= lenToStart && lenToEnd < minDistance)
130  {
131  segmentFraction = 1.0;
132  }
133 }

191 LinearLocation::getCoordinate(const Geometry* linearGeom) const
192 {
193  const LineString* lineComp = dynamic_cast<const LineString *> (linearGeom->getGeometryN(componentIndex));
194  if ( ! lineComp ) {

206 LinearLocation::getSegment(const Geometry* linearGeom) const
207 {
208  const LineString* lineComp = dynamic_cast<const LineString *> (linearGeom->getGeometryN(componentIndex));
209  Coordinate p0 = lineComp->getCoordinateN(segmentIndex);

217  return std::unique_ptr<LineSegment>(new LineSegment(p0, p1));

223 {

228  if (segmentIndex > lineComp->getNumPoints())
229  return false;

234  return false;
