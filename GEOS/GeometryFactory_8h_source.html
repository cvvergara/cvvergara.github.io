322  std::unique_ptr<Geometry> buildGeometry(T from, T toofar) const
323  {
324  bool isHeterogeneous = false;
325  size_t count = 0;
326  int geomClass = -1;
327  for (T i=from; i != toofar; ++i)
328  {
329  ++count;
330  const Geometry* g = *i;
331  if ( geomClass < 0 ) {
332  geomClass = g->getClassSortIndex();
333  }
334  else if ( geomClass != g->getClassSortIndex() ) {
335  isHeterogeneous = true;
336  }
337  }
338 
339  // for the empty geometry, return an empty GeometryCollection
340  if ( count == 0 ) {
341  return std::unique_ptr<Geometry>( createGeometryCollection() );
342  }
343 
344  // for the single geometry, return a clone
345  if ( count == 1 ) {
346  return std::unique_ptr<Geometry>( (*from)->clone() );
347  }
348 
349  // Now we know it is a collection
350 
351  // FIXME:
352  // Until we tweak all the createMulti* interfaces
353  // to support taking iterators we'll have to build
354  // a custom vector here.
355  std::vector<Geometry*> fromGeoms;
356  for (T i=from; i != toofar; ++i) {
357  const Geometry* g = *i;
358  fromGeoms.push_back(const_cast<Geometry*>(g));
359  }
360 
361 
362  // for an heterogeneous ...
363  if ( isHeterogeneous ) {
364  return std::unique_ptr<Geometry>( createGeometryCollection(fromGeoms) );
365  }
366 
367  // At this point we know the collection is not hetereogenous.
368  if ( dynamic_cast<const Polygon*>(*from) ) {
369  return std::unique_ptr<Geometry>( createMultiPolygon(fromGeoms) );
370  } else if ( dynamic_cast<const LineString*>(*from) ) {
371  return std::unique_ptr<Geometry>( createMultiLineString(fromGeoms) );
372  } else if ( dynamic_cast<const Point*>(*from) ) {
373  return std::unique_ptr<Geometry>( createMultiPoint(fromGeoms) );
374  }
375  // FIXME: Why not to throw an exception? --mloskot
376  assert(0); // buildGeomtry encountered an unkwnon geometry type
377  return std::unique_ptr<Geometry>(); // nullptr
378  }
