174  return GeometryTransformer::transformCoordinates(coords, parent);

231 /*public*/
232 void
233 LineStringMapBuilderFilter::filter_ro(const Geometry* geom)
234 {
235  TaggedLineString* taggedLine;
236 
237  if ( const LineString* ls =
238  dynamic_cast<const LineString*>(geom) )
239  {
240  int minSize = ls->isClosed() ? 4 : 2;
241  taggedLine = new TaggedLineString(ls, minSize);
242  }
243  else
244  {
245  return;
246  }
247 
248  // Duplicated Geometry pointers shouldn't happen
249  if ( ! linestringMap.insert(std::make_pair(geom, taggedLine)).second )
250  {
251  std::cerr << __FILE__ << ":" << __LINE__
252  << "Duplicated Geometry components detected"
253  << std::endl;
254 
255  delete taggedLine;
256  }
257 }
