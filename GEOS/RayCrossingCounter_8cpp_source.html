93 void
94 RayCrossingCounter::countSegment(const geom::Coordinate& p1,
95  const geom::Coordinate& p2)
96 {
97  // For each segment, check if it crosses
98  // a horizontal ray running from the test point in
99  // the positive x direction.
100 
101  // check if the segment is strictly to the left of the test point
102  if (p1.x < point.x && p2.x < point.x)
103  return;
104 
105  // check if the point is equal to the current ring vertex
106  if (point.x == p2.x && point.y == p2.y)
107  {
108  isPointOnSegment = true;
109  return;
110  }
111 
112  // For horizontal segments, check if the point is on the segment.
113  // Otherwise, horizontal segments are not counted.
114  if (p1.y == point.y && p2.y == point.y)
115  {
116  double minx = p1.x;
117  double maxx = p2.x;
118 
119  if (minx > maxx)
120  {
121  minx = p2.x;
122  maxx = p1.x;
123  }
124 
125  if (point.x >= minx && point.x <= maxx)
126  isPointOnSegment = true;
127 
128  return;
129  }
130 
131  // Evaluate all non-horizontal segments which cross a horizontal ray
132  // to the right of the test pt.
133  // To avoid double-counting shared vertices, we use the convention that
134  // - an upward edge includes its starting endpoint, and excludes its
135  // final endpoint
136  // - a downward edge excludes its starting endpoint, and includes its
137  // final endpoint
138  if (((p1.y > point.y) && (p2.y <= point.y)) ||
139  ((p2.y > point.y) && (p1.y <= point.y)) )
140  {
141  // For an upward edge, orientationIndex will be positive when p1->p2
142  // crosses ray. Conversely, downward edges should have negative sign.
143  int sign = orientationIndex(p1, p2, point);
144  if (sign == 0)
145  {
146  isPointOnSegment = true;
147  return;
148  }
149 
150  if (p2.y < p1.y)
151  sign = -sign;
152 
153  // The segment crosses the ray if the sign is strictly positive.
154  if (sign > 0)
155  crossingCount++;
156  }
157 }
