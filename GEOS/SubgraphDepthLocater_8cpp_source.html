50  * A segment from a directed edge which has been assigned a depth value

53 class DepthSegment {

57  geom::LineSegment upwardSeg;

59  /*
60  * Compare two collinear segments for left-most ordering.
61  * If segs are vertical, use vertical ordering for comparison.
62  * If segs are equal, return 0.
63  * Segments are assumed to be directed so that the second
64  * coordinate is >= to the first
65  * (e.g. up and to the right).
66  *
67  * @param seg0 a segment to compare
68  * @param seg1 a segment to compare
69  * @return
70  */
71  static int compareX(const geom::LineSegment *seg0, const geom::LineSegment *seg1)
72  {
73  int compare0=seg0->p0.compareTo(seg1->p0);
74  if (compare0!=0) return compare0;
75  return seg0->p1.compareTo(seg1->p1);
76  }
77

79 
80  int leftDepth;
81 
83  DepthSegment(const geom::LineSegment &seg, int depth)
84  :
85  upwardSeg(seg),
86  leftDepth(depth)
87  {
88  // input seg is assumed to be normalized
89  //upwardSeg.normalize();
90  }
91 
104  int compareTo(const DepthSegment& other) const
105  {
110  int orientIndex=upwardSeg.orientationIndex(&(other.upwardSeg));
111 
119  if (orientIndex==0)
120  orientIndex=-1 * other.upwardSeg.orientationIndex(&upwardSeg);
121 
122  // if orientation is determinate, return it
123  if (orientIndex != 0)
124  return orientIndex;
125 
126  // otherwise, segs must be collinear - sort based on minimum X value
127  return compareX(&upwardSeg, &(other.upwardSeg));
128  }

145 SubgraphDepthLocater::getDepth(const Coordinate& p)
146 {
147  vector<DepthSegment*> stabbedSegments;
148  findStabbedSegments(p, stabbedSegments);
149 
150  // if no segments on stabbing line subgraph must be outside all others
151  if (stabbedSegments.size()==0) return 0;
152 
153  sort(stabbedSegments.begin(), stabbedSegments.end(), DepthSegmentLessThen());
154 
155  DepthSegment *ds=stabbedSegments[0];

161 
162  for (vector<DepthSegment *>::iterator
163  it=stabbedSegments.begin(), itEnd=stabbedSegments.end();
164  it != itEnd;

166  {
167  delete *it;
168  }
169
