60 void
61 LinearRing::validateConstruction()
62 {
63  // Empty ring is valid
64  if ( points->isEmpty() ) return;
65 
66  if ( !LineString::isClosed() )
67  {
68  throw util::IllegalArgumentException(
69  "Points of LinearRing do not form a closed linestring"
70  );
71  }
72 
73  if ( points->getSize() < MINIMUM_VALID_SIZE )
74  {
75  std::ostringstream os;
76  os << "Invalid number of points in LinearRing found "
77  << points->getSize() << " - must be 0 or >= 4";
78  throw util::IllegalArgumentException(os.str());
79  }
80 }
