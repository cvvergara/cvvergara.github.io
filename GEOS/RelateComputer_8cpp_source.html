253 void
254 RelateComputer::computeProperIntersectionIM(SegmentIntersector *intersector,IntersectionMatrix *imX)
255 {
256  // If a proper intersection is found, we can set a lower bound on the IM.
257  int dimA=(*arg)[0]->getGeometry()->getDimension();
258  int dimB=(*arg)[1]->getGeometry()->getDimension();
259  bool hasProper=intersector->hasProperIntersection();
260  bool hasProperInterior=intersector->hasProperInteriorIntersection();
261  // For Geometry's of dim 0 there can never be proper intersections.
265  if (dimA==2 && dimB==2) {
266  if (hasProper) imX->setAtLeast("212101212");
267  }
276  else if (dimA==2 && dimB==1) {
277  if (hasProper) imX->setAtLeast("FFF0FFFF2");
278  if (hasProperInterior) imX->setAtLeast("1FFFFF1FF");
279  } else if (dimA==1 && dimB==2) {
280  if (hasProper) imX->setAtLeast("F0FFFFFF2");
281  if (hasProperInterior) imX->setAtLeast("1F1FFFFFF");
282  }
283  /* If edges of LineStrings properly intersect *in an interior point*, all
284  we can deduce is that
285  the interiors intersect. (We can NOT deduce that the exteriors intersect,
286  since some other segments in the geometries might cover the points in the
287  neighbourhood of the intersection.)
288  It is important that the point be known to be an interior point of
289  both Geometries, since it is possible in a self-intersecting geometry to
290  have a proper intersection on one segment that is also a boundary point of another segment.
291  */
292  else if (dimA==1 && dimB==1) {
293  if (hasProperInterior) imX->setAtLeast("0FFFFFFFF");
294  }
