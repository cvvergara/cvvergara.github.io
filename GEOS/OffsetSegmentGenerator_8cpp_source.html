174 /*private*/
175 void
176 OffsetSegmentGenerator::computeOffsetSegment(const LineSegment& seg, int side,
177  double distance, LineSegment& offset)
178 {
179  int sideSign = side == Position::LEFT ? 1 : -1;
180  double dx = seg.p1.x - seg.p0.x;
181  double dy = seg.p1.y - seg.p0.y;
182  double len = sqrt(dx * dx + dy * dy);
183  // u is the vector that is the length of the offset,
184  // in the direction of the segment
185  double ux = sideSign * distance * dx / len;
186  double uy = sideSign * distance * dy / len;
187  offset.p0.x = seg.p0.x - uy;
188  offset.p0.y = seg.p0.y + ux;
189  offset.p1.x = seg.p1.x - uy;
190  offset.p1.y = seg.p1.y + ux;
191 }

196 {
197  LineSegment seg(p0, p1);
198 
199  LineSegment offsetL;
200  computeOffsetSegment(seg, Position::LEFT, distance, offsetL);
201  LineSegment offsetR;
202  computeOffsetSegment(seg, Position::RIGHT, distance, offsetR);
203 
204  double dx=p1.x-p0.x;
205  double dy=p1.y-p0.y;
206  double angle=atan2(dy, dx);
207 
208  switch (bufParams.getEndCapStyle()) {
209  case BufferParameters::CAP_ROUND:
210  // add offset seg points with a fillet between them
211  segList.addPt(offsetL.p1);
212  addFillet(p1, angle+PI/2.0, angle-PI/2.0,
213  CGAlgorithms::CLOCKWISE, distance);
214  segList.addPt(offsetR.p1);
215  break;
216  case BufferParameters::CAP_FLAT:
217  // only offset segment points are added
218  segList.addPt(offsetL.p1);

220  break;
221  case BufferParameters::CAP_SQUARE:
222  // add a square defined by extensions of the offset

226  squareCapSideOffset.y=fabs(distance)*sin(angle);

241 void
242 OffsetSegmentGenerator::addFillet(const Coordinate &p, const Coordinate &p0,
243  const Coordinate &p1, int direction, double radius)
244 {
245  double dx0 = p0.x - p.x;
246  double dy0 = p0.y - p.y;
247  double startAngle = atan2(dy0, dx0);
248  double dx1 = p1.x - p.x;
249  double dy1 = p1.y - p.y;
250  double endAngle = atan2(dy1, dx1);
251 
252  if (direction == CGAlgorithms::CLOCKWISE) {
253  if (startAngle <= endAngle) startAngle += 2.0 * PI;
254  }
255  else { // direction==COUNTERCLOCKWISE
256  if (startAngle >= endAngle) startAngle -= 2.0 * PI;
257  }
258 
259  segList.addPt(p0);
260  addFillet(p, startAngle, endAngle, direction, radius);
261  segList.addPt(p1);
262 }

264 /*private*/
265 void
266 OffsetSegmentGenerator::addFillet(const Coordinate &p, double startAngle,
267  double endAngle, int direction, double radius)
268 {
269  int directionFactor = direction == CGAlgorithms::CLOCKWISE ? -1 : 1;
270 
271  double totalAngle = fabs(startAngle - endAngle);
272  int nSegs = (int) (totalAngle / filletAngleQuantum + 0.5);
273 
274  // no segments because angle is less than increment-nothing to do!
275  if (nSegs<1) return;
276 
277  double initAngle, currAngleInc;
278 
279  // choose angle increment so that each segment has equal length
280  initAngle = 0.0;
281  currAngleInc = totalAngle / nSegs;
282 
283  double currAngle = initAngle;
284  Coordinate pt;
285  while (currAngle < totalAngle) {
286  double angle = startAngle + directionFactor * currAngle;
287  pt.x = p.x + radius * cos(angle);
288  pt.y = p.y + radius * sin(angle);
289  segList.addPt(pt);
290  currAngle += currAngleInc;
291  }
292 }

362 OffsetSegmentGenerator::addOutsideTurn(int orientation, bool addStartPoint)
363 {

399 /* private */
400 void
401 OffsetSegmentGenerator::addInsideTurn(int orientation, bool addStartPoint)
402 {
403  ::geos::ignore_unused_variable_warning(orientation);
404  ::geos::ignore_unused_variable_warning(addStartPoint);
405 
406  // add intersection point of offset segments (if any)
407  li.computeIntersection(offset0.p0, offset0.p1, offset1.p0, offset1.p1);
408  if (li.hasIntersection())
409  {
410  segList.addPt(li.getIntersection(0));
411  return;
412  }
413 
414  // If no intersection is detected, it means the angle is so small
415  // and/or the offset so large that the offsets segments don't
416  // intersect. In this case we must add a "closing segment" to make
417  // sure the buffer curve is continuous,
418  // fairly smooth (e.g. no sharp reversals in direction)
419  // and tracks the buffer correctly around the corner.
420  // The curve connects the endpoints of the segment offsets to points
421  // which lie toward the centre point of the corner.
422  // The joining curve will not appear in the final buffer outline,
423  // since it is completely internal to the buffer polygon.
424  //
425  // In complex buffer cases the closing segment may cut across many
426  // other segments in the generated offset curve.
427  // In order to improve the performance of the noding, the closing
428  // segment should be kept as short as possible.
429  // (But not too short, since that would defeat it's purpose).
430  // This is the purpose of the closingSegLengthFactor heuristic value.
431 
441  if (offset0.p1.distance(offset1.p0) <
442  distance * INSIDE_TURN_VERTEX_SNAP_DISTANCE_FACTOR)
443  {
444  segList.addPt(offset0.p1);
445  }
446  else
447  {
448  // add endpoint of this segment offset
449  segList.addPt(offset0.p1);
450 
451  // Add "closing segment" of required length.
452  if ( closingSegLengthFactor > 0 )
453  {
454  Coordinate mid0(
455  (closingSegLengthFactor*offset0.p1.x + s1.x)/(closingSegLengthFactor + 1),
456  (closingSegLengthFactor*offset0.p1.y + s1.y)/(closingSegLengthFactor + 1)
457  );
458  segList.addPt(mid0);
459 
460  Coordinate mid1(
461  (closingSegLengthFactor*offset1.p0.x + s1.x)/(closingSegLengthFactor + 1),
462  (closingSegLengthFactor*offset1.p0.y + s1.y)/(closingSegLengthFactor + 1)
463  );
464  segList.addPt(mid1);
465  }
466  else
467  {
468  // This branch is not expected to be used
469  // except for testing purposes.
470  // It is equivalent to the JTS 1.9 logic for
471  // closing segments (which results in very poor
472  // performance for large buffer distances)
473  segList.addPt(s1);
474  }
475 
476  // add start point of next segment offset
477  segList.addPt(offset1.p0);
478  }

482 void
483 OffsetSegmentGenerator::addMitreJoin(const geom::Coordinate& p,
484  const geom::LineSegment& offset0,
485  const geom::LineSegment& offset1,
486  double distance)
487 {

521  }

530 /* private */
531 void
532 OffsetSegmentGenerator::addLimitedMitreJoin(
533  const geom::LineSegment& offset0,
534  const geom::LineSegment& offset1,
535  double distance, double mitreLimit)
536 {
537  ::geos::ignore_unused_variable_warning(offset0);
538  ::geos::ignore_unused_variable_warning(offset1);
539 
540  const Coordinate& basePt = seg0.p1;
541 
542  double ang0 = Angle::angle(basePt, seg0.p0);
543  //double ang1 = Angle::angle(basePt, seg1.p1); // unused in JTS, bug ?
544 
545  // oriented angle between segments
546  double angDiff = Angle::angleBetweenOriented(seg0.p0, basePt, seg1.p1);
547  // half of the interior angle
548  double angDiffHalf = angDiff / 2;
549 
550  // angle for bisector of the interior angle between the segments
551  double midAng = Angle::normalize(ang0 + angDiffHalf);
552  // rotating this by PI gives the bisector of the reflex angle
553  double mitreMidAng = Angle::normalize(midAng + PI);
554 
555  // the miterLimit determines the distance to the mitre bevel
556  double mitreDist = mitreLimit * distance;
557  // the bevel delta is the difference between the buffer distance
558  // and half of the length of the bevel segment
559  double bevelDelta = mitreDist * fabs(sin(angDiffHalf));
560  double bevelHalfLen = distance - bevelDelta;
561 
562  // compute the midpoint of the bevel segment
563  double bevelMidX = basePt.x + mitreDist * cos(mitreMidAng);
564  double bevelMidY = basePt.y + mitreDist * sin(mitreMidAng);
565  Coordinate bevelMidPt(bevelMidX, bevelMidY);
566 
567  // compute the mitre midline segment from the corner point to
568  // the bevel segment midpoint
569  LineSegment mitreMidLine(basePt, bevelMidPt);
570 
571  // finally the bevel segment endpoints are computed as offsets from
572  // the mitre midline
573  Coordinate bevelEndLeft;
574  mitreMidLine.pointAlongOffset(1.0, bevelHalfLen, bevelEndLeft);
575  Coordinate bevelEndRight;
576  mitreMidLine.pointAlongOffset(1.0, -bevelHalfLen, bevelEndRight);
577 
578  if (side == Position::LEFT) {
579  segList.addPt(bevelEndLeft);
580  segList.addPt(bevelEndRight);
581  }
582  else {
583  segList.addPt(bevelEndRight);
584  segList.addPt(bevelEndLeft);
585  }
