46 class EnvelopeIntersectsVisitor: public geom::util::ShortCircuitedGeometryVisitor
47 {
48 private:
49 
50  const geom::Envelope &rectEnv;
51  bool intersectsVar;
52 
53  // Declare type as noncopyable
54  EnvelopeIntersectsVisitor(const EnvelopeIntersectsVisitor& other) = delete;
55  EnvelopeIntersectsVisitor& operator=(const EnvelopeIntersectsVisitor& rhs) = delete;
56 
57 protected:
58 
66  void visit(const geom::Geometry &element) override
67  {
68  const geom::Envelope &elementEnv = *(element.getEnvelopeInternal());
69 
70  // skip if envelopes do not intersect
71  if ( ! rectEnv.intersects(elementEnv) ) return;
72 
73  // fully contained - must intersect
74  if ( rectEnv.contains(elementEnv) ) {
75  intersectsVar = true;
76  return;
77  }
78 
79  /*
80  * Since the envelopes intersect and the test element is
81  * connected, if the test envelope is completely bisected by
82  * an edge of the rectangle the element and the rectangle
83  * must touch (This is basically an application of the
84  * Jordan Curve Theorem). The alternative situation
85  * is that the test envelope is "on a corner" of the
86  * rectangle envelope, i.e. is not completely bisected.
87  * In this case it is not possible to make a conclusion
88  * about the presence of an intersection.
89  */
90  if (elementEnv.getMinX() >= rectEnv.getMinX()
91  && elementEnv.getMaxX() <= rectEnv.getMaxX())
92  {
93  intersectsVar=true;
94  return;
95  }
96  if (elementEnv.getMinY() >= rectEnv.getMinY()
97  && elementEnv.getMaxY() <= rectEnv.getMaxY())
98  {
99  intersectsVar = true;
100  return;
101  }
102  }
103 
104  bool isDone() override { return intersectsVar==true; }
105 
106 public:
107 
108  EnvelopeIntersectsVisitor(const geom::Envelope &env)
109  :
110  rectEnv(env),
111  intersectsVar(false)
112  {}
113 
114  bool intersects() { return intersectsVar; }
115 
116 };

139 
140  void visit(const geom::Geometry &geom) override
141  {
142  using geos::algorithm::locate::SimplePointInAreaLocator;
143 
144  const geom::Polygon *poly;
145 
146  // if test geometry is not polygonal this check is not needed
147  if ( nullptr == (poly=dynamic_cast<const geom::Polygon *>(&geom)) ) {
148  return;
149  }
150 
151  const geom::Envelope &elementEnv = *(geom.getEnvelopeInternal());
152 
153  if ( !rectEnv.intersects(elementEnv) ) {
154  return;
155  }
156 
157  // test each corner of rectangle for inclusion
158  for (int i=0; i<4; i++)
159  {
160 
161  const geom::Coordinate &rectPt=rectSeq.getAt(i);
162 
163  if ( !elementEnv.contains(rectPt) ) {
164  continue;
165  }
166 
167  // check rect point in poly (rect is known not to
168  // touch polygon at this point)
169  if ( SimplePointInAreaLocator::containsPointInPolygon(rectPt, poly) )
170  {
171  containsPointVar=true;
172  return;
173  }
174  }
175  }
176 
177  bool isDone() override { return containsPointVar; }
178

180

182  :
183  rectEnv(*(rect.getEnvelopeInternal())),
184  containsPointVar(false),
185  rectSeq(*(rect.getExteriorRing()->getCoordinatesRO()))

265 bool
266 RectangleIntersects::intersects(const geom::Geometry& geom)
267 {
268  if (!rectEnv.intersects(geom.getEnvelopeInternal()))
269  return false;
270 
271  // test envelope relationships
272  EnvelopeIntersectsVisitor visitor(rectEnv);
273  visitor.applyTo(geom);
274  if (visitor.intersects()) return true;
275 
276  // test if any rectangle corner is contained in the target
277  ContainsPointVisitor ecpVisitor(rectangle);
278  ecpVisitor.applyTo(geom);
279  if (ecpVisitor.containsPoint())
280  return true;
281 
282  // test if any lines intersect
283  LineIntersectsVisitor liVisitor(rectangle);
284  liVisitor.applyTo(geom);
285  if (liVisitor.intersects())
286  return true;
287 
288  return false;
289 }
