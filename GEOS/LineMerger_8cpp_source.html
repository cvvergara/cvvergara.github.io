101 LineMerger::merge()
102 {
103  if (mergedLineStrings!=nullptr) return;
104 
105  // reset marks (this allows incremental processing)
106  GraphComponent::setMarkedMap(graph.nodeIterator(), graph.nodeEnd(),
107  false);
108  GraphComponent::setMarked(graph.edgeIterator(), graph.edgeEnd(),
109  false);
110 
111  for (size_t i=0, n=edgeStrings.size(); i<n; ++i)
112  delete edgeStrings[i];
113  edgeStrings.clear();
114 
115  buildEdgeStringsForObviousStartNodes();
116  buildEdgeStringsForIsolatedLoops();
117 
118  auto numEdgeStrings = edgeStrings.size();
119  mergedLineStrings=new vector<LineString*>(numEdgeStrings);
120  for (size_t i=0; i<numEdgeStrings; ++i)
121  {
122  EdgeString *edgeString=edgeStrings[i];
123  (*mergedLineStrings)[i]=edgeString->toLineString();
124  }
125 }

190 void
191 LineMerger::buildEdgeStringsStartingAt(Node *node)
192 {
193  auto edges = node->getOutEdges().getEdges();
194  size_t size = edges.size();
195  for (size_t i=0; i<size; i++)
196  {
197  assert(dynamic_cast<LineMergeDirectedEdge*>(edges[i]));
198  LineMergeDirectedEdge *directedEdge=\
199  static_cast<LineMergeDirectedEdge*> (edges[i]);
200  if (directedEdge->getEdge()->isMarked()) {
201  continue;
202  }
203  edgeStrings.push_back(buildEdgeStringStartingWith(directedEdge));
204  }
205 }

207 EdgeString*
208 LineMerger::buildEdgeStringStartingWith(LineMergeDirectedEdge *start)
209 {
210  EdgeString *edgeString = new EdgeString(factory);
211  LineMergeDirectedEdge *current=start;
212  do {
213  edgeString->add(current);
214  current->getEdge()->setMarked(true);
215  current=current->getNext();
216  } while (current!=nullptr && current!=start);
217  return edgeString;
