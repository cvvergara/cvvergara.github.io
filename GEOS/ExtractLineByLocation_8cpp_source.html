105  }
106 
107  if (lastSegmentIndex >= coordinates->size())

111  }
112 
113  if (! start.isVertex())
114  {
115  newCoordinateArray.add(start.getCoordinate(line));
116  }
117 
118  for (unsigned int i = startSegmentIndex; i <= lastSegmentIndex; i++)
119  {
120  newCoordinateArray.add((*coordinates)[i]);
121  }
122 
123  if (! end.isVertex())
124  {
125  newCoordinateArray.add(end.getCoordinate(line));
126  }
127 
128  // ensure there is at least one coordinate in the result
129  if (newCoordinateArray.size() == 0)

131  newCoordinateArray.add(start.getCoordinate(line));
132  }

148 {

151 
152  if (! start.isVertex())
