101  * Add a LineString forming an edge of the polygon graph.
102  * @param line the line to add
103  */
104 void
105 PolygonizeGraph::addEdge(const LineString *line) {
106  if (line->isEmpty()) return;
107 
108  // TODO(vicky) fix CoordinateSequence::removeRepeatedPoints function
109  auto linePts = CoordinateSequence::removeRepeatedPoints(line->getCoordinatesRO());
110 
111  /*
112  * This would catch invalid linestrings
113  * (containing duplicated points only)
114  */
115  if ( linePts->getSize() < 2 ) {
116  delete linePts;
117  return;
118  }
119 
120  const Coordinate& startPt = linePts->getAt(0);
121  const Coordinate& endPt = linePts->getAt(linePts->getSize() - 1);
122  auto nStart = getNode(startPt);
123  auto nEnd = getNode(endPt);
124  DirectedEdge *de0 = new PolygonizeDirectedEdge(
125  nStart, nEnd, linePts->getAt(1), true);
126  newDirEdges.push_back(de0);
127  DirectedEdge *de1 = new PolygonizeDirectedEdge(
128  nEnd, nStart, linePts->getAt(linePts->getSize() - 2), false);
129  newDirEdges.push_back(de1);
130  Edge *edge = new PolygonizeEdge(line);
131  newEdges.push_back(edge);
132  edge->setDirectedEdges(de0, de1);
133  add(edge);
134 
135  newCoords.push_back(linePts);
136 }

174 std::vector<Node*>
175 PolygonizeGraph::findIntersectionNodes(
176  PolygonizeDirectedEdge *startDE,
177  long label) const {
178  std::vector<Node*> intNodes;
179  auto de = startDE;
180 
181  do {
182  auto node = de->getFromNode();
183  if (getDegree(node, label) > 1) {
184  intNodes.push_back(node);
185  }
186  de = de->getNext();
187  assert(de); // found NULL DE in ring
188  assert(de == startDE || !de->isInRing()); // found DE already in ring
189  } while (de != startDE);
190 
191  return intNodes;

196 PolygonizeGraph::getEdgeRings() const {
197  std::vector<EdgeRing*> edgeRingList;
198  // maybe could optimize this, since most of these pointers should
199  // be set correctly already

233  std::vector<PolygonizeDirectedEdge*> edgeRingStarts;

255 std::vector<const LineString*>
256 PolygonizeGraph::deleteCutEdges() {
257  std::vector<const LineString*> cutLines;
258 
259  computeNextCWEdges();
260 
261  // label the current set of edgerings
262  /* even that is a find it has side efects on the labels */
263  findLabeledEdgeRings(dirEdges); // ignoring the result
264 
265  /*
266  * Cut Edges are edges where both dirEdges have the same label.
267  * Delete them, and record them
268  */
269  for (auto e : dirEdges) {
270  auto de = dynamic_cast<PolygonizeDirectedEdge*>(e);
271 
272  if (de->isMarked()) continue;
273 
274  auto sym = dynamic_cast<PolygonizeDirectedEdge*>(de->getSym());
275 
276  if (de->getLabel() == sym->getLabel()) {
277  de->setMarked(true);
278  sym->setMarked(true);
279 
280  // save the line as a cut edge
281  auto ce = dynamic_cast<PolygonizeEdge*>(de->getEdge());
282 
283  cutLines.push_back(ce->getLine());
284  }
285  }
286  return cutLines;
287 }

304 void
305 PolygonizeGraph::computeNextCWEdges(Node *node) const {
306  auto deStar = node->getOutEdges();
307  PolygonizeDirectedEdge *startDE = nullptr;
308  PolygonizeDirectedEdge *prevDE = nullptr;
309 
310  // the edges are stored in CCW order around the star
311  auto pde = deStar.getEdges();
312  for (auto e : pde) {
313  auto outDE = dynamic_cast<PolygonizeDirectedEdge*>(e);
314  if (outDE->isMarked()) continue;
315  if (!startDE) startDE = outDE;
316  if (prevDE) {
317  dynamic_cast<PolygonizeDirectedEdge*>(prevDE->getSym())->setNext(outDE);
318  }
319  prevDE = outDE;
320  }
321  if (prevDE) {
322  dynamic_cast<PolygonizeDirectedEdge*>(prevDE->getSym())->setNext(startDE);
323  }
324 }

332 void
333 PolygonizeGraph::computeNextCCWEdges(Node *node, long label) const {
334  auto deStar = node->getOutEdges();
335 
336  PolygonizeDirectedEdge *firstOutDE = nullptr;
337  PolygonizeDirectedEdge *prevInDE = nullptr;
338 
339  // the edges are stored in CCW order around the star
340  auto edges = deStar.getEdges();
341 
342  /*
343  * Cycling in reverse order.
344  */
345  for(auto i = edges.size(); i > 0; --i) {
346  auto de = dynamic_cast<PolygonizeDirectedEdge*>(edges[i - 1]);
347  auto sym = dynamic_cast<PolygonizeDirectedEdge*>(de->getSym());
348 
349  auto outDE = (de->getLabel() == label)? de : nullptr;
350  auto inDE = (sym->getLabel() == label)? sym : nullptr;
351 
352  if (!outDE && !inDE) continue; // this edge is not in edgering
353 
354  if (inDE) prevInDE = inDE;
355  if (outDE) {
356  if (prevInDE) {
357  prevInDE->setNext(outDE);
358  prevInDE = nullptr;
359  }
360  if (!firstOutDE) firstOutDE = outDE;
361  }
362  }
363  if (prevInDE) {
364  assert(firstOutDE);
365  prevInDE->setNext(firstOutDE);
366  }
367 }

399 /* public */
400 std::vector<const LineString*>
401 PolygonizeGraph::deleteDangles() {
402  std::vector<const LineString*> dangleLines;
403 
404  auto nodeStack = findNodesOfDegree(1);
405 
406  std::set<const LineString*> uniqueDangles;
407 
408  while (!nodeStack.empty()) {
409  auto node = nodeStack.back();
410  nodeStack.pop_back();
411  deleteAllEdges(node);
412 
413  /* get sorted edges in ascending order by angle with the positive x-axis */
414  auto nodeOutEdges(node->getOutEdges().getEdges());
415 
416  for(auto oe : nodeOutEdges) {
417  auto de(dynamic_cast<PolygonizeDirectedEdge*>(oe));
418  // delete this edge and its sym
419  de->setMarked(true);
420  auto sym(dynamic_cast<PolygonizeDirectedEdge*>(de->getSym()));
421  if (sym) sym->setMarked(true);
422  // save the line as a dangle
423  auto e(dynamic_cast<PolygonizeEdge*>(de->getEdge()));
424  auto ls(e->getLine());
425  if (uniqueDangles.insert(ls).second) {
426  dangleLines.push_back(ls);
427  }
428 
429  auto toNode = de->getToNode();
430  // add the toNode to the list to be processed,
431  // if it is now a dangle
432  if (getDegreeNonDeleted(toNode) == 1) {
433  nodeStack.push_back(toNode);
434  }
435  }
436  }
437  return dangleLines;
438 }
