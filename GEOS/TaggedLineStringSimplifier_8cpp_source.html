85 void
86 TaggedLineStringSimplifier::simplifySection(std::size_t i,
87  std::size_t j, std::size_t depth)
88 {
89  depth += 1;
90 
91 #if GEOS_DEBUG
92  std::cerr << "TaggedLineStringSimplifier[" << this << "] "
93  << " simplifying section " << i << "-" << j
94  << std::endl;
95 #endif
96 
97  vector<std::size_t> sectionIndex(2);
98 
99  if((i+1) == j)
100  {
101 
102 #if GEOS_DEBUG
103  std::cerr << "single segment, no flattening"
104  << std::endl;
105 #endif
106  unique_ptr<TaggedLineSegment> newSeg(new
107  TaggedLineSegment(*(line->getSegment(i))));
108 
109  line->addToResult(std::move(newSeg));
110  // leave this segment in the input index, for efficiency
111  return;
112  }
113 
114  bool isValidToSimplify = true;
115 
124  if (line->getResultSize() < line->getMinimumSize())
125  {
126  std::size_t worstCaseSize = depth + 1;
127  if (worstCaseSize < line->getMinimumSize())
128  isValidToSimplify = false;
129  }
130 
131  double distance;
132 
133  // pass distance by ref
134  std::size_t furthestPtIndex = findFurthestPoint(linePts, i, j, distance);
135 
136 #if GEOS_DEBUG
137  std::cerr << "furthest point " << furthestPtIndex
138  << " at distance " << distance
139  << std::endl;
140 #endif
141 
142  // flattening must be less than distanceTolerance
143  if ( distance > distanceTolerance ) isValidToSimplify = false;
144 
145  // test if flattened section would cause intersection
146  LineSegment candidateSeg(linePts->getAt(i), linePts->getAt(j));
147 
148  sectionIndex[0] = i;
149  sectionIndex[1] = j;
150 
151  if (hasBadIntersection(line, sectionIndex, candidateSeg))
152  isValidToSimplify = false;
153 
154  if (isValidToSimplify)
155  {
156 
157  unique_ptr<TaggedLineSegment> newSeg = flatten(i, j);
158 
159 #if GEOS_DEBUG
160  std::cerr << "isValidToSimplify, adding seg "
161  << newSeg->p0 << ", " << newSeg->p1
162  << " to TaggedLineSegment["<<line<<"] result "
163  << std::endl;
164 #endif
165 
166  line->addToResult(std::move(newSeg));
167  return;
168  }
169 
170  simplifySection(i, furthestPtIndex, depth);
171  simplifySection(furthestPtIndex, j, depth);
172 
173 }
