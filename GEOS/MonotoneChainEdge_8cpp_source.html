122 MonotoneChainEdge::computeIntersectsForChain(int start0, int end0,
123  const MonotoneChainEdge &mce, int start1, int end1,
124  SegmentIntersector &ei)
125 {
126  // terminating condition for the recursion
127  if (end0-start0==1 && end1-start1==1) {
128  ei.addIntersections(e, start0, mce.e, start1);
129  return;
130  }
131 
132  const Coordinate& p00=pts->getAt(start0);
133  const Coordinate& p01=pts->getAt(end0);
134  const Coordinate& p10=mce.pts->getAt(start1);
135  const Coordinate& p11=mce.pts->getAt(end1);
136 
137  // nothing to do if the envelopes of these chains don't overlap
138  env1.init(p00,p01);
139  env2.init(p10,p11);
140 
141  if (!env1.intersects(&env2)) return;
142  // the chains overlap, so split each in half and iterate
143  // (binary search)
144  int mid0=(start0+end0)/2;
145  int mid1=(start1+end1)/2;
146 
147  // Assert: mid != start or end
148  // (since we checked above for end - start <= 1)
149  // check terminating conditions before recursing
150  if (start0<mid0) {
151  if (start1<mid1)
152  computeIntersectsForChain(start0, mid0, mce,
153  start1, mid1, ei);
154  if (mid1<end1)
155  computeIntersectsForChain(start0, mid0, mce,
156  mid1, end1, ei);
157  }
158  if (mid0<end0) {
159  if (start1<mid1)
160  computeIntersectsForChain(mid0, end0, mce,
161  start1, mid1, ei);
162  if (mid1<end1)
163  computeIntersectsForChain(mid0, end0, mce,
164  mid1, end1, ei);
165  }
