165 /*public*/
166 int
167 EdgeEnd::compareDirection(const EdgeEnd* e) const
168 {
169  assert(e);
170  if (dx == e->dx && dy == e->dy)
171  return 0;
172 
173  // if the rays are in different quadrants,
174  // determining the ordering is trivial
175  if (quadrant>e->quadrant) return 1;
176  if (quadrant<e->quadrant) return -1;
177 
178  // vectors are in the same quadrant - check relative
179  // orientation of direction vectors
180  // this is > e if it is CCW of e
181  return CGAlgorithms::computeOrientation(e->p0, e->p1, p1);
182 }
