92 GeometryNoder::toGeometry(SegmentString::NonConstVect& nodedEdges)
93 {
94  const geom::GeometryFactory *geomFact = argGeom.getFactory();
95 
96  std::set< OrientedCoordinateArray > ocas;
97 
98  // Create a geometry out of the noded substrings.
99  std::vector< geom::Geometry* >* lines = new std::vector< geom::Geometry * >();
100  lines->reserve(nodedEdges.size());
101  for (auto &ss : nodedEdges)
102  {
103  const geom::CoordinateSequence* coords = ss->getCoordinates();
104 
105  // Check if an equivalent edge is known
106  OrientedCoordinateArray oca1( *coords );
107  if ( ocas.insert(oca1).second ) {
108  geom::Geometry* tmp = geomFact->createLineString( coords->clone() );
109  lines->push_back( tmp );
110  }
111  }
112 
113  std::unique_ptr<geom::Geometry> noded ( geomFact->createMultiLineString( lines ) );
114 
115  return noded;
116 }
