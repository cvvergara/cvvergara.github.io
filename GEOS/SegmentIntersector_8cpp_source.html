111 /*
112  * A trivial intersection is an apparent self-intersection which in fact
113  * is simply the point shared by adjacent line segments.
114  * Note that closed edges require a special check for the point
115  * shared by the beginning and end segments.
116  */
117 bool
118 SegmentIntersector::isTrivialIntersection(Edge *e0,int segIndex0,Edge *e1,int segIndex1)
119 {
120 // if (e0->equals(e1))
121  if (e0==e1) {
122  if (li->getIntersectionNum()==1) {
123  if (isAdjacentSegments(segIndex0,segIndex1))
124  return true;
125  if (e0->isClosed()) {
126  int maxSegIndex=e0->getNumPoints()-1;
127  if ((segIndex0==0 && segIndex1==maxSegIndex)
128  || (segIndex1==0 && segIndex0==maxSegIndex)) {
129  return true;
130  }
131  }
132  }
133  }
134  return false;
135 }
