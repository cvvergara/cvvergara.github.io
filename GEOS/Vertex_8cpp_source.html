48 int Vertex::classify(const Vertex &p0, const Vertex &p1)
49 {
50  Vertex &p2 = *this;
51  std::unique_ptr<Vertex> a = p1.sub(p0);
52  std::unique_ptr<Vertex> b = p2.sub(p0);
53  double sa = a->crossProduct(*b);
54 
55  if (sa > 0.0)
56  return LEFT;
57  if (sa < 0.0)
58  return RIGHT;
59  if ((a->getX() * b->getX() < 0.0) || (a->getY() * b->getY() < 0.0))
60  return BEHIND;
61  if (a->magn() < b->magn())
62  return BEYOND;
63  if (p0.equals(p2))
64  return ORIGIN;
65  if (p1.equals(p2))
66  return DESTINATION;
67  else
68  return BETWEEN;
69 }

124 std::unique_ptr<Vertex> Vertex::circleCenter(const Vertex &b, const Vertex &c) const
125 {
126  std::unique_ptr<Vertex> a(new Vertex(getX(), getY()));
127  // compute the perpendicular bisector of cord ab
128  std::unique_ptr<HCoordinate> cab = bisector(*a, b);
129  // compute the perpendicular bisector of cord bc
130  std::unique_ptr<HCoordinate> cbc = bisector(b, c);
131  // compute the intersection of the bisectors (circle radii)
132  std::unique_ptr<HCoordinate> hcc(new HCoordinate(*cab, *cbc));

144 double Vertex::interpolateZValue(const Vertex &v0, const Vertex &v1,
145  const Vertex &v2) const
146 {
147  double x0 = v0.getX();
148  double y0 = v0.getY();
149  double a = v1.getX() - x0;
150  double b = v2.getX() - x0;
151  double c = v1.getY() - y0;
152  double d = v2.getY() - y0;
153  double det = a * d - b * c;
154  double dx = this->getX() - x0;
155  double dy = this->getY() - y0;
156  double t = (d * dx - b * dy) / det;
157  double u = (-c * dx + a * dy) / det;
158  double z = v0.getZ() + t * (v1.getZ() - v0.getZ()) + u * (v2.getZ() - v0.getZ());
159  return z;
160 }

162 double Vertex::interpolateZ(const Coordinate &p, const Coordinate &v0,
163  const Coordinate &v1, const Coordinate &v2)
164 {
165  double x0 = v0.x;
166  double y0 = v0.y;

168  double b = v2.x - x0;
169  double c = v1.y - y0;

171  double det = a * d - b * c;
172  double dx = p.x - x0;
173  double dy = p.y - y0;
174  double t = (d * dx - b * dy) / det;
175  double u = (-c * dx + a * dy) / det;
176  double z = v0.z + t * (v1.z - v0.z) + u * (v2.z - v0.z);
