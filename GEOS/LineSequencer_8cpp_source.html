52 bool
53 LineSequencer::isSequenced(const Geometry* geom)
54 {
55  const MultiLineString *mls;
56 
57  if ( nullptr == (mls=dynamic_cast<const MultiLineString *>(geom)) )
58  {
59  return true;
60  }
61 
62  // the nodes in all subgraphs which have been completely scanned
63  Coordinate::ConstSet prevSubgraphNodes;
64  Coordinate::ConstVect currNodes;
65 
66  const Coordinate* lastNode = nullptr;
67 
68  for (std::size_t i=0, n=mls->getNumGeometries(); i<n; ++i)
69  {
70  const LineString* lineptr = \
71  dynamic_cast<const LineString*>(mls->getGeometryN(i));
72  assert(lineptr);
73  const LineString& line = *lineptr;
74 
75 
76  const Coordinate* startNode = &(line.getCoordinateN(0));
77  const Coordinate* endNode = &(line.getCoordinateN(static_cast<int>(line.getNumPoints() - 1)));
78 
83  if (prevSubgraphNodes.find(startNode) != prevSubgraphNodes.end())
84  {
85  return false;
86  }
87  if (prevSubgraphNodes.find(endNode) != prevSubgraphNodes.end())
88  {
89  return false;
90  }
91 
92  if (lastNode != nullptr)
93  {
94  if (! startNode->equals2D(*lastNode))
95  {
96  // start new connected sequence
97  prevSubgraphNodes.insert(currNodes.begin(),
98  currNodes.end());
99  currNodes.clear();
100  }
101  }
102  currNodes.push_back(startNode);
103  currNodes.push_back(endNode);
104  lastNode = endNode;
105  }
106  return true;
107 }

139  Sequences *sequences = new Sequences();
140  planargraph::algorithm::ConnectedSubgraphFinder csFinder(graph);
141  vector<planargraph::Subgraph*> subgraphs;
142  csFinder.getConnectedSubgraphs(subgraphs);
143  for (vector<planargraph::Subgraph*>::const_iterator
144  it=subgraphs.begin(), endIt=subgraphs.end();
145  it!=endIt;

201 Geometry*
202 LineSequencer::buildSequencedGeometry(const Sequences& sequences)
203 {
204  unique_ptr<Geometry::NonConstVect> lines(new Geometry::NonConstVect);
205 
206  for (Sequences::const_iterator
207  i1=sequences.begin(), i1End=sequences.end();
208  i1 != i1End;
209  ++i1)
210  {
211  planargraph::DirectedEdge::NonConstList& seq = *(*i1);
212  for(planargraph::DirectedEdge::NonConstList::iterator i2=seq.begin(),
213  i2End=seq.end(); i2 != i2End; ++i2)
214  {
215  const planargraph::DirectedEdge* de = *i2;
216  assert(dynamic_cast<LineMergeEdge* >(de->getEdge()));
217  LineMergeEdge* e = static_cast<LineMergeEdge* >(de->getEdge());
218  const LineString* line = e->getLine();
219 
220  // lineToAdd will be a *copy* of input things
221  LineString* lineToAdd;
222 
223  if ( ! de->getEdgeDirection() && ! line->isClosed() ) {
224  lineToAdd = reverse(line);
225  } else {
226  Geometry* lineClone = line->clone();
227  lineToAdd = dynamic_cast<LineString *>(lineClone);
228  assert(lineToAdd);
229  }
230 
231  lines->push_back(lineToAdd);
232  }
233  }
234 
235  if ( lines->size() == 0 ) {
236  return nullptr;
237  } else {
238  Geometry::NonConstVect *l=lines.get();
239  lines.release();
240  return factory->buildGeometry(l);
241  }

299 LineSequencer::addReverseSubpath(const planargraph::DirectedEdge *de,
300  planargraph::DirectedEdge::NonConstList& deList,
301  planargraph::DirectedEdge::NonConstList::iterator lit,
302  bool expectedClosed)
303 {
304  using planargraph::Node;

306 
307  // trace an unvisited path *backwards* from this de
308  Node* endNode = de->getToNode();
309 
310  Node* fromNode = nullptr;
311  while (true) {
312  deList.insert(lit, de->getSym());
313  de->getEdge()->setVisited(true);

315  const DirectedEdge* unvisitedOutDE = findUnvisitedBestOrientedDE(fromNode);
316 
317  // this must terminate, since we are continually marking edges as visited
318  if (unvisitedOutDE == nullptr) break;

320  }

331 planargraph::DirectedEdge::NonConstList*
332 LineSequencer::findSequence(planargraph::Subgraph& graph)
333 {
334  using planargraph::DirectedEdge;
335  using planargraph::Node;
336  using planargraph::GraphComponent;
337 
338  GraphComponent::setVisited(graph.edgeBegin(),
339  graph.edgeEnd(), false);
340 
341  const Node* startNode = findLowestDegreeNode(graph);
342 
343  const auto startDE = *(startNode->getOutEdges().begin());
344  const DirectedEdge *startDESym = startDE->getSym();
345 
346  DirectedEdge::NonConstList *seq = new DirectedEdge::NonConstList();
347 
348  DirectedEdge::NonConstList::iterator lit=seq->begin();
349  addReverseSubpath(startDESym, *seq, lit, false);
350 
351  lit=seq->end();
352  while (lit != seq->begin()) {
353  const DirectedEdge* prev = *(--lit);
354  const DirectedEdge* unvisitedOutDE = findUnvisitedBestOrientedDE(prev->getFromNode());
355  if (unvisitedOutDE != nullptr)
356  addReverseSubpath(unvisitedOutDE->getSym(), *seq, lit, true);
357  }
358 
359  // At this point, we have a valid sequence of graph DirectedEdges,
360  // but it is not necessarily appropriately oriented relative to
361  // the underlying geometry.
362  DirectedEdge::NonConstList* orientedSeq = orient(seq);
363 
364  if (orientedSeq != seq) delete seq;
365 
366  return orientedSeq;
367 }

370 planargraph::DirectedEdge::NonConstList*
371 LineSequencer::orient(planargraph::DirectedEdge::NonConstList* seq)
372 {
373  using namespace geos::planargraph;
374 
375  const DirectedEdge* startEdge = seq->front();
376  const DirectedEdge* endEdge = seq->back();
377  Node* startNode = startEdge->getFromNode();
378  Node* endNode = endEdge->getToNode();
379 
380  bool flipSeq = false;
381  bool hasDegree1Node = \
382  startNode->getDegree() == 1 || endNode->getDegree() == 1;
383 
384  if (hasDegree1Node)
385  {
386  bool hasObviousStartNode = false;
387 
388  // test end edge before start edge, to make result stable
389  // (ie. if both are good starts, pick the actual start
390  if (endEdge->getToNode()->getDegree() == 1 &&
391  endEdge->getEdgeDirection() == false)
392  {
393  hasObviousStartNode = true;
394  flipSeq = true;
395  }
396  if (startEdge->getFromNode()->getDegree() == 1 &&
397  startEdge->getEdgeDirection() == true)
398  {
399  hasObviousStartNode = true;
400  flipSeq = false;
401  }
402 
403  // since there is no obvious start node,
404  // use any node of degree 1
405  if (! hasObviousStartNode)
406  {
407  // check if the start node should actually
408  // be the end node
409  if (startEdge->getFromNode()->getDegree() == 1)
410  flipSeq = true;
411  // if the end node is of degree 1, it is
412  // properly the end node
413  }
414 
415  }
416 
417 
418  // if there is no degree 1 node, just use the sequence as is
419  // (Could insert heuristic of taking direction of majority of
420  // lines as overall direction)
421 
422  if (flipSeq)
423  {
424  return reverse(*seq);
425  }
426  return seq;
427 }
