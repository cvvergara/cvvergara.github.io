224 bool
225 Point::equalsExact(const Geometry *other, double tolerance) const
226 {
227  if (!isEquivalentClass(other)) {
228  return false;
229  }
230 
231  // assume the isEquivalentClass would return false
232  // if other is not a point
233  assert(dynamic_cast<const Point*>(other));
234 
235  if ( isEmpty() ) return other->isEmpty();
236  else if ( other->isEmpty() ) return false;
237 
238  const Coordinate* this_coord = getCoordinate();
239  const Coordinate* other_coord = other->getCoordinate();
240 
241  // assume the isEmpty checks above worked :)
242  assert(this_coord && other_coord);
243 
244  return equal(*this_coord, *other_coord, tolerance);
245 }
