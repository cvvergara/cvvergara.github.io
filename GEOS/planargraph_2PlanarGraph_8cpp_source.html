88 /*
89  * Removes a node from the graph, along with any associated
90  * DirectedEdges and Edges.
91  */
92 void
93 PlanarGraph::remove(Node *node)
94 {
95  // unhook all directed edges
96  auto & outEdges=node->getOutEdges().getEdges();
97  for(auto de : outEdges) {
98  DirectedEdge *sym = de->getSym();
99  // remove the diredge that points to this node
100  if (sym) remove(sym);
101  find_and_erase(de, dirEdges);
102  find_and_erase(de->getEdge(), edges);
103  }
104  // remove the node from the graph
105  nodeMap.remove(node->getCoordinate());
106  //nodes.remove(node);
107 }
