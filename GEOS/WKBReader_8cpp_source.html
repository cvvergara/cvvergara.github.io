54 ostream &
55 WKBReader::printHEX(istream &is, ostream &os)
56 {
57  static const char hex[] = "0123456789ABCDEF";
58 
59  std::streampos pos = is.tellg(); // take note of input stream get pointer
60  is.seekg(0, ios::beg); // rewind input stream
61 
62  char each=0;
63  while(is.read(&each, 1))
64  {
65  const unsigned char c=each;
66  int low = (c & 0x0F);
67  int high = (c >> 4);
68  os << hex[high] << hex[low];
69  }
70 
71  is.clear(); // clear input stream eof flag
72  is.seekg(pos); // reset input stream position
73 
74  return os;
75 }

131 Geometry *
132 WKBReader::readHEX(istream &is)
133 {
134  // setup input/output stream
135  stringstream os(ios_base::binary|ios_base::in|ios_base::out);
136 
137  while( true )
138  {
139  const int input_high = is.get();
140  if (input_high == char_traits<char>::eof())
141  break;
142 
143  const int input_low = is.get();
144  if (input_low == char_traits<char>::eof())
145  throw ParseException("Premature end of HEX string");
146 
147  const char high = static_cast<char>(input_high);
148  const char low = static_cast<char>(input_low);
149 
150  const unsigned char result_high = ASCIIHexToUChar(high);
151  const unsigned char result_low = ASCIIHexToUChar(low);
152 
153  const unsigned char value =
154  static_cast<char>((result_high<<4) + result_low);
155 
156 #if DEBUG_HEX_READER
157  cout<<"HEX "<<high<<low<<" -> DEC "<<(int)value<<endl;
158 #endif
159  // write the value to the output stream
160  os << value;
161  }
162 
163  // now call read to convert the geometry
164  return this->read(os);
165 }

174 Geometry *
175 WKBReader::readGeometry()
176 {
177  // determine byte order
178  unsigned char byteOrder = dis.readByte();
179 
180 #if DEBUG_WKB_READER
181  cout<<"WKB byteOrder: "<<(int)byteOrder<<endl;
182 #endif
183 
184  // default is machine endian
185  if (byteOrder == WKBConstants::wkbNDR)
186  dis.setOrder(ByteOrderValues::ENDIAN_LITTLE);
187  else if (byteOrder == WKBConstants::wkbXDR)
188  dis.setOrder(ByteOrderValues::ENDIAN_BIG);
189 
190  int typeInt = dis.readInt();
191  int geometryType = typeInt & 0xff;
192 
193 #if DEBUG_WKB_READER
194  cout<<"WKB geometryType: "<<geometryType<<endl;
195 #endif
196 
197  bool hasZ = ((typeInt & 0x80000000) != 0);
198  if (hasZ) inputDimension = 3;
199  else inputDimension = 2; // doesn't handle M currently
200 
201 #if DEBUG_WKB_READER
202  cout<<"WKB hasZ: "<<hasZ<<endl;
203 #endif
204 
205 #if DEBUG_WKB_READER
206  cout<<"WKB dimensions: "<<inputDimension<<endl;
207 #endif
208 
209  bool hasSRID = ((typeInt & 0x20000000) != 0);
210 
211 #if DEBUG_WKB_READER
212  cout<<"WKB hasSRID: "<<hasSRID<<endl;
213 #endif
214 
215  int SRID = 0;
216  if (hasSRID) SRID = dis.readInt(); // read SRID
217 
218 
219  // allocate space for ordValues
220  if ( ordValues.size() < inputDimension )
221  ordValues.resize(inputDimension);
222 
223  Geometry *result;
224 
225  switch (geometryType) {
226  case WKBConstants::wkbPoint :
227  result = readPoint();
228  break;
229  case WKBConstants::wkbLineString :
230  result = readLineString();
231  break;
232  case WKBConstants::wkbPolygon :
233  result = readPolygon();
234  break;
235  case WKBConstants::wkbMultiPoint :
236  result = readMultiPoint();
237  break;
238  case WKBConstants::wkbMultiLineString :
239  result = readMultiLineString();
240  break;
241  case WKBConstants::wkbMultiPolygon :
242  result = readMultiPolygon();
243  break;
244  case WKBConstants::wkbGeometryCollection :
245  result = readGeometryCollection();
246  break;
247  default:
248  stringstream err;
249  err << "Unknown WKB type " << geometryType;
250  throw ParseException(err.str());
251  }
252 
253  result->setSRID(SRID);
254  return result;
255 }

302

307  holes = new vector<Geometry *>(numRings-1);
308  for (int i=0; i<numRings-1; i++)
309  (*holes)[i] = (Geometry *)readLinearRing();

312  delete (*holes)[i];
313  delete holes;

315  throw;
316  }
317  }
318  return factory.createPolygon(shell, holes);

420 CoordinateSequence *
421 WKBReader::readCoordinateSequence(int size)
422 {
423  CoordinateSequence *seq = factory.getCoordinateSequenceFactory()->create(size, inputDimension);
424  auto targetDim = seq->getDimension();
425  if ( targetDim > inputDimension )
426  targetDim = inputDimension;
427  for (int i=0; i<size; i++) {
428  readCoordinate();
429  for (unsigned int j=0; j<targetDim; j++) {
430  seq->setOrdinate(i, j, ordValues[j]);
431  }
432  }
