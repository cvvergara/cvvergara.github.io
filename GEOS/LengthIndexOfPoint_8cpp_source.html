80 double LengthIndexOfPoint::indexOfFromStart(const Coordinate& inputPt, double minIndex) const
81 {
82  double minDistance = numeric_limits<double>::max();
83 
84  double ptMeasure = minIndex;
85  double segmentStartMeasure = 0.0;
86  LineSegment seg;
87  LinearIterator it(linearGeom);
88  while (it.hasNext())
89  {

93  seg.p1 = it.getSegmentEnd();
94  double segDistance = seg.distance(inputPt);
95  double segMeasureToPt = segmentNearestMeasure(&seg, inputPt, segmentStartMeasure);
96  if (segDistance < minDistance

98  {
99  ptMeasure = segMeasureToPt;

101  }
102  segmentStartMeasure += seg.getLength();
103  }
104  it.next();
105  }
