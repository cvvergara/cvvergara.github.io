78 /*
79  * Find and mark L edges which are "covered" by the result area (if any).
80  * L edges at nodes which also have A edges can be checked by checking
81  * their depth at that node.
82  * L edges at nodes which do not have A edges can be checked by doing a
83  * point-in-polygon test with the previously computed result areas.
84  */
85 void
86 LineBuilder::findCoveredLineEdges()
87 {
88 // first set covered for all L edges at nodes which have A edges too
89  map<Coordinate*,Node*,CoordinateLessThen> &nodeMap=op->getGraph().getNodeMap()->nodeMap;
90  map<Coordinate*,Node*,CoordinateLessThen>::iterator it=nodeMap.begin();
91  map<Coordinate*,Node*,CoordinateLessThen>::iterator endIt=nodeMap.end();
92  for ( ; it!=endIt; ++it)
93  {
94  Node *node=it->second;
95  //node.print(System.out);
96  assert(dynamic_cast<DirectedEdgeStar*>(node->getEdges()));
97  DirectedEdgeStar* des=static_cast<DirectedEdgeStar*>(node->getEdges());
98  des->findCoveredLineEdges();
99  //((DirectedEdgeStar*)node->getEdges())->findCoveredLineEdges();
100  }
101 
102  /*
103  * For all L edges which weren't handled by the above,
104  * use a point-in-poly test to determine whether they are covered
105  */
106  vector<EdgeEnd*> *ee=op->getGraph().getEdgeEnds();
107  for(size_t i=0, s=ee->size(); i<s; ++i)
108  {
109  assert(dynamic_cast<DirectedEdge*>((*ee)[i]));
110  DirectedEdge *de=static_cast<DirectedEdge*>((*ee)[i]);
111  Edge *e=de->getEdge();
112  if (de->isLineEdge() && !e->isCoveredSet()) {
113  bool isCovered=op->isCoveredByA(de->getCoordinate());
114  e->setCovered(isCovered);
115  }
116  }

159 void
160 LineBuilder::collectBoundaryTouchEdge(DirectedEdge *de,
161  OverlayOp::OpCode opCode, vector<Edge*> *edges)
162 {
163  if (de->isLineEdge()) return; // only interested in area edges
164  if (de->isVisited()) return; // already processed
165 
166  // added to handle dimensional collapses
167  if (de->isInteriorAreaEdge()) return;
168 
169  // if the edge linework is already included, don't include it again
170  if (de->getEdge()->isInResult()) return;
171 
172  // sanity check for labelling of result edgerings
173  assert( ! ( de->isInResult() || de->getSym()->isInResult() )
174  ||
175  ! de->getEdge()->isInResult() );
176 
177 
178  // include the linework if it's in the result of the operation
179  const Label& label = de->getLabel();
180  if ( OverlayOp::isResultOfOp(label, opCode)
181  && opCode == OverlayOp::opINTERSECTION )
182  {
183  edges->push_back(de->getEdge());
184  de->setVisitedEdge(true);
185  }

204 /*
205  * If the given CoordinateSequence has mixed 3d/2d vertexes
206  * set Z for all vertexes missing it.
207  * The Z value is interpolated between 3d vertexes and copied
208  * from a 3d vertex to the end.
209  */
210 void
211 LineBuilder::propagateZ(CoordinateSequence *cs)
212 {
213 #if GEOS_DEBUG
214  cerr<<"LineBuilder::propagateZ() called"<<endl;
215 #endif
216 
217  vector<size_t> v3d; // vertex 3d
218  size_t cssize = cs->getSize();
219  for (size_t i = 0; i < cssize; ++i)
220  {
221  if ( !ISNAN(cs->getAt(i).z) ) v3d.push_back(i);
222  }
223 
224 #if GEOS_DEBUG
225  cerr<<" found "<<v3d.size()<<" 3d vertexes"<<endl;
226 #endif
227 
228  if ( v3d.size() == 0 )
229  {
230 #if GEOS_DEBUG
231  cerr<<" nothing to do"<<endl;
232 #endif
233  return;
234  }
235 
236  Coordinate buf;
237 
238  // fill initial part
239  if ( v3d[0] != 0 )
240  {
241  double z = cs->getAt(v3d[0]).z;
242  for (size_t j = 0; j < v3d[0]; ++j)
243  {
244  buf = cs->getAt(j);
245  buf.z = z;
246  cs->setAt(buf, j);
247  }
248  }
249 
250  // interpolate inbetweens
251  size_t prev=v3d[0];
252  for (size_t i = 1; i < v3d.size(); ++i)
253  {
254  auto curr = v3d[i];
255  auto dist = curr - prev;
256  if (dist > 1)
257  {
258  const Coordinate &cto = cs->getAt(curr);
259  const Coordinate &cfrom = cs->getAt(prev);
260  double gap = cto.z - cfrom.z;
261  double zstep = gap / static_cast<double>(dist);
262  double z = cfrom.z;
263  for (size_t j = prev+1; j < curr; ++j)
264  {
265  buf = cs->getAt(j);
266  z+=zstep;
267  buf.z = z;
268  cs->setAt(buf, j);
269  }
270  }
271  prev = curr;
272  }
273 
274  // fill final part
275  if ( prev < cssize-1 )
276  {
277  double z = cs->getAt(prev).z;
278  for (size_t j=prev+1; j<cssize; j++)
279  {
280  buf = cs->getAt(j);
281  buf.z = z;
282  cs->setAt(buf, j);
283  }
284  }
