200 /*public*/
201 void
202 PlanarGraph::addEdges(const vector<Edge*>& edgesToAdd)
203 {
204  // create all the nodes for the edges
205  for (vector<Edge*>::const_iterator it=edgesToAdd.begin(),
206  endIt=edgesToAdd.end(); it!=endIt; ++it)
207  {
208  Edge *e=*it;
209  assert(e);
210  edges->push_back(e);
211 
212  // PlanarGraph destructor will delete all DirectedEdges
213  // in edgeEndList, which is where these are added
214  // by the ::add(EdgeEnd) call
215  std::unique_ptr<DirectedEdge> de1(new DirectedEdge(e, true));
216  std::unique_ptr<DirectedEdge> de2(new DirectedEdge(e, false));
217  de1->setSym(de2.get());
218  de2->setSym(de1.get());
219 
220  // First, ::add takes the ownership, then follows with operations that may throw.
221  add(de1.release());
222  add(de2.release());
223  }
224 }
