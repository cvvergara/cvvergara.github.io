44 HCoordinate::intersection(const Coordinate &p1, const Coordinate &p2,

57  // unrolled computation
58 
59  double px = p1.y - p2.y;
60  double py = p2.x - p1.x;
61  double pw = p1.x * p2.y - p2.x * p1.y;
62 
63  double qx = q1.y - q2.y;
64  double qy = q2.x - q1.x;
65  double qw = q1.x * q2.y - q2.x * q1.y;
66 
67  double x = py * qw - qy * pw;
68  double y = qx * pw - px * qw;
69  double w = px * qy - qx * py;
70 
71  double xInt = x/w;
72  double yInt = y/w;
73 
74  if ( (!FINITE(xInt)) || (!FINITE(yInt)) )
75  {
76  throw NotRepresentableException();
77  }
78 
79  ret = Coordinate(xInt, yInt);
