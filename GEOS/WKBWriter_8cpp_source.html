84 WKBWriter::write(const Geometry &g, ostream &os)
85 {
86  outputDimension = defaultOutputDimension;

89 
90  outStream = &os;

94  return writePoint(*x);

97  if ( const LineString* x = dynamic_cast<const LineString*>(&g) )

99  return writeLineString(*x);

103  {
104  return writePolygon(*x);
105  }
106 
107  if ( const MultiPoint* x = dynamic_cast<const MultiPoint*>(&g) )
108  {
109  return writeGeometryCollection(*x, WKBConstants::wkbMultiPoint);
110  }
111 
112  if ( const MultiLineString* x = dynamic_cast<const MultiLineString*>(&g) )

160 void
161 WKBWriter::writePolygon(const Polygon &g)
162 {
163  writeByteOrder();
164 
165  writeGeometryType(WKBConstants::wkbPolygon, g.getSRID());
166  writeSRID(g.getSRID());
167 
168  if (g.isEmpty()) {
169  writeInt(0);
170  return;
171  }
172 
173  std::size_t nholes = g.getNumInteriorRing();
174  writeInt(static_cast<int>(nholes + 1));
175 
176  const LineString* ls = g.getExteriorRing();
177  assert(ls);
178 
179  const CoordinateSequence* cs=ls->getCoordinatesRO();
180  assert(cs);
181 
182  writeCoordinateSequence(*cs, true);
183  for (std::size_t i=0; i<nholes; i++)
184  {
185  ls = g.getInteriorRingN(i);
186  assert(ls);
187 
188  cs = ls->getCoordinatesRO();
189  assert(cs);
190 
191  writeCoordinateSequence(*cs, true);
192  }
