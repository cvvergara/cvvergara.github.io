57 /*public*/
58 void
59 OffsetCurveBuilder::getLineCurve(const CoordinateSequence *inputPts,
60  double nDistance, vector<CoordinateSequence*>& lineList)
61 {
62  distance = nDistance;
63 
64  // a zero or (non-singlesided) negative width buffer of a line/point is empty
65  if (distance == 0.0) return;
66  if (distance < 0.0 && ! bufParams.isSingleSided()) return;
67 
68  double posDistance = std::abs(distance);
69 
70  std::unique_ptr<OffsetSegmentGenerator> segGen = getSegGen(posDistance);
71  if (inputPts->getSize() <= 1) {
72  computePointCurve(inputPts->getAt(0), *segGen);
73  } else {
74  if (bufParams.isSingleSided()) {
75  bool isRightSide = distance < 0.0;
76  computeSingleSidedBufferCurve(*inputPts, isRightSide, *segGen);
77  }
78  else {
79  computeLineBufferCurve(*inputPts, *segGen);
80  }
81  }
82 
83  segGen->getCoordinates(lineList);
84 }

87 void
88 OffsetCurveBuilder::computePointCurve(const Coordinate& pt,
89  OffsetSegmentGenerator& segGen)
90 {
91  switch (bufParams.getEndCapStyle()) {
92  case BufferParameters::CAP_ROUND:
93  segGen.createCircle(pt, distance);
94  break;
95  case BufferParameters::CAP_SQUARE:
96  segGen.createSquare(pt, distance);
97  break;
98  default:
99  // otherwise curve is empty (e.g. for a butt cap);
100  break;
101  }

104 /*public*/
105 void
106 OffsetCurveBuilder::getSingleSidedLineCurve(const CoordinateSequence* inputPts,
107  double distance, vector<CoordinateSequence*>& lineList, bool leftSide,
108  bool rightSide)
109 {
110  // A zero or negative width buffer of a line/point is empty.
111  if ( distance <= 0.0 ) return ;
112 
113  if ( inputPts->getSize() < 2 )
114  {
115  // No cap, so just return.
116  return ;
117  }
118 
119  double distTol = simplifyTolerance(distance);
120 
121  std::unique_ptr<OffsetSegmentGenerator> segGen = getSegGen(distance);
122 
123  if ( leftSide ) {
124  //--------- compute points for left side of line
125  // Simplify the appropriate side of the line before generating
126  std::unique_ptr<CoordinateSequence> simp1_ =
127  BufferInputLineSimplifier::simplify( *inputPts, distTol );
128  const CoordinateSequence& simp1 = *simp1_;
129 
130 
131  int n1 = static_cast<int>(simp1.size()) - 1;
132  if ( ! n1 )
133  throw util::IllegalArgumentException("Cannot get offset of single-vertex line");
134  segGen->initSideSegments(simp1[0], simp1[1], Position::LEFT);
135  segGen->addFirstSegment();
136  for (int i = 2; i <= n1; ++i) {
137  segGen->addNextSegment(simp1[i], true);
138  }
139  segGen->addLastSegment();
140  }
141 
142  if ( rightSide ) {
143 
144  //---------- compute points for right side of line
145  // Simplify the appropriate side of the line before generating
146  std::unique_ptr<CoordinateSequence> simp2_ =
147  BufferInputLineSimplifier::simplify( *inputPts, -distTol );
148  const CoordinateSequence& simp2 = *simp2_;
149 
150  int n2 = static_cast<int>(simp2.size()) - 1;
151  if ( ! n2 )
152  throw util::IllegalArgumentException("Cannot get offset of single-vertex line");
153  segGen->initSideSegments(simp2[n2], simp2[n2-1], Position::LEFT);
154  segGen->addFirstSegment();
155  for (int i = n2-2; i >= 0; --i) {
156  segGen->addNextSegment(simp2[i], true);
157  }
158  segGen->addLastSegment();
159  }
160 
161  segGen->getCoordinates(lineList);
162 }

165 void
166 OffsetCurveBuilder::getRingCurve(const CoordinateSequence *inputPts,
167  int side, double nDistance,
168  vector<CoordinateSequence*>& lineList)
169 {
170  distance = nDistance;
171 
172  // optimize creating ring for zero distance
173  if (distance == 0.0) {
174  lineList.push_back(inputPts->clone());
175  return;
176  }
177 
178  if (inputPts->getSize() <= 2)
179  {
180  getLineCurve(inputPts, distance, lineList);
181  return;
182  }
183 
184  std::unique_ptr<OffsetSegmentGenerator> segGen = getSegGen(std::abs(distance));
185  computeRingBufferCurve(*inputPts, side, *segGen);
186  segGen->getCoordinates(lineList);
187 }

196 /*private*/
197 void
198 OffsetCurveBuilder::computeLineBufferCurve(const CoordinateSequence& inputPts,
199  OffsetSegmentGenerator& segGen)
200 {
201  double distTol = simplifyTolerance(distance);
202 
203  //--------- compute points for left side of line
204  // Simplify the appropriate side of the line before generating
205  std::unique_ptr<CoordinateSequence> simp1_ =
206  BufferInputLineSimplifier::simplify(inputPts, distTol);
207  const CoordinateSequence& simp1 = *simp1_;
208 
209 
210  int n1 = static_cast<int>(simp1.size()) - 1;
211  segGen.initSideSegments(simp1[0], simp1[1], Position::LEFT);
212  for (int i = 2; i <= n1; ++i) {
213  segGen.addNextSegment(simp1[i], true);
214  }
215  segGen.addLastSegment();
216  // add line cap for end of line
217  segGen.addLineEndCap(simp1[n1-1], simp1[n1]);
218 
219  //---------- compute points for right side of line
220  // Simplify the appropriate side of the line before generating
221  std::unique_ptr<CoordinateSequence> simp2_ =
222  BufferInputLineSimplifier::simplify(inputPts, -distTol);
223  const CoordinateSequence& simp2 = *simp2_;
224 
225  int n2 = static_cast<int>(simp2.size()) - 1;
226  segGen.initSideSegments(simp2[n2], simp2[n2-1], Position::LEFT);
227  for (int i = n2-2; i >= 0; --i) {
228  segGen.addNextSegment(simp2[i], true);
229  }
230  segGen.addLastSegment();
231  // add line cap for start of line
232  segGen.addLineEndCap(simp2[1], simp2[0]);
233 
234  segGen.closeRing();
235 }

237 /*private*/
238 void
239 OffsetCurveBuilder::computeRingBufferCurve(const CoordinateSequence& inputPts,
240  int side, OffsetSegmentGenerator& segGen)
241 {
242  // simplify input line to improve performance
243  double distTol = simplifyTolerance(distance);
244  // ensure that correct side is simplified
245  if (side == Position::RIGHT)
246  distTol = -distTol;
247  std::unique_ptr<CoordinateSequence> simp_ =
248  BufferInputLineSimplifier::simplify(inputPts, distTol);
249  const CoordinateSequence& simp = *simp_;
250 
251  int n = static_cast<int>(simp.size())-1;
252  segGen.initSideSegments(simp[n-1], simp[0], side);
253  for (int i = 1; i <= n; i++) {
254  bool addStartPoint = i != 1;
255  segGen.addNextSegment(simp[i], addStartPoint);
256  }
257  segGen.closeRing();
258 }
