77 GeometryTransformer::transform(const Geometry* nInputGeom)
78 {
79  using geos::util::IllegalArgumentException;
80 
81 #if GEOS_DEBUG
82  std::cerr << "GeometryTransformer::transform(Geometry " << nInputGeom << ");" << std::endl;
83 #endif
84 
85  inputGeom = nInputGeom;
86  factory = inputGeom->getFactory();
87 
88  if ( const Point* p=dynamic_cast<const Point*>(inputGeom) )
89  return transformPoint(p, nullptr);
90  if ( const MultiPoint* mp=dynamic_cast<const MultiPoint*>(inputGeom) )
91  return transformMultiPoint(mp, nullptr);
92  if ( const LinearRing* lr=dynamic_cast<const LinearRing*>(inputGeom) )
93  return transformLinearRing(lr, nullptr);
94  if ( const LineString* ls=dynamic_cast<const LineString*>(inputGeom) )
95  return transformLineString(ls, nullptr);
96  if ( const MultiLineString* mls=dynamic_cast<const MultiLineString*>(inputGeom) )
97  return transformMultiLineString(mls, nullptr);
98  if ( const Polygon* p=dynamic_cast<const Polygon*>(inputGeom) )
99  return transformPolygon(p, nullptr);
100  if ( const MultiPolygon* mp=dynamic_cast<const MultiPolygon*>(inputGeom) )
101  return transformMultiPolygon(mp, nullptr);
102  if ( const GeometryCollection* gc=dynamic_cast<const GeometryCollection*>(inputGeom) )
103  return transformGeometryCollection(gc, nullptr);
104 
105  throw IllegalArgumentException("Unknown Geometry subtype.");
106 }

253 Geometry::Ptr
254 GeometryTransformer::transformPolygon(
255  const Polygon* geom,
256  const Geometry* parent)
257 {
258  ::geos::ignore_unused_variable_warning(parent);
259 
260 #if GEOS_DEBUG
261  std::cerr << "GeometryTransformer::transformPolygon(Polygon " << geom <<", Geometry " << parent << ");" << std::endl;
262 #endif
263 
264  bool isAllValidLinearRings = true;
265 
266  const LinearRing* lr = dynamic_cast<const LinearRing*>(
267  geom->getExteriorRing());
268  assert(lr);
269 
270  Geometry::Ptr shell = transformLinearRing(lr, geom);
271  if ( shell.get() == nullptr
272  || ! dynamic_cast<LinearRing*>(shell.get())
273  || shell->isEmpty() )
274  {
275  isAllValidLinearRings = false;
276  }
277 
278  vector<Geometry*>* holes = new vector<Geometry*>();
279  for (unsigned int i=0, n=static_cast<unsigned int>(geom->getNumInteriorRing()); i<n; i++)
280  {
281  const LinearRing* lr = dynamic_cast<const LinearRing*>(
282  geom->getInteriorRingN(i));
283  assert(lr);
284 
285  Geometry::Ptr hole(transformLinearRing(lr, geom));
286 
287  if ( hole.get() == nullptr || hole->isEmpty() ) {
288  continue;
289  }
290 
291  if ( ! dynamic_cast<LinearRing*>(hole.get()) )
292  {
293  if ( skipTransformedInvalidInteriorRings )
294  continue;
295  isAllValidLinearRings = false;
296  }
297 
298  holes->push_back(hole.release());
299  }
300 
301  if ( isAllValidLinearRings)
302  {
303  Geometry* sh = shell.release();
304  LinearRing* lr = dynamic_cast<LinearRing*>(sh);
305  assert(lr);
306  return Geometry::Ptr(factory->createPolygon(lr, holes));
307  }
308  else
309  {
310  // would like to use a manager constructor here
311  vector<Geometry*>* components = new vector<Geometry*>();
312  if ( shell.get() != nullptr ) {
313  components->push_back(shell.release());
314  }
315 
316  components->insert(components->end(),
317  holes->begin(), holes->end());
318 
319  delete holes; // :(
320 
321  return Geometry::Ptr(factory->buildGeometry(components));
322  }
