201 /*public*/
202 void
203 TopologyLocation::merge(const TopologyLocation &gl)
204 {
205  // if the src is an Area label & and the dest is not, increase the dest to be an Area
206  size_t sz=location.size();
207  size_t glsz=gl.location.size();
208  if (glsz>sz) {
209  location.resize(3);
210  location[Position::LEFT]=Location::UNDEF;
211  location[Position::RIGHT]=Location::UNDEF;
212  }
213  for (size_t i=0; i<sz; ++i) {
214  if (location[i]==Location::UNDEF && i<glsz)
215  location[i]=gl.location[i];
216  }
217 }
