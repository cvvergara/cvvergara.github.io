150 void
151 BufferOp::bufferReducedPrecision()
152 {
153 
154  // try and compute with decreasing precision,
155  // up to a min, to avoid gross results
156  // (not in JTS, see http://trac.osgeo.org/geos/ticket/605)
157 #define MIN_PRECISION_DIGITS 6
158  for (int precDigits=MAX_PRECISION_DIGITS; precDigits >= MIN_PRECISION_DIGITS; precDigits--)
159  {
160 #if GEOS_DEBUG
161  std::cerr<<"BufferOp::computeGeometry: trying with precDigits "<<precDigits<<std::endl;
162 #endif
163  try {
164  bufferReducedPrecision(precDigits);
165  } catch (const util::TopologyException& ex) {
166  saveException=ex;
167  // don't propagate the exception - it will be detected by fact that resultGeometry is null
168  }
169 
170  if (resultGeometry!=nullptr) {
171  // debug
172  //if ( saveException ) std::cerr<<saveException->toString()<<std::endl;
173  return;
174  }
175  }
176  // tried everything - have to bail
177  throw saveException;
