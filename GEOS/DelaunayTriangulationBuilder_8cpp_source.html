99 void
100 DelaunayTriangulationBuilder::create()
101 {
102  if(subdiv != nullptr || siteCoords == nullptr)
103  return;
104 
105  Envelope siteEnv;
106  siteCoords ->expandEnvelope(siteEnv);
107  IncrementalDelaunayTriangulator::VertexList* vertices = toVertices(*siteCoords);
108  subdiv = new quadedge::QuadEdgeSubdivision(siteEnv, tolerance);
109  IncrementalDelaunayTriangulator triangulator = IncrementalDelaunayTriangulator(subdiv);
110  triangulator.insertSites(*vertices);
111  delete vertices;
112 }
