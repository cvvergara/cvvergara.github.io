71 bool
72 QuadtreeNestedRingTester::isNonNested()
73 {
74  buildQuadtree();
75  for(size_t i=0, ni=rings.size(); i<ni; ++i)
76  {
77  const LinearRing *innerRing=rings[i];
78  const CoordinateSequence *innerRingPts=innerRing->getCoordinatesRO();
79  const Envelope *envi=innerRing->getEnvelopeInternal();
80 
81  vector<void*> results;
82  qt->query(envi, results);
83  for(size_t j=0, nj=results.size(); j<nj; ++j)
84  {
85  LinearRing *searchRing=(LinearRing*)results[j];
86  const CoordinateSequence *searchRingPts=searchRing->getCoordinatesRO();
87 
88  if (innerRing==searchRing) continue;
89 
90  const Envelope *e1=innerRing->getEnvelopeInternal();
91  const Envelope *e2=searchRing->getEnvelopeInternal();
92  if (!e1->intersects(e2)) continue;
93 
94  const Coordinate *innerRingPt=IsValidOp::findPtNotNode(innerRingPts,
95  searchRing, graph);
96 
97  // Unable to find a ring point not a node of the search ring
98  assert(innerRingPt!=nullptr);
99 
100  bool isInside=CGAlgorithms::isPointInRing(*innerRingPt,searchRingPts);
101  if (isInside) {
102  /*
103  * innerRingPt is const just because the input
104  * CoordinateSequence is const. If the input
105  * Polygon survives lifetime of this object
106  * we are safe.
107  */
108  nestedPt=const_cast<Coordinate *>(innerRingPt);
109  return false;
110  }
111  }
112  }
113  return true;
114 }
