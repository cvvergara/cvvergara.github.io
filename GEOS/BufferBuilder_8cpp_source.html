129 Geometry*
130 BufferBuilder::bufferLineSingleSided( const Geometry* g, double distance,
131  bool leftSide )
132 {
133  // Returns the line used to create a single-sided buffer.
134  // Input requirement: Must be a LineString.
135  const LineString* l = dynamic_cast< const LineString* >( g );
136  if ( !l ) throw util::IllegalArgumentException("BufferBuilder::bufferLineSingleSided only accept linestrings");
137 
138  // Nothing to do for a distance of zero
139  if ( distance == 0 ) return g->clone();
140 
141  // Get geometry factory and precision model.
142  const PrecisionModel* precisionModel = workingPrecisionModel;
143  if ( !precisionModel ) precisionModel = l->getPrecisionModel();
144 
145  assert( precisionModel );
146  assert( l );
147 
148  geomFact = l->getFactory();
149 
150  // First, generate the two-sided buffer using a butt-cap.
151  BufferParameters modParams = bufParams;
152  modParams.setEndCapStyle(BufferParameters::CAP_FLAT);
153  modParams.setSingleSided(false); // ignore parameter for areal-only geometries
154  std::unique_ptr<Geometry> buf;
155 
156  // This is a (temp?) hack to workaround the fact that
157  // BufferBuilder BufferParamaters are immutable after
158  // construction, while we want to force the end cap
159  // style to FLAT for single-sided buffering
160  {
161  BufferBuilder tmp(modParams);
162  buf.reset( tmp.buffer( l, distance ) );
163  }
164 
165  // Create MultiLineStrings from this polygon.
166  std::unique_ptr<Geometry> bufLineString ( buf->getBoundary() );
167 
168 #ifdef GEOS_DEBUG_SSB
169  std::cerr << "input|" << *l << std::endl;
170  std::cerr << "buffer|" << *bufLineString << std::endl;
171 #endif
172 
173  // Then, get the raw (i.e. unnoded) single sided offset curve.
174  OffsetCurveBuilder curveBuilder( precisionModel, modParams );
175  std::vector< CoordinateSequence* > lineList;
176 
177  {
178  std::unique_ptr< CoordinateSequence > coords(g->getCoordinates());
179  curveBuilder.getSingleSidedLineCurve(coords.get(), distance,
180  lineList, leftSide, !leftSide);
181  coords.reset();
182  }
183 
184  // Create a SegmentString from these coordinates.
185  SegmentString::NonConstVect curveList;
186  for ( unsigned int i = 0; i < lineList.size(); ++i )
187  {
188  CoordinateSequence* seq = lineList[i];
189 
190  // SegmentString takes ownership of CoordinateSequence
191  SegmentString* ss = new NodedSegmentString(seq, nullptr);
192  curveList.push_back( ss );
193  }
194  lineList.clear();
195 
196 
197  // Node these SegmentStrings.
198  Noder* noder = getNoder( precisionModel );
199  noder->computeNodes( &curveList );
200 
201  SegmentString::NonConstVect* nodedEdges = noder->getNodedSubstrings();
202 
203  // Create a geometry out of the noded substrings.
204  std::vector< Geometry* >* singleSidedNodedEdges =
205  new std::vector< Geometry * >();
206  singleSidedNodedEdges->reserve(nodedEdges->size());
207  for ( std::size_t i = 0, n = nodedEdges->size(); i < n; ++i )
208  {
209  SegmentString* ss = ( *nodedEdges )[i];
210 
211  Geometry* tmp = geomFact->createLineString(
212  ss->getCoordinates()->clone()
213  );
214  delete ss;
215 
216  singleSidedNodedEdges->push_back( tmp );
217  }
218 
219  delete nodedEdges;
220 
221  for (size_t i=0, n=curveList.size(); i<n; ++i) delete curveList[i];
222  curveList.clear();
223 
224  std::unique_ptr<Geometry> singleSided ( geomFact->createMultiLineString(
225  singleSidedNodedEdges ) );
226 
227 #ifdef GEOS_DEBUG_SSB
228  std::cerr << "edges|" << *singleSided << std::endl;
229 #endif
230 
231  // Use the boolean operation intersect to obtain the line segments lying
232  // on both the butt-cap buffer and this multi-line.
233  //Geometry* intersectedLines = singleSided->intersection( bufLineString );
234  // NOTE: we use Snapped overlay because the actual buffer boundary might
235  // diverge from original offset curves due to the addition of
236  // intersections with caps and joins curves
237  using geos::operation::overlay::snap::SnapOverlayOp;
238  std::unique_ptr<Geometry> intersectedLines = SnapOverlayOp::overlayOp(*singleSided, *bufLineString, OverlayOp::opINTERSECTION);
239 
240 #ifdef GEOS_DEBUG_SSB
241  std::cerr << "intersection" << "|" << *intersectedLines << std::endl;
242 #endif
243 
244  // Merge result lines together.
245  LineMerger lineMerge;
246  lineMerge.add( intersectedLines.get() );
247  std::unique_ptr< std::vector< LineString* > > mergedLines (
248  lineMerge.getMergedLineStrings() );
249 
250  // Convert the result into a std::vector< Geometry* >.
251  std::vector< Geometry* >* mergedLinesGeom = new std::vector< Geometry* >();
252  const Coordinate& startPoint = l->getCoordinatesRO()->front();
253  const Coordinate& endPoint = l->getCoordinatesRO()->back();
254  while( !mergedLines->empty() )
255  {
256  // Remove end points if they are a part of the original line to be
257  // buffered.
258  CoordinateSequence::Ptr coords(mergedLines->back()->getCoordinates());
259  if ( nullptr != coords.get() )
260  {
261  // Use 98% of the buffer width as the point-distance requirement - this
262  // is to ensure that the point that is "distance" +/- epsilon is not
263  // included.
264  //
265  // Let's try and estimate a more accurate bound instead of just assuming
266  // 98%. With 98%, the episilon grows as the buffer distance grows,
267  // so that at large distance, artifacts may skip through this filter
268  // Let the length of the line play a factor in the distance, which is still
269  // going to be bounded by 98%. Take 10% of the length of the line from the buffer distance
270  // to try and minimize any artifacts.
271  const double ptDistAllowance = std::max(distance - l->getLength()*0.1, distance * 0.98);
272  // Use 102% of the buffer width as the line-length requirement - this
273  // is to ensure that line segments that is length "distance" +/-
274  // epsilon is removed.
275  const double segLengthAllowance = 1.02 * distance;
276 
277  // Clean up the front of the list.
278  // Loop until the line's end is not inside the buffer width from
279  // the startPoint.
280  while ( coords->size() > 1 &&
281  coords->front().distance( startPoint ) < ptDistAllowance )
282  {
283  // Record the end segment length.
284  double segLength = coords->front().distance( ( *coords )[1] );
285  // Stop looping if there are no more points, or if the segment
286  // length is larger than the buffer width.
287  if ( coords->size() <= 1 || segLength > segLengthAllowance )
288  {
289  break;
290  }
291  // If the first point is less than buffer width away from the
292  // reference point, then delete the point.
293  coords->deleteAt( 0 );
294  }
295  while ( coords->size() > 1 &&
296  coords->front().distance( endPoint ) < ptDistAllowance )
297  {
298  double segLength = coords->front().distance( ( *coords )[1] );
299  if ( coords->size() <= 1 || segLength > segLengthAllowance )
300  {
301  break;
302  }
303  coords->deleteAt( 0 );
304  }
305 
306  // Clean up the back of the list.
307  while ( coords->size() > 1 &&
308  coords->back().distance( startPoint ) < ptDistAllowance )
309  {
310  double segLength = coords->back().distance(
311  ( *coords )[coords->size()-2] );
312  if ( coords->size() <= 1 || segLength > segLengthAllowance )
313  {
314  break;
315  }
316  coords->deleteAt( coords->size()-1 );
317  }
318  while ( coords->size() > 1 &&
319  coords->back().distance( endPoint ) < ptDistAllowance )
320  {
321  double segLength = coords->back().distance(
322  ( *coords )[coords->size()-2] );
323  if ( coords->size() <= 1 || segLength > segLengthAllowance )
324  {
325  break;
326  }
327  coords->deleteAt( coords->size()-1 );
328  }
329 
330  // Add the coordinates to the resultant line string.
331  if ( coords->size() > 1 )
332  {
333  mergedLinesGeom->push_back( geomFact->createLineString( coords.release() ) );
334  }
335  }
336 
337  geomFact->destroyGeometry( mergedLines->back() );
338  mergedLines->pop_back();
339  }
340 
341  // Clean up.
342  if ( noder != workingNoder ) delete noder;
343  buf.reset();
344  singleSided.reset();
345  intersectedLines.reset();
346 
347  if ( mergedLinesGeom->size() > 1 )
348  {
349  return geomFact->createMultiLineString( mergedLinesGeom );
350  }
351  else if ( mergedLinesGeom->size() == 1 )
352  {
353 
354  Geometry* single = (*mergedLinesGeom)[0];
355  delete mergedLinesGeom;
356  return single;
357  }
358  else
359  {
360  delete mergedLinesGeom;
361  return geomFact->createLineString();
362  }
363 }

367 BufferBuilder::buffer(const Geometry *g, double distance)
368  // throw(GEOSException *)
369 {
370  const PrecisionModel *precisionModel=workingPrecisionModel;
371  if (precisionModel==nullptr)
372  precisionModel=g->getPrecisionModel();
373 
374  assert(precisionModel);
375  assert(g);
376 
377  // factory must be the same as the one used by the input
378  geomFact=g->getFactory();
379 
380  { // This scope is here to force release of resources owned by
381  // OffsetCurveSetBuilder when we're doing with it
382 
383  OffsetCurveBuilder curveBuilder(precisionModel, bufParams);
384  OffsetCurveSetBuilder curveSetBuilder(*g, distance, curveBuilder);
385 
386  GEOS_CHECK_FOR_INTERRUPTS();
387 
388  std::vector<SegmentString*>& bufferSegStrList=curveSetBuilder.getCurves();
389 
390 #if GEOS_DEBUG
391  std::cerr << "OffsetCurveSetBuilder got " << bufferSegStrList.size()
392  << " curves" << std::endl;
393 #endif
394  // short-circuit test
395  if (bufferSegStrList.size()<=0) {
396  return createEmptyResultGeometry();
397  }
398 
399 #if GEOS_DEBUG
400  std::cerr<<"BufferBuilder::buffer computing NodedEdges"<<std::endl;
401 #endif
402 
403  computeNodedEdges(bufferSegStrList, precisionModel);
404 
405  GEOS_CHECK_FOR_INTERRUPTS();
406 
407  } // bufferSegStrList and contents are released here
408 
409 #if GEOS_DEBUG > 1
410  std::cerr << std::endl << edgeList << std::endl;
411 #endif
412 
413  Geometry* resultGeom=nullptr;
414  std::unique_ptr< std::vector<Geometry*> > resultPolyList;
415  std::vector<BufferSubgraph*> subgraphList;
416 
417  try {
418  PlanarGraph graph(OverlayNodeFactory::instance());
419  graph.addEdges(edgeList.getEdges());
420 
421  GEOS_CHECK_FOR_INTERRUPTS();
422 
423  createSubgraphs(&graph, subgraphList);
424 
425 #if GEOS_DEBUG
426  std::cerr<<"Created "<<subgraphList.size()<<" subgraphs"<<std::endl;
427 #if GEOS_DEBUG > 1
428  for (size_t i=0, n=subgraphList.size(); i<n; i++)
429  std::cerr << std::setprecision(10) << *(subgraphList[i]) << std::endl;
430 #endif
431 #endif
432 
433  GEOS_CHECK_FOR_INTERRUPTS();
434 
435  { // scope for earlier PolygonBuilder cleanupt
436  PolygonBuilder polyBuilder(geomFact);
437  buildSubgraphs(subgraphList, polyBuilder);
438 
439  resultPolyList.reset( polyBuilder.getPolygons() );
440  }
441 
442  // Get rid of the subgraphs, shouldn't be needed anymore
443  for (size_t i=0, n=subgraphList.size(); i<n; i++) delete subgraphList[i];
444  subgraphList.clear();
445 
446 #if GEOS_DEBUG
447  std::cerr << "PolygonBuilder got " << resultPolyList->size()
448  << " polygons" << std::endl;
449 #if GEOS_DEBUG > 1
450  for (size_t i=0, n=resultPolyList->size(); i<n; i++)
451  std::cerr << (*resultPolyList)[i]->toString() << std::endl;
452 #endif
453 #endif
454 
455  // just in case ...
456  if ( resultPolyList->empty() ) return createEmptyResultGeometry();
457 
458  // resultPolyList ownership transferred here
459  resultGeom=geomFact->buildGeometry(resultPolyList.release());
460 
461  } catch (const util::GEOSException& /* exc */) {
462 
463  // In case they're still around
464  for (size_t i=0, n=subgraphList.size(); i<n; i++)
465  delete subgraphList[i];
466  subgraphList.clear();
467 
468  throw;
469  }
470 
471  return resultGeom;
472 }

568  insertUniqueEdge(edge);
569  }

572

576 /*private*/
577 void
578 BufferBuilder::insertUniqueEdge(Edge *e)
579 {
580  //<FIX> MD 8 Oct 03 speed up identical edge lookup
581  // fast lookup
582  Edge *existingEdge = edgeList.findEqualEdge(e);
583  // If an identical edge already exists, simply update its label
584  if (existingEdge != nullptr) {
585  Label& existingLabel = existingEdge->getLabel();
586  Label labelToMerge = e->getLabel();
587 
588  // check if new edge is in reverse direction to existing edge
589  // if so, must flip the label before merging it
590  if (! existingEdge->isPointwiseEqual(e))
591  {
592  labelToMerge = e->getLabel();
593  labelToMerge.flip();
594  }
595 
596  existingLabel.merge(labelToMerge);
597 
598  // compute new depth delta of sum of edges
599  int mergeDelta = depthDelta(labelToMerge);
600  int existingDelta = existingEdge->getDepthDelta();
601  int newDelta = existingDelta + mergeDelta;
602  existingEdge->setDepthDelta(newDelta);
603 
604  // we have memory release responsibility
605  delete e;
606 
607  } else { // no matching existing edge was found
608 
609  // add this new edge to the list of edges in this graph
610  edgeList.add(e);
611 
612  e->setDepthDelta(depthDelta(e->getLabel()));
613  }
614 }

628  graph->getNodes(nodes);
