181 /*private*/
182 void
183 PolygonBuilder::buildMinimalEdgeRings(
184  vector<MaximalEdgeRing*> &maxEdgeRings,
185  vector<EdgeRing*> &newShellList, vector<EdgeRing*> &freeHoleList,
186  vector<MaximalEdgeRing*> &edgeRings)
187 {
188  for(size_t i=0, n=maxEdgeRings.size(); i<n; ++i)
189  {
190  MaximalEdgeRing *er = maxEdgeRings[i];
191 #if GEOS_DEBUG
192  cerr<<"buildMinimalEdgeRings: maxEdgeRing "<<i<<" has "<<er->getMaxNodeDegree()<<" maxNodeDegree"<<endl;
193 #endif
194  if (er->getMaxNodeDegree()>2)
195  {
196  er->linkDirectedEdgesForMinimalEdgeRings();
197  vector<MinimalEdgeRing*> minEdgeRings;
198  er->buildMinimalRings(minEdgeRings);
199  // at this point we can go ahead and attempt to place
200  // holes, if this EdgeRing is a polygon
201  EdgeRing *shell=findShell(&minEdgeRings);
202  if(shell != nullptr)
203  {
204  placePolygonHoles(shell, &minEdgeRings);
205  newShellList.push_back(shell);
206  }
207  else
208  {
209  freeHoleList.insert(freeHoleList.end(),
210  minEdgeRings.begin(),
211  minEdgeRings.end() );
212  }
213  delete er;
214  }
215  else
216  {
217  edgeRings.push_back(er);
218  }
219  }

224 PolygonBuilder::findShell(vector<MinimalEdgeRing*> *minEdgeRings)
225 {
226  int shellCount=0;

228 
229 #if GEOS_DEBUG

231 #endif
232 
233  for (size_t i=0, n=minEdgeRings->size(); i<n; ++i)
234  {
235  EdgeRing *er=(*minEdgeRings)[i];
236  if ( ! er->isHole() )
237  {
238  shell=er;
239  ++shellCount;
240  }
241  }
242 
243  if ( shellCount > 1 ) {
244  throw util::TopologyException("found two shells in MinimalEdgeRing list");
245  }
246 
247  return shell;

327 /*private*/
328 EdgeRing*
329 PolygonBuilder::findEdgeRingContaining(EdgeRing *testEr,
330  vector<EdgeRing*>& newShellList)
331 {
332  LinearRing *testRing=testEr->getLinearRing();
333  const Envelope *testEnv=testRing->getEnvelopeInternal();
334  const Coordinate& testPt=testRing->getCoordinateN(0);
335  EdgeRing *minShell=nullptr;
336  const Envelope *minEnv=nullptr;
337  for(size_t i=0, n=newShellList.size(); i<n; i++)
338  {
339  LinearRing *lr=nullptr;
340  EdgeRing *tryShell=newShellList[i];
341  LinearRing *tryRing=tryShell->getLinearRing();
342  const Envelope *tryEnv=tryRing->getEnvelopeInternal();
343  if (minShell!=nullptr) {
344  lr=minShell->getLinearRing();
345  minEnv=lr->getEnvelopeInternal();
346  }
347  bool isContained=false;
348  const CoordinateSequence *rcl = tryRing->getCoordinatesRO();
349  if (tryEnv->contains(testEnv)
350  && CGAlgorithms::isPointInRing(testPt,rcl))
351  isContained=true;
352  // check if this new containing ring is smaller than
353  // the current minimum ring
354  if (isContained) {
355  if (minShell==nullptr
356  || minEnv->contains(tryEnv)) {
357  minShell=tryShell;
358  }
359  }
360  }
361  return minShell;
362 }
