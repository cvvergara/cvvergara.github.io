237 /* private */
238 void
239 Polygonizer::polygonize()
240 {
241  // check if already computed
242  if (!polyList.empty()) return;
243 
244  polyList.clear();
245 
246  // if no geometries were supplied it's possible graph could be null
247  if (graph.empty()) return;
248 
249  dangles = graph.deleteDangles();
250  cutEdges = graph.deleteCutEdges();
251 
252  auto edgeRingList = graph.getEdgeRings();
253 
254 #if GEOS_DEBUG
255  cerr<<"Polygonizer::polygonize(): "<<edgeRingList.size()<<" edgeRings in graph"<<endl;
256 #endif
257 
258  auto validEdgeRingList = findValidRings(edgeRingList);
259 
260 #if GEOS_DEBUG
261  cerr<<" "<<validEdgeRingList.size()<<" valid"<<endl;
262  cerr<<" "<<invalidRingLines.size()<<" invalid"<<endl;
263 #endif
264 
265  findShellsAndHoles(validEdgeRingList);
266 
267 #if GEOS_DEBUG
268  cerr<<" "<<holeList.size()<<" holes"<<endl;
269  cerr<<" "<<shellList.size()<<" shells"<<endl;
270 #endif
271 
272  assignHolesToShells(holeList, shellList);
273 
274  for (const auto &er : shellList)
275  {
276  polyList.push_back(er->getPolygon());
277  }
278 }

281 vector<EdgeRing*>
282 Polygonizer::findValidRings(const vector<EdgeRing*>& edgeRingList) const
283 {
284  vector<EdgeRing*> validEdgeRingList;
285  for (const auto &er : edgeRingList)
286  {
287  if (er->isValid())
288  {
289  validEdgeRingList.push_back(er);
290  }
291  else
292  {
293  invalidRingLines.push_back(er->getLineString());
294  }
295  GEOS_CHECK_FOR_INTERRUPTS();
296  }
297  return validEdgeRingList;
