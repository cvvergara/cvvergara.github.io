99 bool
100 CGAlgorithms::isCCW(const CoordinateSequence* ring)
101 {
102  // # of points without closing endpoint
103  const std::size_t nPts=ring->getSize()-1;
104 
105  // sanity check
106  if (nPts < 3)
107  {
108  throw util::IllegalArgumentException("Ring has fewer than 3 points, so orientation cannot be determined");
109  }
110 
111  // find highest point
112  const Coordinate *hiPt=&ring->getAt(0);
113  int hiIndex=0;
114  for (std::size_t i=1; i<=nPts; ++i)
115  {
116  const Coordinate *p=&ring->getAt(i);
117  if (p->y > hiPt->y) {
118  hiPt = p;
119  hiIndex = static_cast<int>(i);
120  }
121  }
122 
123  // find distinct point before highest point
124  int iPrev = hiIndex;
125  do {
126  iPrev = iPrev - 1;
127  if (iPrev < 0)
128  iPrev = static_cast<int>(nPts);
129  } while (ring->getAt(iPrev)==*hiPt && iPrev!=hiIndex);
130 
131  // find distinct point after highest point
132  int iNext = hiIndex;
133  do {
134  iNext = (iNext + 1) % static_cast<int>(nPts);
135  } while (ring->getAt(iNext)==*hiPt && iNext != hiIndex);
136 
137  const Coordinate *prev=&ring->getAt(iPrev);
138  const Coordinate *next=&ring->getAt(iNext);
139 
140  /*
141  * This check catches cases where the ring contains an A-B-A
142  * configuration of points.
143  * This can happen if the ring does not contain 3 distinct points
144  * (including the case where the input array has fewer than 4 elements),
145  * or it contains coincident line segments.
146  */
147  if ( prev->equals2D(*hiPt) || next->equals2D(*hiPt) ||
148  prev->equals2D(*next) )
149  {
150  return false;
151  // MD - don't bother throwing exception,
152  // since this isn't a complete check for ring validity
153  //throw IllegalArgumentException("degenerate ring (does not contain 3 distinct points)");
154  }
155 
156  int disc = computeOrientation(*prev, *hiPt, *next);
157 
168  bool isCCW=false;
169 
170  if (disc == 0) {
171  // poly is CCW if prev x is right of next x
172  isCCW = (prev->x > next->x);
173  } else {
174  // if area is positive, points are ordered CCW
175  isCCW = (disc > 0);
176  }
177 
178  return isCCW;
179 }

189 /*public static*/
190 double
191 CGAlgorithms::distancePointLine(const Coordinate& p, const Coordinate& A,
192  const Coordinate& B)
193 {
194  //if start==end, then use pt distance
195  if (A==B) return p.distance(A);
196 
197  // otherwise use comp.graphics.algorithms Frequently Asked Questions method
198  /*(1) AC dot AB
199  r = ---------
200  ||AB||^2
201  r has the following meaning:
202  r=0 P = A
203  r=1 P = B
204  r<0 P is on the backward extension of AB
205  r>1 P is on the forward extension of AB
206  0<r<1 P is interior to AB
207  */
208  double r=((p.x-A.x)*(B.x-A.x)+(p.y-A.y)*(B.y-A.y))/
209  ((B.x-A.x)*(B.x-A.x)+(B.y-A.y)*(B.y-A.y));
210  if (r<=0.0) return p.distance(A);
211  if (r>=1.0) return p.distance(B);
212  /*(2)
213  (Ay-Cy)(Bx-Ax)-(Ax-Cx)(By-Ay)
214  s = -----------------------------
215  L^2
216 
217  Then the distance from C to P = |s|*L.
218  */
219  double s=((A.y-p.y)*(B.x-A.x)-(A.x-p.x)*(B.y-A.y))/
220  ((B.x-A.x)*(B.x-A.x)+(B.y-A.y)*(B.y-A.y));
221  return fabs(s)*sqrt(((B.x-A.x)*(B.x-A.x)+(B.y-A.y)*(B.y-A.y)));
222 }

226 CGAlgorithms::distancePointLinePerpendicular(const Coordinate& p,const Coordinate& A,const Coordinate& B)
227 {
228  // use comp.graphics.algorithms Frequently Asked Questions method
229  /*(2)
230  (Ay-Cy)(Bx-Ax)-(Ax-Cx)(By-Ay)

234  Then the distance from C to P = |s|*L.

243 /*public static*/
244 double
245 CGAlgorithms::distanceLineLine(const Coordinate& A, const Coordinate& B,
246  const Coordinate& C, const Coordinate& D)
247 {
248  // check for zero-length segments
249  if (A==B) return distancePointLine(A,C,D);
250  if (C==D) return distancePointLine(D,A,B);
251 
252  // AB and CD are line segments
253  /* from comp.graphics.algo
254 
255  Solving the above for r and s yields
256  (Ay-Cy)(Dx-Cx)-(Ax-Cx)(Dy-Cy)
257  r = ----------------------------- (eqn 1)
258  (Bx-Ax)(Dy-Cy)-(By-Ay)(Dx-Cx)
259 
260  (Ay-Cy)(Bx-Ax)-(Ax-Cx)(By-Ay)
261  s = ----------------------------- (eqn 2)
262  (Bx-Ax)(Dy-Cy)-(By-Ay)(Dx-Cx)
263  Let P be the position vector of the intersection point, then
264  P=A+r(B-A) or
265  Px=Ax+r(Bx-Ax)
266  Py=Ay+r(By-Ay)
267  By examining the values of r & s, you can also determine some other
268 limiting conditions:
269  If 0<=r<=1 & 0<=s<=1, intersection exists
270  r<0 or r>1 or s<0 or s>1 line segments do not intersect
271  If the denominator in eqn 1 is zero, AB & CD are parallel
272  If the numerator in eqn 1 is also zero, AB & CD are collinear.
273 
274  */
275  double r_top=(A.y-C.y)*(D.x-C.x)-(A.x-C.x)*(D.y-C.y);
276  double r_bot=(B.x-A.x)*(D.y-C.y)-(B.y-A.y)*(D.x-C.x);
277  double s_top=(A.y-C.y)*(B.x-A.x)-(A.x-C.x)*(B.y-A.y);
278  double s_bot=(B.x-A.x)*(D.y-C.y)-(B.y-A.y)*(D.x-C.x);
279  if ((r_bot==0)||(s_bot==0)) {
280  return std::min(distancePointLine(A,C,D),
281  std::min(distancePointLine(B,C,D),
282  std::min(distancePointLine(C,A,B), distancePointLine(D,A,B))));
283  }
284  double s=s_top/s_bot;
285  double r=r_top/r_bot;
286  if ((r<0)||( r>1)||(s<0)||(s>1)) {
287  //no intersection
288  return std::min(distancePointLine(A,C,D),
289  std::min(distancePointLine(B,C,D),
290  std::min(distancePointLine(C,A,B), distancePointLine(D,A,B))));
291  }
292  return 0.0; //intersection exists
293 }

295 /*public static*/
296 double
297 CGAlgorithms::signedArea(const CoordinateSequence* ring)
298 {
299  size_t npts=ring->getSize();
300 
301  if (npts<3) return 0.0;
302 
303  Coordinate pp;
304  Coordinate cp = ring->getAt(0);
305  Coordinate np = ring->getAt(1);
306  double x0 = cp.x;
307  np.x -= x0;
308  double sum=0.0;
309  for (size_t i=1; i<npts; ++i)
310  {
311  pp.y = cp.y;
312  cp.x = np.x;
313  cp.y = np.y;
314  ring->getAt(i, np);
315  np.x -= x0;
316  sum += cp.x * (np.y - pp.y);
317  }
318  return -sum/2.0;
319 }

322 double
323 CGAlgorithms::length(const CoordinateSequence* pts)
324 {
325  // optimized for processing CoordinateSequences
326 
327  size_t npts=pts->getSize();
328  if (npts <= 1) return 0.0;
329 
330  double len = 0.0;
331 
332  const Coordinate& p0 = pts->getAt(0);
333  double x0 = p0.x;
334  double y0 = p0.y;
335 
336  for(size_t i = 1; i < npts; ++i)
337  {
338  const Coordinate& p = pts->getAt(i);
339  double x1 = p.x;
340  double y1 = p.y;
341  double dx = x1 - x0;
342  double dy = y1 - y0;
343 
344  len += sqrt(dx * dx + dy * dy);
345 
346  x0 = x1;
347  y0 = y1;
348  }
349 
350  return len;
351 }
