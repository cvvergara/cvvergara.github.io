47 double
48 PrecisionModel::makePrecise(double val) const
49 {
50 #if GEOS_DEBUG
51  cerr<<"PrecisionModel["<<this<<"]::makePrecise called"<<endl;
52 #endif
53 
54  if (modelType==FLOATING_SINGLE) {
55  float floatSingleVal = static_cast<float>(val);
56  return static_cast<double>(floatSingleVal);
57  }
58  if (modelType == FIXED) {
59  // Use whatever happens to be the default rounding method
60  const double ret = util::round(val*scale)/scale;
61  return ret;
62  }
63  // modelType == FLOATING - no rounding necessary
64  return val;
65 }

154 /*public*/
155 int
156 PrecisionModel::getMaximumSignificantDigits() const
157 {
158  int maxSigDigits = 16;
159  if (modelType == FLOATING) {
160  maxSigDigits = 16;
161  } else if (modelType == FLOATING_SINGLE) {
162  maxSigDigits = 6;
163  } else if (modelType == FIXED) {
164 
165  double dgtsd = std::log(getScale()) / std::log(double(10.0));
166  const int dgts = static_cast<int>(
167  dgtsd > 0 ? std::ceil(dgtsd)
168  : std::floor(dgtsd)
169  );
170  maxSigDigits = dgts;
171  }
172  return maxSigDigits;
173 }
