76 WKTWriter::toLineString(const CoordinateSequence& seq)
77 {
78  stringstream buf(ios_base::in|ios_base::out);
79  buf << "LINESTRING ";
80  unsigned int npts = static_cast<unsigned int>(seq.getSize());
81  if ( npts == 0 )
82  {
83  buf << "EMPTY";
84  }
85  else
86  {
87  buf << "(";
88  for (unsigned int i=0; i<npts; ++i)
89  {
90  if (i) buf << ", ";
91  buf << seq.getX(i) << " " << seq.getY(i);

95  }
96  buf << ")";
97  }
98 
99  return buf.str();

180 void
181 WKTWriter::appendGeometryTaggedText(const Geometry *geometry, int level,
182  Writer *writer)
183 {
184  outputDimension = std::min( defaultOutputDimension,
185  geometry->getCoordinateDimension() );
186 
187  indent(level, writer);
188  if ( const Point* point = dynamic_cast<const Point*>(geometry) )
189  {
190  appendPointTaggedText(point->getCoordinate(),level,writer);
191  }
192  else if ( const LinearRing* lr =
193  dynamic_cast<const LinearRing*>(geometry) )
194  {
195  appendLinearRingTaggedText(lr, level, writer);
196  }
197  else if ( const LineString* ls =
198  dynamic_cast<const LineString*>(geometry) )
199  {
200  appendLineStringTaggedText(ls, level, writer);
201  }
202  else if ( const Polygon* x =
203  dynamic_cast<const Polygon*>(geometry) )
204  {
205  appendPolygonTaggedText(x, level, writer);
206  }
207  else if ( const MultiPoint* x =
208  dynamic_cast<const MultiPoint*>(geometry) )
209  {
210  appendMultiPointTaggedText(x, level, writer);
211  }
212  else if ( const MultiLineString* x =
213  dynamic_cast<const MultiLineString*>(geometry) )
214  {
215  appendMultiLineStringTaggedText(x, level, writer);
216  }
217  else if ( const MultiPolygon* x =
218  dynamic_cast<const MultiPolygon*>(geometry) )
219  {
220  appendMultiPolygonTaggedText(x, level, writer);
221  }
222  else if ( const GeometryCollection* x =
223  dynamic_cast<const GeometryCollection*>(geometry) )
224  {
225  appendGeometryCollectionTaggedText(x, level, writer);
226  }
227  else
228  {
229  assert(0); // Unsupported Geometry implementation
230  }
231 }

353  writer->write("EMPTY");
